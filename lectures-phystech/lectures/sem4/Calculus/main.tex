\section{Пространство Лебега}
\subsection{Пространства \(L_p\)}
\begin{definition}
    Пусть \(E \subset \R^m\). Будем говорить, что \(f: E \ra \Cm\) измерима (интегрируема) по Лебегу, если \(\Re f, \Im f\) измеримы (интегрируемы) по Лебегу.
\end{definition}

\begin{definition}
    В случае интегрируемости положим \(\int_E f = \int_E \Re f + i \int_E \Im f\). Полученный интеграл линеен и аддитивен по множествам
\end{definition}

\begin{note}
    \[\left\|\int_E f \right\| \le \int_E |f|\]
\end{note}
\begin{proof}
    \[\int_E f = \left\|\int_E f\right\| e^{i \theta} \Ra \int_E fe^{-i \theta} = \left\|\int_E f\right\| = \int_E \Re(e^{-i\theta} f) \le \int_E |e^{i\theta} f| = \int_E |f|\]
\end{proof}

\begin{definition}
    Пусть \(1 \le p \le < \infty\) и \(E\) измеримо. Определим:
    \[L_p(E) = \{f: E \ra \Cm | \int_E |f|^p < \infty\}\]
    В таком случае положим \(\|f\|_p = \left( \int_E |f|^p \right)^{\frac{1}{p}}\)
\end{definition}

Пусть \(f, g \in L_p(E)\). Если \(\lambda \in \Cm\), то \(\lambda f \in L_p(E)\) и ввиду \(|f + g|^p \le (2\max\{|f|, |g|\})^p \le 2^p(|f|^p + |g|^p)\) выполнено \(f + g \in L_p\). Получили, что \(L_p\) является линейным пространством относительно \(+, \lambda\cdot\).



\begin{lemma}
    Пусть \(a, b \ge 0\). Если \(1 < p < \infty, \frac{1}{p} + \frac{1}{q} = 1\), то \(ab \le \frac{a^p}{p} + \frac{b^q}{q}\), причем равенство достигается тогда и только тогда, когда \(a^p = b^q\)
\end{lemma}
\begin{proof}
    Можно считать, что \(ab > 0\). Ввиду выпуклости экспоненты, имеем:
    \[e^{\frac{x}{p} + \frac{y}{q}} \le \frac{1}{p}e^x + \frac{1}{q}e^y\]
    Положим \(x = p\ln a, y = q\ln b\) и получаем желаемое.
\end{proof}

\begin{theorem}[Гельдер]
    Пусть \(1 < p < \infty, \frac{1}{p} + \frac{1}{q} = 1\). Если \(f \in L_p(E), g \in L_q(E)\), то \(fg \in L_1(E)\) и \(\|fg\|_1 \le \|f\|_p\|g\|_q\)
\end{theorem}
\begin{proof}
    Если \(\|f\|_p = 0 \Ra f = 0\) п.в. \(\Ra fg = 0\) п.в. \(\Ra \) утверждение доказано. Аналогично и для случая \(\|g\|_q = 0\). В противном случае, получаем:
    \[\|f\|_p\|g\|_q > 0\]
    По предыдущей лемме, имеем:
    \[\frac{|f(x)|}{\|f\|_p}\frac{|g(x)|}{\|g\|_q} \le \frac{1}{p}\left( \frac{|f(x)|}{\|f\|_p} \right)^p + \frac{1}{q}\left( \frac{|g(x)|}{\|g\|_q} \right)^q\]
    Проинтегрировав, получим:
    \[\frac{1}{\|f\|_p\|g\|_q}\int_E |fg| \le \frac{1}{p}\left( \int_E \frac{|f|^p}{\|f\|_p^p} \right) + \frac{1}{q}\left( \int_E \frac{|g|^q}{\|g\|_q^q} \right) = 1\]
\end{proof}

\begin{note}
    В неравенстве гельдера равенство имеет место тогда и только тогда, когда \(|f|^p = c|g|^q\) п.в. на \(E\) для некоторого \(c > 0\).
\end{note}

\begin{theorem}[Минковский]
    Пусть \(1 \le p < \infty\). Если \(f, g \in L_p(E)\), то \(f + g \in L_p(E)\) и \(\|f + g\|_p \le \|f\|_p + \|g\|_p\).
\end{theorem}
\begin{proof}
    При \(p = 1\) применим \(|f + g| \le |f| + |g|\). Далее при \(p \ge 1\): Применим неравенство Гельдера для \(p, q = \frac{p}{p - 1} \Ra \frac{1}{p} + \frac{1}{q} = 1\). 
    \[\|f + g\|_p^p = \int_E |f + g|^p \le \int_E |f||f + g|^{p - 1} + \int_E |g||f + g|^{p - 1} \le\]
    \[\le \left( \int_E |f|^p \right)^{\frac{1}{p}}\left( \int_E |f + g|^p \right)^{\frac{1}{q}} + \left( \int_E |g|^p \right)^{\frac{1}{p}}\left( \int_E |f + g|^p \right)^{\frac{1}{q}} = \]
    \[ = \|f + g\|_p^{p - 1}(\|f\|_p + \|g\|_p)\]
\end{proof}

\begin{note}
    Равенство в теореме Минковского выполняется тогда и только тогда, когда \(f = cg\) п.в. для некоторой \(c > 0\)
\end{note}

\begin{definition}
    На \(L_p\) введем отношение \(\sim\). Будем говорить, что \(f \sim g \Lra f = g\) п.в. на \(E\).
\end{definition}

\begin{note}
    \(\sim\) --- отношение эквивалентности на \(L_p\), согласованное с операциями сложения и умножения на скаляр. Факторпространство \(L_p(E)/_\sim\) будем также обозначать \(L_p(E)\)
\end{note}

\begin{corollary}
    Пространство \(L_p\) относительно нормы \(\|\cdot\|_p\) является нормированным линейным пространством
\end{corollary}
\begin{proof}\indent
    \begin{enumerate}
        \item \(\|f\|_p \ge 0, \|f\|_p = 0 \Lra f \sim 0\)
        \item \(\|\lambda f\|_p = \lambda \|f\|_p\)
        \item \(\|f + g\|_p \le \|f\|_p + \|g\|_p\)
    \end{enumerate}
\end{proof}

\begin{problem}
    Если \(\mu(E) < \infty, 1 \le p < q < \infty\), то \(L_q(E) \subsetneq L_p(E)\)
\end{problem}

\begin{reminder}

\end{reminder}

\end{theorem}
\begin{proof}

\end{proof}

\end{reminder}

\begin{lemma}
    Пусть \(f \in L_p, \epsilon > 0\). Тогда \(\exists\) простая финитная функция \(\phi\) такая, что \(\|f - \phi\|_p < \epsilon\).
\end{lemma}
\begin{proof}
    Можно считать, что \(f\) вещественнозначная (иначе приближаем \(\Re f, \Im f\)) отдельно. Т.к. \(|f - I_{B_k(0)}|^p \le |f|^p\). Тогда по Теореме Лебега о мажорируемой сходимости, \(\|f - fI_{B_k(0)}\|_p \ra 0, k \ra \infty\). Заменяя функцию \(f\) на \(fI_{B_k(0)}\) для достаточно большого \(k\), заключаем, что она финитная. Пусть сначала \(f \ge 0\). По теореме о приближении, найдется последовательность \(\{\phi_k\}\) --- простых функций, т.ч. \(0 \le \phi_1 \le \dots \), \(\phi_k \ra f\). Т.к. \(|f - \phi_k|^p \le |f|^p\). По теореме Лебега о мажорируемой сходимости \(f - \phi_k \ra 0, k \ra \infty\), причем все \(\phi_k\) финитны, т.к. \(0 \le \phi_k \le f\) на \(E\). Пусть \(f\) теперь произвольного знака \(\Ra f = f^+ - f^-\). Тогда по доказанному найдутся простые финитные функции \(\phi^+, \phi^-\), такие, что \(\|f^+ - \phi^+\|_p < \frac{\epsilon}{2}, \|f^- - \phi^-\|_p < \frac{\epsilon}{2}\). Положим \(\phi = \phi^+ - \phi^-\). Тогда \(\phi\) --- простая финитная функция и по неравенству треугольника:
    \[\|f - \phi\|_p \le \|f^+ - \phi^+\| + \|f^- - \phi^-\| < \epsilon\]
\end{proof}

\begin{theorem}
    Пусть \(f \in L_p, \epsilon > 0\). Тогда \(\exists g \in C_c^\infty(\R^m)\), т.ч. \(\|f - g\|_p < \epsilon\)
\end{theorem}
\begin{proof}
    По предыдущей лемме, любую функцию можно приблизить финитной простой функцией. Всякая простая функция есть линейная комбинация индикаторов. Из этого заключаем, что достаточно доказать теорему для случая \(f = I_A\), где \(A\) --- ограниченное измеримое множество. По свойству регулярности меры Лебега, \(\exists G, H\) --- открытые, такие, что \(G \supset A, H \supset A^c\) и \(\mu(G \setminus A) < \frac{\epsilon}{2}, \mu(H \setminus A^c) < \frac{\epsilon}{2}\). Положим \(k = H^c\) --- замкнутое и ограниченное (т.к. лежит в \(A\)), т.е. компакт, лежащий в \(A\). \(\mu(G \setminus A) \le \mu(G \setminus A) + \mu(H \setminus A^c) < \epsilon\). По теореме о гладком разбиении единицы, \(\exists g \in C^\infty(\R^m)\) с носителем в \(G\), такая, что \(0 \le g \le 1\) и \(g|_K = 1\). Поэтому \(\|I_A - g|_E\|_p^p = \int_E |I_A - g|^p \le \int_{(G \cap E) \setminus K} |I_A - g|^p \le \mu(G \setminus K) < \epsilon\)
\end{proof}

\begin{theorem}[Кантора]
    Пусть \(f: U \ra \Cm\), где \(U \subset \R^n\) --- открыто. Тогда, если \(K\) --- компакт и \(f\) непрерывна на нем, то \(f\) равномерно непрерывна на нем, т.е.
    \[\forall \epsilon > 0 \exists \delta > 0 \forall x \in K \forall h \in \R^n (h < \delta \Ra |f(x) - f(x - h)| < \delta)\]
\end{theorem}
\begin{proof}
    От противного. Тогда \(\exists \{x_k\} \subset K \exists \{h_k\} \subset \R^m |f(x_k) - f(x_k - h_k)| \ge \epsilon_0, h_k \ra 0\). Т.к. \(K\) --- компакт, то \(\exists x_{n_k} \ra x_0 \in K\) и \(x_{n_k} + h_{n_k} \ra x_0\). Тогда по непрерывности, \(f(x_{n_k}) \ra f(x_0), f(x_{n_k} - h_{n_k}) \ra f(x_0) \Ra |f(x_{n_k}) - f(x_{n_k} - h_{n_k})| \ra 0\)
\end{proof}

\begin{definition}
    Пусть \(f: \R^m \ra \Cm, h \in \R^m\). Функция \(f_h: \R^m \ra \Cm, f_h(x) = f(x - h)\) называется сдвигом функции \(f\) на \(h\).
\end{definition}

\begin{note}
    Функции \(f, f_h\) одновременно лежат в \(L_p(\R^m)\) и \(\|f\|_p = \|f_h\|_p\)
\end{note}

\begin{theorem}[Непрерывность сдвига]
    Если \(f \in L_p(\R^m)\), то \(\|f_h - f\|_p \ra 0\) при \(h \ra 0\).
\end{theorem}
\begin{proof}
    Зафиксируем \(\epsilon > 0\). \(\exists g \in C_c(\R^m)\) (можно считать, что \(g = 0\) вне \(B_r(0)\)), такая, что \(\|f - g\|_p < \frac{\epsilon}{3}\). По неравенству треугольника:
    \[\|f_h - f\|_p \le \|f_h - g_h\|_p + \|g_h - g\|_p + \|g - f\|_p < \frac{2}{3}\epsilon + \|g_h - g\|\]
    Т.к. \(g\) непрерывна на \(\overline{B_r(0)}\), то \(\exists \delta \in (0, 1] \forall h \in \R^m (|h| < \delta \Ra |g(x) - g(x - h)| < \frac{\epsilon}{3M})\), где \(M^p = \mu()B_{r + 1}(0)\). Поэтому:
    \[\|g_h - g\|^p_p = \int |g(x - h) - g(x)|^pdx \le \left( \frac{\epsilon}{3M} \right)^p\mu(B_{r + 1}) = \left( \frac{\epsilon}{3} \right)^p\]
    Следовательно, при \(|h| < \delta\), выполнено \(\|f_h - f\|_p < \epsilon\)
\end{proof}

\begin{corollary}[Непрерывность сдвига для периодических функций]
    Пусть \(f\) --- \(2\pi\)-периодичная измеримая на \(\R\) функция. Если \(\int_{[-\pi, \pi]} |f(x)|^p dx < \infty\), то 
    \[\int_{[-\pi, \pi]} |f(x - h) - f(x)|^pdx \ra 0, h \ra 0\]
\end{corollary}
\begin{proof}
    Пусть \(g = \left\{\begin{array}{l}
        f, x \in (-2\pi, 2\pi) \\
        0, \text{иначе}
    \end{array}\right.\). Тогда \(g \in L_p(\R)\). Также, при \(x \in (-\pi, \pi)\) и \(|h| < \pi\) имеем: \(|g(x - h) - g(x)| = |f(x - h) - f(x)|\), а значит,
    \[\int_{[-\pi, \pi]} |f(x - h) - f(x)|^pdx = \int_{[-\pi, \pi]} |g(x - h) - g(x)|^pdx \le \|g_h - g\|_p \ra 0, h \ra 0\]
\end{proof}

\begin{theorem}[Римана об осцилляции]
    Пусть \(I\) --- промежуток в \(\R\), \(f \in L_1(I)\). Тогда 
    \[\int_I f(x)e^{i\lambda x}dx \ra 0, \lambda \ra \pm\infty\]
    В частности, 
    \[\int_I f(x)\cos\lambda xdx \ra 0, \int_I f(x)\sin \lambda xdx \ra 0, \lambda \ra \pm\infty\]
\end{theorem}
\begin{proof}
    Рассмотрим случай \(I = \R\). Сделаем в интеграле замену \(x = t - \frac{\pi}{\lambda}\), тогда:
    \[\int_I f(x)e^{i\lambda x}dx = \int_I f\left( t - \frac{\pi}{\lambda} \right)e^{i\lambda \left( t - \frac{\pi}{\lambda} \right)}dx = -\int_I f\left( t - \frac{\pi}{\lambda} \right)e^{i\lambda t}dx\]
    Следовательно,
    \[\int_I f(x)e^{i\lambda x}dx = \frac{1}{2}\left(  \int_I f(x)e^{i\lambda x}dx -\int_I f\left( x - \frac{\pi}{\lambda} \right)e^{i\lambda x}dx\right)= \frac{1}{2}\left(  \int_I \left( f(x) - f\left( x - \frac{\pi}{\lambda} \right) \right)e^{i\lambda x}dx\right)\]
    Следовательно, 
    \[\left|\int_I f(x)e^{i\lambda x}dx\right| \le \frac{1}{2}\left(  \int_I \left| f(x) - f\left( x - \frac{\pi}{\lambda} \right) \right|dx\right) = \|f_h - f\|_1 \ra 0\]
\end{proof}

\subsection{Свертка и аппроксимация функции}
\begin{definition}
    Пусть \(f, g\) --- измеримы в \(\R^m\), функция \(f * g\) определяемая по формуле
    \[(f * g)(x) = \int_{\R^m} f(x - t)g(t)dt\]
    называется сверткой функций \(f, g\).
\end{definition}

\begin{note}
    Покажем измеримость функции \(f(x - t)g(t)\) в \(\R^{2m}\). Т.к. \(g(t)\) измерима в \(\R^{2m}\), достаточно показать измеримость \(f(x - t)\). Положим \(E_a = \{x \in \R^m: f(x) < a\}\) и определим оператор \(L: \R^{2m} \ra \R^{2m}, L(x, t) = (x - t, t)\). Тогда \(\{(x - t): f(x - t) < a\} = \{(x, t): x - t \in E_a\} = L^{-1}(E_a \times \R^m)\) --- измеримо как образ измеримого при диффеоморфизме \(L^{-1}\)
\end{note}

\begin{lemma}
    Пусть \(f, g \in L_1(\R^m)\), тогда \(f * g\) определена почти всюду. Более того, \(f * g \in L_1(\R^m)\) и \(\|f * g\|_1 \le \|f\|_1\|g\|_1\)
\end{lemma}
\begin{proof}
    Определим \(H(x) = \int_{\R^m}|f(x - t)||g(t)|dt\). По теореме Тонелли:
    \[\int_{\R^m}H(x)dx = \int_{\R^m}\left( \int_{\R^m}|f(x - t)| dx\right)|g(t)|dt = \int_{\R^m}\left( \int_{\R^m}|f(y)| dy\right)|g(t)|dt = \|f\|_1\|g\|_1\]
    Получаем:
    \[\left|\int_{\R^m}f(x - t)g(t)\right| \le \int_{\R^m}H(x)dx = \|f\|_1\|g\|_1\]
    Тогда \(|f(x - t)g(t)|\) конечна почти всюду \(\Ra (f * g)(x)\) определена почти всюду.
\end{proof}

\begin{theorem}
    \begin{enumerate}
        \item Если \(f \in L_p(\R^m), g \in L_q(\R^m), \frac{1}{p} + \frac{1}{q} = 1\), то \((f * g)\) существует, равномерно непрерывна на \(\R^m\) и \(|(f * g)(x)| \le \|f\|_p\|g\|_q\)
        \item Если \(f \in L_p(\R^m), g\) измерима и ограничена в \(\R^m\), то \((f * g)\) существует, равномерно непрерывна на \(\R^m\) и \(|(f * g)(x)| \le \|f\|_p\|g\|_\infty\), где \(\|g\|_\infty = \sup_{\R^m} |g|\)
    \end{enumerate}
\end{theorem}
\begin{proof}\indent
    \begin{enumerate}
        \item \[\int_{\R^m} |f(x - t)g(t)|dt \le \left( \int_{\R^m}|f(x - t)|^p dt\right)\frac{1}{p} \left( \int_{\R^m}|g(t)|^q dt\right)\frac{1}{q}= \|f\|_p\|g\|_q < \infty\]
        Тогда \(f * g\) определена всюду на \(\R^m\), и справедлива оценка из условия. Докажем равномерную непрерывность:
        \[|(f * g)(x - h) - (f * g)(x)| = \left| \int_{\R^m}(f(x - h - t) - f(x - t))g(t)dt \right| = \]
        \[= |((f_h - f) * g)(x)| \le \|f_h - f\|_p\|g\|_q \ra 0, h \ra 0\]
        Последнее стремление равномерное, т.е. не зависит от \(x\), поэтому утверждение доказано
        \item \[\int_{\R^m} |f(x - t)g(t)|dt \le \|g\|_\infty \|f\|_1 < \infty\]
        Равномерное стремление доказывается аналогично пункту 1.
    \end{enumerate}
\end{proof}

\begin{proposition}
    Пусть \(f, g, h \in L_1(\R^m)\), тогда:
    \begin{enumerate}
        \item \((f * g) = (g * f)\)
        \item \((f * g) * h = f * (g * h)\)
    \end{enumerate}
\end{proposition}
\begin{proof}\indent
    \begin{enumerate}
        \item \[(f * g) = \int_{\R^m} f(x - t)g(t) dt = \left\{\begin{array}{c}
            x - t = y \\
            |J| = 1
        \end{array}\right\} = \int_{\R^m} f(y)g(x - y)dy = (g * f)\]
        \item Следует из теоремы Фубини
    \end{enumerate}
\end{proof}

\begin{definition}
    Последовательность \(\{K_n\}_{n = 1}^\infty\) интегрируемых на \(\R^m\) функций называется аппроксимацией единицей (\(\delta\)-образным семейством), если 
    \begin{enumerate}
        \item \(K_n \ge 0 \forall n\)
        \item \(\int_{\R^m} K_ndx = 1\)
        \item \(\lim_{n \ra \infty} \int_{|x| \ge \delta} K_n(x)dx \ra 0\) для всех \(\delta > 0\).
    \end{enumerate}
\end{definition}

\begin{example}
    Рассмотрим \(\phi \in L_1(\R), \phi \ge 0\) с \(\int_{\R^m}\phi(x)dx = 1\), положим \(K_n(x) = n^m\phi(nx)\). Тогда:
    \[\int_{|x| \ge \delta} = \int_{|x| \ge \delta} n^m\phi(nx)dx = \left\{\begin{array}{c}
        y = nx \\
        |J| = n^m
    \end{array}\right\} = \int_{|y| \ge n\delta}\phi dy = \int_{\R^m}\phi I_{\{|y| \ge n\delta\}} dy \ra 0\]
\end{example}

\begin{theorem}
    Пусть \(\{K_n\}_{n = 1}^\infty\) --- аппроксимация единицы и \(f\) --- ограниченная измеримая функиция на \(\R^n\). Тогда справедливы утверждения:
    \begin{enumerate}
        \item Если \(f\) непрерывна в \(x \in \R^m\), то \(K_n * f(x) \ra f(x)\).
        \item Если \(f\) непрервна в каждой точке компакта \(K \subset \R^m\), то \(K_n * f \rightrightarrows f\) на \(K\)
    \end{enumerate}
\end{theorem}
\begin{proof}
    Из пункта 2 определения \(K_n\) следует, что \(f(x) = \int_{\R^m} f(x)K_n(t)dt\), поэтому \(K_n * f(x) - f(x) = \int_{\R^m} (f(x - t) - f(x))K_ndt\). Зафиксируем \(\epsilon > 0\). Пользуясь равномерной непрерывностью \(f\) на \(K\), найдем такое \(\delta > 0\), что \(\forall x \in K \forall t \in \R^m \left(|t| < \delta \Ra |f(x - t) - f(x)| \le \frac{\epsilon}{2}\right)\). Разобьем последний интеграл следующим образом:
    \[K_n * f(x) - f(x) = \left( \int_{|t| < \delta} + \int_{|t| > \delta} \right)(f(x - t) - f(x))K_n(t)dt = I_1 + I_2\]
    Имеем:
    \[|I_1| \le \int_{|t| < \delta} |f(x - t) - f(x)||K_n(t)|dt \le \frac{\epsilon}{2}\int_{\R^m}K_n(t)dt = \frac{\epsilon}{2}\]
    \[|I_2| \le \int_{|t| \ge \delta} (f(x - t) + |f(x)|)K_n(t)dt \le 2C \int_{|t| \ge \delta} K_n(t)dt\]
    Из пункт 3 опрелеления \(K_n\), найдем \(N = N(\delta)\) такое, что \(\forall n \ge N \left( 2C \int_{|t| \ge \delta} K_n(t)dt \le < \frac{\epsilon}{2} \right)\), тогда при таких \(n\) выполнено \(|K_n * f(x) - f(x)| < \epsilon\), т.е. \(K_n * f \rightrightarrows f\) на \(K\).

    Пункт 1 следует из пункта 2, где в качестве компакта рассматривается точка.
\end{proof}

\begin{proposition}

\end{proposition}

\end{proof}

\end{definition}

\section{Тригонометриеский ряд Фурье}
\begin{definition}
    Ряд:
    \begin{equation}
        \frac{a_0}{2} + \sum_{k = 1}^\infty a_k\cos kx + b_k\sin kx
    \end{equation}
    Называется тригонометрическим рядом с коэффициентами \(a_k, b_k \in \Cm\).
\end{definition}

По формулам Эйлера: \(\cos kx = \frac{e^{ikx} + e^{-ikx}}{2}, \sin kx = -i\left( \frac{e^{ikx} - e^{-ikx}}{2} \right)\), поэтому частичные суммы можно переписать в виде:
\[S_n(x) = \frac{a_0}{2} + \sum_{k = 1}^n a_k\cos kx + b_k\sin kx = \sum_{k = -n}^n c_ke^{ikx}\]
Где \(c_{\pm k} = \frac{a_k \mp ib_k}{2}, \frac{a_0}{2} = c_0\). 

\begin{definition}
        Ряд:
    \begin{equation}
        \sum_{k = -\infty}^{\infty} c_ke^{ikx}
    \end{equation}
    Называется тригонометрическим рядом в комплексной форме, и его сумма считается как 
    \[\sum_{k = -\infty}^{\infty} c_ke^{ikx} = \lim_{n \ra \infty} \sum_{-n}^n c_ke^{ikx}\]
\end{definition}

\begin{lemma}
    Пусть \(f \in L_1(-\pi, \pi)\) и тригонометрический ряд в форме (2.1) или (2.2) сходится почти всюду к \(f\) и существует \(g \in L_1(-\pi, \pi)\), такая, что \(|S_n(x)| \le g \forall n\) для почти всех \(x\). Тогда:
    \[a_k = \frac{1}{\pi}\int_{-\pi}^\pi f(t)\cos kt, b_k = \frac{1}{\pi}\int_{-\pi}^\pi f(t)\sin kt, c_k = \frac{1}{2\pi}\int_{-\pi}^\pi f(t)e^{-ikx}\]
\end{lemma}
\begin{proof}
    Зафиксируем \(k \in \Z\). По условию, \(S_n(x) e^{-ikx} \ra f(x)e^{-ikx}\) почти всюду. При этом, \(|S_n(x) e^{-ikx}| \le g\). Поэтому, по теореме Лебега о мажорируемой сходимости, имеем: 
    \[\lim_{n \ra \infty} \int_{-\pi}^{\pi}S_n(x)e^{-ikx}dx = \int_{-\pi}^{\pi}f(x)e^{-ikx}dx\]
    Так как \(\int_{-\pi}^{\pi}e^{imx}e^{-ikx}dx = \left\{\begin{array}{l}
        0, k \ne m \\
        2\pi, k = m
    \end{array}\right.\), то 
    \[\int_{-\pi}^{\pi}S_ne^{-ikx}dx = 2\pi c_k, n \ge |k|\]
    Следовательно, \(\int_{-\pi}^{\pi} f(x)e^{-ikx}dx = 2\pi c_k\)
\end{proof}

\begin{definition}
    Пусть \(f \in L_1(T)\). Числа
    \[a_k(f) = \frac{1}{\pi}\int_{-\pi}^\pi f(t)\cos kt, b_k(f) = \frac{1}{\pi}\int_{-\pi}^\pi f(t)\sin kt, c_k(f) = \frac{1}{2\pi}\int_{-\pi}^\pi f(t)e^{-ikx}\]
    Называются коэффициентами Фурье функции \(f\), а ряд
    \[S(f, x) = \frac{a_0(f)}{2} + \sum_{k = 1}^\infty a_k(f)\cos kx + b_k(f)\sin kx = \sum_{k = -\infty}^{\infty} c_k(f)e^{ikx}\]
    Называется тригонометрическим рядом Фурье функции \(f\)
\end{definition}

\begin{note}
    По лемме Римана об осцилляции, \(a_k(f), b_k(f) \ra 0\) при \(k \ra \infty\) и \(c_k(f) \ra 0\) при \(|k| \ra 0\).
\end{note}

\subsection{Поточечная сходимость рядов Фурье}
\begin{definition}
    \(D_n(t) = \frac{1}{2}\sum_{k = -n}^n e^{ikt} = \frac{1}{2} + \sum_{k = 1}^n \cos kt\) называется \(n\)-ым ядром Дирихле.
\end{definition}

\begin{note}
    Функция \(D_n\) непрерывна, \(2\pi\)-периодическая, \(\int_{-\pi}^{\pi} D_n(t)dt = \pi\).
\end{note}

Имеем:
\[S_n(f, x) = \sum_{k = -n}^n\left( \frac{1}{2\pi}\int_{-\pi}^{\pi}f(s)e^{-iks}ds \right)e^{ikx} = \frac{1}{2\pi}\int_{-\pi}^\pi f(s)\sum_{k = -n}^n e^{ik(x - s)}ds = \frac{1}{\pi}\int_{-\pi}^{\pi} f(s)D_n(x - s)ds = \]
\[\frac{1}{\pi}D_n * f(x) = \frac{1}{\pi}f * D_n(x)\]

\[D_n(t) = \frac{1}{2}e^{-nit}\left( 1 + e^{it} + \dots + e^{2nit} \right) = \frac{1}{2}e^{-nit} \frac{e^{2(n + 1)it} - 1}{e^{it} - 1} = \frac{e^{\left( n + \frac{1}{2} \right)it} - e^{-\left( n + \frac{1}{2} \right)it}}{2 \left( e^{\frac{it}{2}} - e^{-\frac{it}{2}} \right)} = \frac{\sin\left( n + \frac{1}{2} \right)t}{2 \sin \frac{t}{2}}\]

\begin{lemma}
    Пусть \(f \in L_1(T)\) и \(0 < \delta < \pi\). Тогда \(\forall x \in \R\) выполнено:
    \[S_n(f, x) = \frac{1}{\pi}\int_{-\delta}^{\delta}\frac{f(x - t)}{t}\sin\left( n + \frac{1}{2} \right)t dt + \epsilon_n(x), \lim_{n \ra \infty} \epsilon_n(x) = 0\]
\end{lemma}
\begin{proof}
    Имеем:
    \[S_n(f, x) = \frac{1}{\pi}\int_{-\pi}^{\pi}f(x - t)\frac{\sin\left( n + \frac{1}{2} \right)t}{2\sin\frac{t}{2}}dt = \]
    \[ = \frac{1}{\pi}\int_{|t| < \delta} \frac{f(x - t)}{t}\sin\left( n + \frac{1}{2} \right)tdt + \frac{1}{\pi}\int_{\delta \le |t| \le \pi} \frac{f(x - t)}{t}\sin\left( n + \frac{1}{2} \right)t + \]
    \[+ \frac{1}{\pi}\int_{-\pi}^\pi f(x - t)\left( \frac{1}{2\sin \frac{t}{2}} - \frac{1}{t} \right)\sin\left( n + \frac{1}{2} \right)t\]
    \(h(t) = f(x - t)\) интегрируема на \((-\pi, \pi)\). Положим \(g(t) = \frac{1}{2\sin \frac{t}{2}} - \frac{1}{t}\). По непрерывности, доопределим \(g(0) = 0\) --- то есть \(g\) непрерывна на \([-\pi, \pi] \Ra\) ограничена.  Аналогично, \(\frac{f(x - t)}{t}\) интегрируема на \(\pi \ge |t| \ge \delta\). Но тогда по лемма Римана об осцилляции, получаем желаемое.
\end{proof}

\begin{proposition}[Признак Дини]

\end{proposition}

\end{proof}

\begin{corollary}

\end{proof}

\begin{proposition}[Признак Дирихле]
    Пусть \(f \in L_1(T), x \in \R\). Если \(f\) монотонна на \([x - \delta, x + \delta]\) при некотором \(\delta > 0\), то \(S(f, x) = \frac{f(x + 0) + f(x - 0)}{2}\)
\end{proposition}
\begin{proof}
    Пусть \(f\) нестрого возрастает на \([x - \delta, x + \delta]\). Рассмотрим:
    \[\int_0^\sigma \frac{f(x + t) - f(x + 0)}{t}\sin\left( n + \frac{1}{2} \right)tdt, \sigma \in (0, \delta)\]
    Также, пусть \(I(n) = \int_0^n \frac{\sin t}{t}dt \ra_{n \ra \infty} \int_0^{+\infty} \frac{\sin t}{t}dt \Ra \exists M: |I(n)| \le M\) на \([0, +\infty)\). Следовательно:
    \[\left| \int_0^x \frac{\sin \left( n + \frac{1}{2} \right)t}{t}dt \right| = \left| \int_0^{x\left( n + \frac{1}{2} \right)} \frac{\sin u}{u}du \right| \le M\]
    Т.к. монотонная функция имеет множество точек разрыва меры 0, и разрывы могут быть только первого рода, то зааменим \(f(x)\) на \(f(x + 0)\), т.к. интеграл от этого не изменится. По неравенству Абеля:
    \[\left| \int_0^\sigma \frac{f(x + t)- f(x + 0)}{t}\sin\left( n + \frac{1}{2} \right)t dt\right| \le 2M |f(x + \sigma) - f(x + 0)|\]
    Заметим, что по теореме Римана об осцилляции,
    \[\int_\sigma^\delta \frac{f(x + t) - f(x + 0)}{t}\sin\left( n + \frac{1}{2} \right)tdt \ra 0, n \ra \infty\]
    Получаем, что 
    \[\int_0^\delta \frac{f(x + t) - f(x + 0)}{t}\sin\left( n + \frac{1}{2} \right)tdt \ra 0\]
    Аналогично доказывается, что 
    \[\int_0^\delta \frac{f(x - t) - f(x - 0)}{t}\sin\left( n + \frac{1}{2} \right)tdt \ra 0\]
    Из чего по теореме следует утверждение.
\end{proof}

\end{lemma}
\begin{proof}
    \(\phi_n(x) \ra 0 \forall x\) по лемме Римана об осцилляции. Предположим противное, т.е.

    У последовательности \(x_k\) имеется сходящаяся подпоследовательность, поэтому считаем, что \(x_k \ra x_0\).
    \[\epsilon \le |\phi_{n_k}(x_k)| \le |\phi_{n_k}(x_k) - \phi_{n_k}(x_0)| + \underbrace{|\phi_{n_k}(x_0)|}_{\ra 0}\]

\end{proof}

\begin{definition}
    Говорят, что функция \(f\) удовлетворяет условию Гельдера на \(E \subset \R\) с показателем \(\alpha \in (0, 1]\), если \(\exists C > 0: \forall x, y \in E: |f(x) - f(y)| \le C|x - y|^\alpha\)
\end{definition}

\begin{example}
    Рассмотрим \(f = \sqrt{x}, E = (0, +\infty)\). Данная функция удовлетворяет условию Гельдера с \(\alpha = \frac{1}{2}\) (следует из неравенства \(|\sqrt{x} - \sqrt{y}| \le \sqrt{|x - y|}\)).
\end{example}

\begin{definition}
    Функция \(f\) называется кусочно-непрерывной на \([a, b]\), если \(\exists I\) --- разбиение \([a, b]\) точками \(a = x_0 < x_1 < \dots < x_n = b\), что \(f\) непрерывна на \((x_i, x_{i + 1})\) и существуют и конечны пределы \(f(x_i \pm 0)\) при \(0 < i < n\) и \(f(a + 0), f(b - 0)\).
\end{definition}

\begin{example}
    Если для непрерывной функции \(f\), производная \(f'\) кусочно-непрерывная, то \(f\) удовлетворяет условию Гельдера с \(\alpha = 1\).
\end{example}

\begin{theorem}[Признак Липшица]
    Пусть \(f \in L_1(T)\) и удовлетворяет условию Гельдера на \([a, b]\). Тогда \(S_n(f, x) \rightrightarrows f(x)\) на любом \([a', b'] \subset (a, b)\).
\end{theorem}
\begin{proof}
    Существует \(\delta < 0\), такое, что \(|t| < \delta \Ra x - t \in [a, b]\) для \(x \in [a', b'] \subset (a, b)\). Для \(\delta_1 \in (0, \delta)\), имеем \(|f(x \pm t) - f(x)| \le C|t|^\alpha\). Тогда:
    \[\left| \int_0^{\delta_1} \frac{f(x + t) + f(x - t) - 2f(x)}{t}\sin\left( n + \frac{1}{2} \right)tdt \right| \le\]
    \[\le \int_0^{\delta_1} \frac{|f(x + t) - f(x)| + |f(x - t) - f(x)|}{t}dt \le\]
    \[\le 2C\int_0^{\delta_1}t^{\alpha - 1}dt = \frac{2C}{\alpha}\delta_1^\alpha < \frac{\epsilon}{2}\]
    Также, 
    \[\int_{\delta_1}^\delta \frac{f(x + t) + f(x - t) - 2f(x)}{t}\sin\left( n + \frac{1}{2} \right)tdt = \]
    \[=\int_{-\pi}^{\pi} (f(x + t) + f(x - t) - 2f(x))\frac{I_{[\delta_1, \delta]}}{t}\sin\left( n + \frac{1}{2} \right)dt \rightrightarrows 0\]
    Для доказательства равномерной сходимости достаточно разбить интеграл на три части и воспользовалься предыдущей леммой. Тогда
    \[\exists N: \forall n \ge N \forall x \in [a', b']: \left| \int_{-}^{\pi} (f(x + t) + f(x - t) - 2f(x))\frac{I_{[\delta_1, \delta]}}{t}\sin\left( n + \frac{1}{2} \right)dt \right| < \frac{\epsilon}{2}\]
    А значит:
    \[\left| \int_{0}^{\delta} \frac{f(x + t) + f(x - t) - 2f(x)}{t}\sin\left( n + \frac{1}{2} \right)tdt \right|  < \epsilon\]
    Что и требовалось
\end{proof}

\begin{corollary}
    Пусть \(f \in C(T)\) и \(f'\) кусочно-непрерывна на \([-\pi, \pi] \Ra S_n(f, x) \rightrightarrows f(x)\) на \(\R\).
\end{corollary}

\subsection{Почленное дифференцирование и интегрирование рядов Фурье}

\begin{theorem}
    Пусть \(f \in C(T)\) и \(f'\) кусочно непрерывна на \([-\pi, \pi]\). Тогда \(S(f', x) = S'(f, x)\) --- почленно продифференцированный ряд Фурье
\end{theorem}
\begin{proof}
    По формуле интегрирования по частям:
    \[2\pi c_n(f') = \int_{-\pi}^{\pi} f'(x)e^{-inx}dx = f(x)e^{-inx}|_{-\pi}^\pi + in \int_{-\pi}^\pi f(x)e^{-inx}dx = 2\pi i n c_n(f)\]
    Итак, \(c_n(f')e^{inx} = (c_n(f)e^{inx})'\), что и требовалось
\end{proof}

\begin{corollary}[О порядке убывания коэффициентов Фурье]
    Пусть \(f, f', \dots f^{(m - 1)} \in C(T)\) и \(f^{(m)}, f^{(m + 1)}\) --- кусочно-непрерывны на \([-\pi, \pi]\). Тогда \(c_n(f) = O\left( \frac{1}{|n|^{m + 1}} \right)\) при \(n \ra \infty\)
\end{corollary}
\begin{proof}
    Последовательно применяя предыдущую теорему, имеем: \(c_n(f^{(m)}) = (in)^m c_n(f)\). К точкам разрыва \(f^{(m + 1)}\) добавим еще точки \(-\pi , \pi\) и упорядочим по возрастанию: \(-\pi = x_0 < x_1 < \dots < x_N = \pi\). По формуле интегрирования по частям:
    \[\int_{x_{j - 1}}^{x_j} f^{(m + 1)}(x)e^{-inx}dx = f^{(m)}(x)e^{-inx}|^{x_j - 0}_{x_{j - 1} + 0} + in\int_{x_{j - 1}}^{x_j}f^{(m)}(x) e^{-inx}dx\]
    \[c_n(f^{m + 1}) = \lambda_n + inc_n(f^{(m)}), \lambda_n = \frac{1}{2\pi}\sum_{j = 1}^N \left(f^{(m)}(x_j - 0)e^{-inx_j} - f^{(m)}(x_{j - 1} + 0)e^{-inx_{j - 1}}\right)\]
    Заметим, что \(\lambda_n\) ограничена. Т.к. \(c_n(f^{(m + 1)}) = o(1)\), то \(inc_n(f^{(m)}) = O(1)\).
\end{proof}

\begin{problem}
    Если в условии следствия \(c_n(f) = O\left( \frac{1}{|n|^{m + 1 + \epsilon}} \right)\) для некоторого \(\epsilon > 0\), то \(f^{(m)} \in C(T)\).
\end{problem}

\begin{exercise}
    По свойствам интеграла с переменным пределом получить, что \(f^{(m)}\) кусочно-непрерывна, если \(f^{(m + 1)}\) кусочно-непрерывна
\end{exercise}

\begin{lemma}
    Пусть \(f \in L_1(T), F(x) = \int_{-\pi}^x (f(t) - c_0(f))dt, x \in \R\). Тогда \(F\) непрерывна, \(2\pi\)-периодическая и является суммой своего ряда Фурье:
    \[F(x) = c_0(F) + \sum_{n \ne 0} \frac{c_n(F)}{in}e^{inx}\]
\end{lemma}
\begin{proof}
    Положим \(f_0 = f - c_0(f)\). Найдем \(c_n(F)\) при \(n \ne 0\):
    \[\int_{-\pi}^\pi F(x)e^{-inx}dx = \int_{-\pi}^\pi\left( \int_{-\pi}^x f_0(t)e^{-inx} dt\right)dx\]
    Применяя теорему Фубини, получаем:
    \[\int_{-\pi}^\pi\left( \int_{t}^\pi f_0(t)e^{-inx} dx\right)dt = \int_{-\pi}^\pi f_0(t) \frac{e^{-int} - (-1)^n}{in}dt =\]
    \[= \int_{-\pi}^\pi(f(t) - c_0(f))\frac{e^{-int}}{in}dt = \frac{1}{in}\int_{-\pi}^\pi f(t)e^{-int}dt\]
    Следовательно, \(c_n(F) = \frac{c_n(f)}{in}, n \ne 0\). Положим \(f = f^+ - f^- \Ra F(x) = \int_{-\pi}^x f^+(t)dt - \int_{-\pi}^x f^-(t)dt - c_0(x + \pi)\), то сходимость ряда Фурье к \(F\) следует из признака Дирихле (как сумма рядов Фурьте для монотонных функиций).
\end{proof}

\begin{note}
    Положим \(x = 0\) в формуле из леммы:
    \[F(0) = c_0(F) + \sum_{n \ne 0} \frac{c_n(F)}{in}\]
    Следовательно, сходится ряд:
    \[-\sum_{n \ne 0}\frac{c_n(f)}{in} = -\sum_{n = 1}^\infty \frac{c_n(f) - c_{-n}(f)}{in} = \sum_{n = 1}^\infty \frac{b_n(f)}{n}\]
\end{note}

\begin{theorem}
    Если \(f \in L_1(T), -\infty \le a < b \le \infty\), то
    \[\int_a^b f(x)dx = \sum_{n = -\infty}^\infty c_n(f)\int_a^be^{inx}dx\]
\end{theorem}
\begin{proof}
    \[F(b) - F(a) = \sum_{n \ne 0} \frac{c_n(f)}{in}(e^{inb} - e^{ina}) = \sum_{n \ne 0} c_n(f) \int_a^be^{inx}dx = \int_a^b f(x)dx - c_0(f)(b - a)\]
\end{proof}

\begin{corollary}[Теорема единственности]
    Если \(f, g \in L_1(T)\) и \(c_n(f) = c_n(g) \forall n \in \Z\), то \(f = g\) почти всюду.
\end{corollary}
\begin{proof}
    \(\forall x \in [-\pi, \pi] \int_{-\pi}^x f(x) dt = \int_{-\pi}^x g(t) dt\). Из теории меры нам известно, что это гарантирует равенство функций \(f, g\) почти всюду.
\end{proof}

\subsection{Суммирование рядов Фурье методом средних арифметических}

Ряд Фурье непрерывной функции может расходиться в некоторых точках (и даже на множестве меры \(0\)). Однако он всегда суммируем методом средних арифметических.

\begin{definition}
    Пусть \(f \in L_1(T)\), тогда определим последовательность:
    \[\sigma_N(f, x) = \frac{S_1(f, x) + \dots + S_N(f, x)}{N}\]
    Данная последовательность называется суммой Фейера.
\end{definition}

\begin{note}
    \(\sigma_N(f, x) = \sum_{n = -N}^N\left( 1 - \frac{|n|}{N + 1} \right)c_n(f)e^{inx}\)
\end{note}
Т.к. \(S_n(f) = \frac{1}{\pi}f * D_n\), то \(\sigma_N(f) = \frac{1}{\pi(N + 1)}\sum_{n = 0}^N f * D_n = \frac{1}{\pi} f * F_N\), где \(F_N = \frac{1}{N + 1}\sum_{n = 0}^N D_n\) --- ядро Фейера. Получим формулу для ядра Фейера без значка суммирования. Имеем:
\[(N + 1)F_N(t) = \sum_{n = 0}^N \frac{\sin\left( n + \frac{1}{2} \right)t\sin\frac{t}{2}}{2\sin^2\frac{t}{2}} = \sum_{n = 0}^N \frac{\cos nt - \cos(n + 1)t}{4\sin^2\frac{t}{2}} =\frac{1 - \cos(N + 1)t}{4\sin^2\frac{t}{2}} =\]
\[= \frac{\sin^2\frac{N + 1}{2}t}{2\sin^2\frac{t}{2}}, t \ne 2\pi, m \in \Z\]
Таким образом, получили:
\[F_N(t) = \left\{\begin{array}{l}
    \frac{1}{2(N + 1)} \frac{\sin^2 \frac{N + 1}{2}t}{\sin^2 \frac{t}{2}}, t \ne 2\pi m \\
    \frac{N + 1}{2}, t = 2\pi m
\end{array}\right.\]

\begin{note}[Свойства ядра Фейера]
    \begin{enumerate}
        \item \(F_N \ge 0\)
        \item \[\int_{-\pi}^\pi F_N(t)dt = \pi\]
        \item \[\forall \delta > 0 \int_{\delta < |t| \le \pi} F_N(t)dt \ra 0, N \ra +\infty\]
    \end{enumerate}
\end{note}
\begin{proposition}
    \begin{enumerate}
        \setcounter{enumi}{2}
        \item \(0 \le F_N(t) \le \frac{1}{2(N + 1)} \frac{1}{\sin^2 \frac{\delta}{2}}\)
    \end{enumerate}
\end{proposition}

\begin{corollary}
    \(\left\{ \frac{1}{\pi}F_N \right\}\) --- аппроксимация единицей.
\end{corollary}

Тогда получаем следствие:

\begin{theorem}[Фейер]
    Пусть \(f_1 \in L_1(T)\), тогда справедливы утверждения:
    \begin{enumerate}
        \item Если \(f\) непрерывна в точке \(x\), то \(\sigma_N(f, x) \ra f(x)\).
        \item Если \(f\) непрерывна на компакте \(K\), то \(\sigma_N(f, x) \rightrightarrows_K f\). В частности, если \(f \in C(T)\), то \(\sigma_N(f, x) \rightrightarrows f\) на \(\R\)
    \end{enumerate}
\end{theorem}

\begin{definition}
    \(T(x) = \alpha_0 + \sum_{k = 1}^n\alpha_k \cos kx + \beta_k \sin kx = \sum_{k = -n}^n \gamma_k e^{ikx}\) называется тригонометрическим многочленом.
\end{definition}

\begin{corollary}[1-ая теорема Вейерштрасса]
    Если \(f \in C(T)\), то \(\forall \epsilon > 0 \exists T\) --- тригонометрический многочлен, такой, что \(\|f - T\|_{\infty} < \epsilon\), где \(\|f\|_\infty = \sup_{[-\pi, \pi]} |f|\).
\end{corollary}
\begin{proof}
    По теореме Фейера, \(\|f - \sigma_N(f)\|_{\infty} \ra 0\) при \(N \ra \infty\), поэтому в качестве \(T(x)\) положим \(\sigma_N(f, x)\).
\end{proof}

\begin{corollary}[2-ая теорема Вейерштрасса]
    Если \(f \in C[a, b]\), то \(\forall \epsilon > 0 \exists P(x) = \sum_{k = 0}^n \alpha_kx^k\) --- алгебраический многочлен, т.ч. \(\|f - P\|_{C[a, b]} < \epsilon\)
\end{corollary}
\begin{proof}
    Замена \(x = a + \frac{b - a}{\pi}t\) сводит отрезок \([a, b]\) к случаю \([0, \pi], f \in C[0, \pi]\). Рассмотрим \(\tilde{f}(x) = \left\{\begin{array}{l}
        f(x), x \in [0, \pi] \\
        f(-x), x \in [-\pi, 0]
    \end{array}\right.\) с периодом \(2\pi\) на \(\R\). Тогда \(\tilde{f} \in C(T)\) и по предыдущему следствию \(\exists T\) --- тригонометрический многочлен, такой, что \(\|\tilde{f} - T\|_\infty < \frac{\epsilon}{2}\). Функцию \(T\) можно представить рядом Маклорена на \(\R\): \(T(x) = \sum_{k = 0}^\infty \alpha_kx^k\), причем данный ряд сходится равномерно на любом подотрезке \(\R\). Тогда \(\exists P(x) = \sum_{k = 0}^n \alpha_kx^k: \|T - P\|_{C[-\pi, \pi]} < \frac{\epsilon}{2}\). Следовательно, \(\|f - P\|_{C[0, \pi]} \le \|\tilde{f} - P\|_{C[-\pi, \pi]} \le \|\tilde{f} - T\|_[C[-\pi, \pi]] + \|T - P\|_{C[-\pi, \pi]} = \epsilon\)
\end{proof}

\begin{theorem}
    \(L_p(E)\) банахово.
\end{theorem}
\begin{proof}
    Пусть \(\{f_n\}\) --- фундаментальная последовательность в \(L_p(E)\). Покажем, что \(\exists f \in L_p(E), \exists \{f_{n_k}\}: f_{n_k} \ra f\) почти всюду на \(E\). Из определения фундаментальной последовательности по индукции строим подпоследовательность \(\{f_{n_k}\}\), т.ч. \(\|f_{n_{k + 1}} - f_{n_k}\| < \frac{1}{2^k}\).
    
    Заметим, что \(\{f_{n_k}\}\) сходится \(\Lra\) сходится почти всюду ряд
    \[(*) = f_{n_1}(x) + \sum_{k = 1}^\infty (f_{n_{k + 1}}(x) - f_{n_k}(x))\]

    Покажем, что ряд \((*)\) сходится абсолютно почти всюду. Рассмотрим \(F = \sum_{k = 1}^\infty |f_{n_{k + 1}} - f_{n_k}|, F_n = \sum_{k = 1}^n |f_{n_{k + 1}} - f_{n_k}|\). Последовательность \(\{F_n^p\}\) нестрого возрастает, тогда по теореме Леви,
    \[\lim_{n \ra \infty} \int F_n^p d\mu = \int F^p d\mu\]
    
    Интегралы в левой части ограничены:
    \[\|F_n\|_p \le \sum_{k = 1}^n\|f_{n_{k + 1}} - f_{n_k}\|_p < \sum_{k = 1}^\infty \frac{1}{2^k} = 1 \Ra \|F\|_p \le 1\]
\lecture{7}
    Получили, что \(F\) интегрируема, но тогда она конечна почти всюду и ряд \((*)\) сходится абсолютно почти всюду на \(E\), а значит, существует \(f(x) = \lim_{k \ra \infty} f_{n_k}(x)\), кроме того, \(|f| \le |f_{n_1}| + |F| \Ra |f|^p \le \underbrace{2^p(|f_{n_k}|^p + |F|^p)}_{\text{интегрируемая}}\). Следовательно \(|f|^p\) интегрируема и \(f \in L_p(E)\).

    Покажем, что \(f_n \ra f\) в \(L_p(E)\). Из определения фундаментальности: \(\forall \epsilon > 0 \exists N: \forall n, m > N: (\|f_n - f_m\|_p < \epsilon)\). Зафиксируем \(n > N\). Имеем \(|f_n - f_{n_k}|^p \ra |f_n - f|^p\). Заметим, что \(|f_n - f_{n_k}|^p \le 2^p(|f_n|^p + |f_{n_k}|^p) \le 2^p(|f_n|^p + |F|^p)\). Тогда по теореме Лебега о мажорируемой сходимости \(\|f_n - f_{n_k}\|_p \ra \|f_n - f\|_p\). Следовательно, \(\|f_n - f\|_p \le \epsilon\). 
\end{proof}

\begin{definition}
    \(L_\infty(E) = \{f: E \ra \Cm \vert \exists \tilde{f}\text{ ограниченная на }E,\;f = \tilde{f}\text{ почти всюду }\} /_\sim\), т.е. фактормножество по отношению \(\sim\). Введем на этом пространстве норму \(\|f\|_\infty = \inf_{\tilde{f} \sim f} \sup_E|\tilde{f}|\).
\end{definition}

\begin{definition}
    Пусть \(X\) --- нормированное пространство, \(\{\phi_k\}_{k = 1}^\infty \subset X\). Система \(\{\phi_k\}_{k = 1}^\infty\) называется полной в \(X\), если \(\forall f \in X \forall \epsilon > 0 \exists\alpha_1, \dots \alpha_n: \left\| f - \sum_{k = 1}^n \alpha_k\phi_k \right\| < \epsilon\)
\end{definition}

\begin{theorem}
    Тригонометрическая система в вещественной форме \(\left\{ \frac{1}{2}, \cos kx, \sin kx \right\}\) и в комплексной форме \(\left\{ e^{ikx} \right\}_{k \in \Z}\) полны в \(L_p(T), 1 \le p < \infty\)
\end{theorem}
\begin{proof}
    Пусть \(f \in L_P(T), \epsilon > 0\). Тогда \(f \in L(-\pi, \pi)\). По теореме о приближении, \(\exists g \in C_c^\infty(\R)\), такая, что \(\|f - g\|_p < \epsilon\). Можно считать, что носитель \(g\) лежит в \((-\pi, \pi)\) (достаточно вспомнить доказательство теоремы о приближении). Существует \(h \in C(T)\), такая, что \(g = h\) на \([-\pi, \pi]\). По теореме Вейерштрасса, \(\exists T\) --- тригонометрический многочлен, такой, что \(\|h - T\|_\infty < \epsilon\). Тогда \(\|h - T\|_p = \left( \int_{-\pi}^\pi |h - T|^p \right)^{\frac{1}{p}} < \epsilon (2\pi)^{\frac{1}{p}}\). Тогда по неравенству треугольника, имеем \(\|f - T\|_p \le \|f - g\|_p + \|g - T\|_p = \|f - g\|_p + \|h - T\|_p < (1 + (2\pi)^{\frac{1}{p}})\epsilon\).
\end{proof}

\section{Ортогональные ряды}

\begin{definition}
    Пусть \(V\) --- линейное пространство над \(\R\) или \(\Cm\), функция \((\cdot, \cdot): V \times V \ra \mathbb{K}\) называется скалярным произведение, если она удовлетворяет следующим свойствам:
    \begin{enumerate}
        \item \(\forall x \in V: (x, x) \ge 0\), причем \((x, x) = 0 \Lra x = 0\)
        \item \(\forall x, x', y \in V \forall \lambda: (\lambda x + x', y) = \lambda(x, y) + (x', y)\)
        \item \(\forall x, y \in V: \overline{(x, y)} = (y, x)\).
    \end{enumerate}
    Пространство \(V\) с введенным на нем скалярным произведением называется евклидовым
\end{definition}

\begin{note}
    Всякое евклидовое пространство является нормированным относительно нормы \(\|x\| = \sqrt{(x, x)}\) (евклидова норма). Действительно, неравенство треугольника следует из неравенства КБШ: \(|(x, y)| \le \|x\|\cdot\|y\|\)
\end{note}

\begin{note}
    Скалярное произведение непрерывно.
\end{note}
\begin{proof}
    Пусть \(x_n \ra x, y_n \ra y\). Тогда \(|(x_n, y_n) - (x, y)| = |(x_n, y_n - y) + (x_n - x_n, y)| \le\) по КБШ \(\le \|x_n\|\cdot\|y_n - y\| + \|x_n - x\|\cdot\|y\| \ra_{n \ra \infty} 0\).
\end{proof}

\begin{definition}
    Векторы \(x, y \in V\) называются ортогональными, если \((x, y) = 0\). В таком случае пишут \(x \perp y\).
\end{definition}

\begin{lemma}[Теорема Пифагора]
    Пусть \(x_1, \dots x_n \in V\), такие, что при \(i \ne j: x_i \perp x_j\). Тогда:
    \[\|x_1 + \dots + x_n\|^2 = \|x_1\|^2 + \dots + \|x_n\|^2\]
\end{lemma}
\begin{proof}
    \[\left\|\sum_{k = 1}^n x_k\right\|^2 = \left( \sum_{k = 1}^n x_k, \sum_{k = 1}^n x_k \right) = \sum_{i, j = 1}^n (x_i, x_j) = \sum_{k = 1}^n \|x_k\|^2\]
\end{proof}

\begin{definition}
    Семейство \(\{e_k\}_{k = 1}^\infty\) называется ортогональной системой, если при \(i \ne j\) верно \(e_i \perp e_j\) и \(\forall k: e_k \ne \vec{0}\).
\end{definition}

\begin{note}
    Элементы ортогональной системы линейно независимы.
\end{note}
\begin{proof}
    Действительно, пусть \(\lambda_1e_1 + \dots + \lambda_me_m = \vec{0}\). Рассмотрим \((e_i, \lambda_1e_1 + \dots + \lambda_me_m) = \lambda_i\|e_i\|^2 = 0 \Ra \lambda_i = 0\).
\end{proof}

\begin{definition}
    Пусть \(\{e_k\}_{k = 1}^\infty\) --- ортогональная система, \(x \in V\). Тогда:
    \begin{enumerate}
        \item \(\widehat{x} = \frac{(x, e_k)}{\|e_k\|^2}\) --- \(k\)-ый коэффициент Фурье \(x\) по \(\{e_k\}\).
        \item \(\sum_{k = 1}^\infty \widehat{x}_ke_k\) --- ряд Фурье \(x\) по \(\{e_k\}\), в таком случае пишут \(x \sim \sum_{k = 1}^\infty \widehat{x}_ke_k\)
    \end{enumerate}
\end{definition}

\begin{theorem}[Свойство минимальности]
    \[\min_{c_1, \dots c_n} = \left\| x - \sum_{k = 1}^n c_ke_k \right\| = \left\| x - \sum_{k = 1}^n \widehat{x}_ke_k \right\|\]
    Причем равенство возможно только при \(c_k = \widehat{x}_k\). Кроме того, справедливо тождество Бесселя:
    \[\left\| x - \sum_{k = 1}^n \widehat{x}_ke_k \right\|^2 = \|x\|^2 - \sum_{k = 1}^n |\widehat{x}_k|^2 \|e_k\|^2 \]
\end{theorem}
\begin{proof}
    Положим \(S_n = \sum_{k = 1}^n \widehat{x}_ke_k, L = \langle e_1, \dots e_n \rangle\)

    Покажем, что \(\forall y \in L: x - S_n \perp y\). Рассмотрим \((x - S_n, e_m) = (x, e_m) - (S_n, e_m) = (x, e_m) - \widehat{x}_m\|e_m\|^2 = 0\). Т.к. \(y \in \langle e_1, \dots e_n \rangle\), из линейности скалярного произведения получаем, что \(y \perp x - S_n\).

    Положим \(y = \sum_{k = 1}^n c_ke_k\). Тогда \(S_n - y \in L\), а значит, \(x - S_n \perp S_n - y\). По теореме Пифагора \(\|x - y\|^2 = \|(x - S_n) + (S_n - y)\|^2 = \|x - S_n\|^2 + \|S_n - y\|^2 = \|x - S_n\|^2 + \sum_{k = 1}^n|\widehat{x}_k - c_k|^2\|e_k\|^2 = (*)\). Получаем, что \(\|x - y\| \ge \|x - S_n\|\), причем равенство возможно тогда и только тогда, когда \(c_k = \widehat{x}_k\). Тождество Бесселя получаем из \((*)\) при \(y = 0\)
\end{proof}

\begin{corollary}[Неравенство Бесселя]
    Если \(\{e_k\}_{k = 1}^\infty\) --- ортогональная система, \(x \in V\), то 
    \[\sum_{k = 1}^\infty |\widehat{x}_k|^2\|e_k\|^2 \le \|x\|^2\]
\end{corollary}
\begin{proof}
    Из тождества Бесселя следует, что 
    \[\|x\|^2 \le \sum_{k = 1}^n |\widehat{x}_k|^2 \|e_k\|^2\]
    Переходим в данном неравенстве к пределу и получаем желаемое
\end{proof}

\end{corollary}

\end{proof}

        0, k \ne n \\
        2\pi, k = n

    Таким образом, \(f \sim \frac{a_0}{2} + \sum_{k = 1}^\infty a_k\cos kx + b_k\sin kx = \sum_{k = -\infty}^\infty c_ke^{ikx}\)

\begin{theorem}
    Пусть \(\{e_k\}_{k = 1}^\infty\) --- ортогональная система и \(x \in V\). Тогда следующие условия эквивалентны:
    \begin{enumerate}
        \item \(\forall \epsilon > 0 \exists c_1, \dots c_n: \left\|x - \sum_{k = 1}^n c_ke_k\right\| < \epsilon\)
        \item \(x = \sum_{k = 1}^\infty \hat{x_k}e_k\)
        \item \(\|x\|^2 = \sum_{k = 1}^\infty |x_k|^2 \|e_k\|^2\) --- выполнено равенство Парсеваля
    \end{enumerate}
\end{theorem}
\begin{proof}
    \begin{enumerate}
        \item[\((1) \Ra (2)\)] Пусть \(\epsilon > 0\). По свойству минимальности коэффициентов Фурье \(\exists N: \left\| x - \sum_{k = 1}^N \hat{x_k}e_k \right\| < \epsilon\). Снова в силу минимальности \(\forall n \ge N\):
        \[\left\| x - \sum_{k = 1}^n \hat{x_k}e_k \right\| \le \left\| x - \sum_{k = 1}^N \hat{x_k}e_k \right\| < \epsilon\]
        Это доказывает \(x = \sum_{k = 1}^\infty \hat{x_k}e_k\)
        \item[\((2) \Ra (1)\)] очевидно
        \item[\((2) \Lra (3)\)] Следует из тождетва Бесселя:
        \[\left\| x - \sum_{k = 1}^n \widehat{x}_ke_k \right\|^2 = \|x\|^2 - \sum_{k = 1}^n |\widehat{x}_k|^2 \|e_k\|^2 \]
    \end{enumerate}
\end{proof}

\begin{definition}

\begin{theorem}[Рисс-Фишер]
    Пусть \(H\) --- гильбертово пространство, \(\{e_k\}_{k = 1}^\infty\) --- ортогональная система. Если. \(\alpha_k\) таковы, что \(\sum_{k = 1}^\infty |\alpha_k|\|e_k\|^2 < \infty\), то существует \(x \in H: x = \sum_{k = 1}^{\infty} \alpha_ke_k\)
\end{theorem}
\begin{proof}
    Положим:
    \[S_n = \sum_{k = 1}^n \alpha_k e_k, \sigma_n = \sum_{k = 1}^n |\alpha_k|^2 \|e_k\|^2 \Ra \forall n, p \in \N: S_{n + p} - S_n =  \sum_{k = n}^{n + p} \alpha_k e_k\]
    По теореме Пифагора получаем, что
    \[\|S_{n + p} - S_n\| = \sum_{k = n}^{n + p} |\alpha_k|^2 \|e_k\|^2 = |\sigma_{n + p} - \sigma_n|\]
    Получаем, что \(S_n\) --- фундаментальная, но тогда у нее есть предел.
\end{proof}

\begin{definition}
    Система \(\{e_k\}_{k = 1}^\infty\) в евклидовом пространстве \(V\) называется замкнутой, если \(\forall k \in \N: (x, e_k) = 0\), то \(x = 0\).
\end{definition}

\begin{theorem}
    Всякая полная ортогональная система в евклидовом пространстве замкнута. Если пространство гильбертово, то всякая замкнутая система полна.
\end{theorem}
\begin{proof}
    Пусть \(\{e_k\}_{k = 1}^\infty\) --- ортогональная система и \(x \in H\). Тогда по следствию из теоремы, получаем:
    \[x = \sum_{k = 1}^\infty \hat{x_k}e_k\]
    Если \((x, e_k) = 0 \forall k \Ra \hat{x_k} = 0 \forall k \Ra x = 0\). Пусть теперь \(\{e_k\}_{k = 1}^\infty\) --- ортогональная замкнутая система в гильбертовом пространстве. Докажем, что она полна. Заметим, что достаточно проверить, что \(\{e_k\}_{k = 1}^\infty\) --- базис. Пусть \(x \in H\). По неравенству Бесселя:
    \[\sum_{k = 1}^\infty |\hat{x_k}|^2\|e_k\|^2 < \infty \Ra \exists y \in H: y = \sum_{k = 1}^\infty \hat{x_k}e_k\]
    По следствию из предыдущей теоремы, получаем \(\hat{x_k} = \hat{y_k} \forall k \Ra \widehat{(y_k - x_k)} = 0 \forall k\), т.е. \(\forall k (y - x, e_k) = 0 \Ra\) из замкнутости \(y - x = \vec{0}\), т.е \(x = \sum_{k = 1}^\infty \hat{x_k}e_k\)
\end{proof}

\subsection{\(L_2\)-теория рядов Фурье}
\(L_2\) --- гильбертово пространство, а системы \(\left\{ \frac{1}{2}, \sin kx, \cos kx \right\}_{k = 1}^\infty, \{e^{ikx}\}_{k \in \Z}\) являются ортогональными базисами, причем 
\[f = \frac{a_0}{2} + \sum_{k = 1}^\infty a_k\cos kx + b_k\sin kx = \sum_{k = -\infty}^\infty c_ke^{ikx}\]

Равенство выполняется в смысле нормы \(L_2\): частичные суммы сходятся по норме. Запишем равенство Парсеваля:
\[\|f\|_2^2 = |a_0|^2\left( \frac{1}{2}, \frac{1}{2} \right) + \sum_{k = 1}^\infty |a_k|^2 (\cos kx, \cos kx) + |b_k|^2 (\sin kx, \sin kx) = \sum_{k = -\infty}^\infty c_k \left( e^{ikx}, e^{ikx} \right)\]
\[\frac{1}{\pi}\|f\|^2 = \frac{|a_0|^2}{2} + \sum_{k = 1}^\infty |a_k|^2 + |b_k|^2 = 2\sum_{k = -\infty}^\infty |c_k|^2\]

\begin{corollary}
    Если \(f, g \in L_2(T), c_k(f) = c_k(g)\), то \(f = g\) почти всюду
\end{corollary}

\begin{theorem}
    Пусть \(f \in C(T)\) и \(f'\) кусочно непрерывна. Тогда ряд Фурье \(S(f, x)\) сходится к функции \(f\) равномерно на \(\R\), причем \(\|S_n(f) - f\|_\infty \le \frac{1}{\sqrt{\pi n}}\|f'\|_2\)
\end{theorem}
\begin{proof}
    По следствию из признака Дини, \(f(x) = \sum_{k = -\infty}^\infty c_k(f)e^{ikx}\) и по теореме о дифференцировании рядя Фурье \(c_k(f') = ik c_k(f) \forall k \in \Z\). Тогда \(\forall x \in \R\) имеем:
    \[|S_n(f, x) - f(x)| = \left| \sum_{|k| \ge n + 1}^\infty c_k(f)e^{ikx} \right| \le \sum_{|k| \ge n + 1}^\infty |c_k(f)| = \sum_{|k| \ge n + 1}^\infty \frac{1}{|k|}|ikc_k(f)| = \sum_{|k| \ge n + 1}^\infty \frac{1}{|k|}|c_k(f')|\]
    Применяя КБШ, получаем:
    \[\left( \sum_{|k| \ge n + 1} \frac{1}{k^2} \right)^{\frac{1}{2}}\left( \sum_{|k| \ge n + 1} |c_k(f')|^2 \right)^{\frac{1}{2}}\]
    По неравеству Бесселя:
    \[\sum_{|k| \ge n + 1} |c_k(f')| \le \frac{1}{2\pi}\|f'\|_2^2\]
    Также:
    \[\sum_{k = n + 1}^\infty \frac{1}{k^2} \le \sum_{k = n + 1}^\infty \int_{k - 1}^k \frac{dx}{x^2} = \int_0^\infty \frac{dx}{x^2} = \frac{1}{n}\]
    Итак, \(\forall x \in \R\) справедлива оценка:
    \[|S_n(f, x) - f(x)| \le \left( \frac{2}{n} \right)^{\frac{1}{2}}\left( \frac{1}{2\pi} \right)^{\frac{1}{2}}\|f'\|_2\]
\end{proof}

\section{Интегралы с параметром}
Пусть \(E \subset \R^m\) измерима, \(A\) --- произвольное множество и пусть \(f: E \times A \ra \Cm\), такая, что \(\forall \alpha \in A: f(\cdot, \alpha) \in L_1(E)\). Рассмотрим функцию \(I(\alpha) = \int_E f(x, \alpha) dx, \alpha \in A\)

\begin{theorem}[О непрерывности интеграла с переменным пределом]
    Пусть \(E \subset \R^m\) измерима, \(A\) --- промежуток в \(\R, \alpha_0 \in A, f: E \times A \ra \Cm\) такая, что 
    \begin{enumerate}
        \item \(f(\cdot, \alpha)\) измерима на \(E \forall \alpha \in A\)
        \item \(f(x, \cdot)\) непрерывна в \(\alpha_0\) для почти всех \(x\)
        \item \(\exists \phi: E \ra \R\) --- интегрируема на \(E\), что для почти всех \(x \in E \forall \alpha \in A \left( |f(x, \alpha)| \le \phi(x) \right)\)
    \end{enumerate}
    Тогда \(I\) непрерывна в точке \(\alpha_0\)
\end{theorem}
\begin{proof}
    Из пунктов 1, 3 следует, что функция \(f(\cdot, \alpha)\) интегрируема на \(E\) для любого \(\alpha \Ra I\) определена. Возьмем \(\{\alpha_k\} \subset A, \alpha_k \ra \alpha_0\). Рассмотрим \(f_k = f(\cdot, \alpha_k), k \in \N_0\). По условию \(f_k(x) \ra f_0(x)\) и \(|f_k(x)| \le \phi\) для почти всех \(x\). Тогда по теореме Лебега о мажорируемой сходимости:
    \[I(\alpha_k) = \int_E f_k(x)dx \ra \int_E f_0(x)dx = I(\alpha_0)\]
    Тогда по определению предела по Гейне, получаем непрерывность функции \(I\) в точке \(\alpha_0\)
\end{proof}

\begin{corollary}
    Пусть \(E \subset \R^m\) --- компакт, \(A \subset \R\) --- промежуток. Если \(f \in C(E \times A)\), то \(I \in C(A)\)
\end{corollary}
\begin{proof}
    Зафиксируем \(\alpha_0 \in A\) и выберем \(\delta > 0\) так, что \(J = A \cap [\alpha_0 - \delta, \alpha_0 + \delta]\) является отрезком. По теореме Вейерштрасса, \(f\) ограничена на \(E \times J\), поэтому \(I\) непрерывна в точке \(\alpha_0\) (в качестве \(\phi\) берем \(\max_{E\times J} |f|\))
\end{proof}

\begin{example}
    Положим \(I(\alpha) = \int_0^{+\infty} \alpha e^{-\alpha x}dx\). Заметим, что \(I(0) = 0\), а при \(\alpha > 0: I(\alpha) = e^{-\alpha x}\big|_0^{+\infty} = 1\). Таким образом, условие 3 критично для выполнения утверждения теоремы.
\end{example}

\begin{theorem}[О дифференцируемости]
    Пусть \(E \subset \R^m\) --- измеримо, \(A \subset \R\) --- промежуток. Пусть \(f: E \times A \ra \Cm\), такая, что 
    \begin{enumerate}
        \item \(f(\cdot, \alpha)\) интегрируема на \(E \forall \alpha \in A\).
        \item \(f(x, \cdot)\) дифференцируема на \(A\) для почти всех \(x\)
        \item \(\exists \phi: E \ra \R: \left| \frac{\partial f}{\partial \alpha}(x, \alpha) \right| \le \phi(x)\) для п.в. \(x \in E\) и всех \(\alpha \in A\).
    \end{enumerate}
    Тогда функция \(I\) дифференцируема на \(A\) и \(I'(\alpha) = \int_E \frac{\partial f}{\partial \alpha}(x, \alpha) dx\)
\end{theorem}
\begin{proof}
    Зафиксируем \(\alpha_0 \in A\) и выберем последовательность \(\{\alpha_k\} \subset A: \alpha_k \ra \alpha_0\). Определим \(g_k(x) = \frac{f(x, \alpha_k) - f(x, \alpha_0)}{\alpha_k - \alpha_0}\). Тогда \(g_k(x) \ra \frac{\partial f}{\partial \alpha}(x, \alpha_0)\) для почти всех \(x \in E\). По теореме Лагранжа о среднем \(g_k(x) = \frac{\partial f}{\partial \alpha}(x, c_k)\) для некоторого \(c_k\) между \(\alpha_0, \alpha_k\). Поэтому, \(|g_k(x)| \le \phi(x)\) для почти всех \(x\). По теореме Лебега о мажорируемой сходимости, заключаем:
    \[\frac{I(\alpha_k) - I(\alpha_0)}{\alpha_k - \alpha_0} = \int_E \frac{f(x, \alpha_k) - f(x, \alpha_0)}{\alpha_k - \alpha_0} dx \ra_{k \ra \infty} \int_E \frac{\partial f}{\partial \alpha}(x, \alpha_0) dx\]
    Но тогда по определению предела по Гейне, получаем:
    \[I'(\alpha_0) = \lim_{\alpha \ra \alpha_0} \frac{I(\alpha) - I(\alpha_0)}{\alpha - \alpha_0} = \int_E \frac{\partial f}{\partial \alpha}(x, \alpha_0) dx\]
\end{proof}

\begin{corollary}
    Пусть \(E \subset \R^m\) --- компакт, \(A \subset \R\) --- промежуток. Если \(f, \frac{\partial f}{\partial \alpha} \in C(E \times A)\), то \(I \in C^1(A), I'(\alpha) = \int_E \frac{\partial f}{\partial \alpha}(x, \alpha_0) dx\)
\end{corollary}

\begin{example}
    Рассмотрим \(I(\alpha) = \int_0^{+\infty} \alpha^2e^{-\alpha x} dx\)
\end{example}

\begin{note}
    Из доказательства теорем выше, можно видеть, что можно поменять местами кванторы ''для п.в. \(x \in E\), \(\forall \alpha \in A\)'', т.е. они будут верны и для формулировок ''\(\forall \alpha \in A\), для п.в. \(x \in E\)''
\end{note}

\subsection{Равномерная сходимость несобственного интеграла с переменным пределом}
\begin{definition}
    Пусть \(-\infty < a < b \le +\infty\), \(A\) --- множество и задана функция \(f: [a, b) \times A \ra \Cm\), т.ч. \(\forall \alpha \in A: f(\cdot, \alpha) \in L_1([a, b'])\) для всех \(b' \in (a, b)\). Пусть \(\forall \alpha \in A\) существует:
    \[\int_a^b f(x, \alpha) dx = \lim_{y \ra b-0} \int^y_a f(x, \alpha) dx\]
    и если предел конечен (лежит в \(\Cm\)), то интеграл \(\int_a^b f(x, \alpha)\) называется сходящимся. Аналогично определяется несобственный интеграл по промежутку \((a, b]\).
\end{definition}

\begin{definition}
    Говорят, что \(\int_a^b f(x, \alpha) dx\) сходится равномерно на \(A\) по \(\alpha \in A\), если интеграл сходится \(\forall \alpha \in A\) и \(\forall \epsilon > 0 \exists b' \in [a, b) \forall y \in (b', b) \forall \alpha \in A \left| \int_a^y f(x, \alpha) dx - \int_a^b f(x, \alpha) dx \right| < \epsilon\), что эквивалентно \(\sup_{\alpha \in A} \left| \int_y^b f(x, \alpha) dx \right| \ra_{y \ra b - 0} 0\)
\end{definition}

\begin{note}
    Определение равномерной сходимости имеет место, если \(\{f(\cdot, \alpha)\}_{\alpha \in A}\) допускает интегрируемую мажоранту, т.е. \(\phi: [a, b) \to \R\) интегрируема. \(\forall x \in [a, b) \ \forall \alpha \in A: |f(x, \alpha)| \le \phi(x)\) или \(\sup_{\alpha \in A} \left| \int_y^b f(x, \alpha) dx \right| \le \int_y^b \phi(x)dx \to_{y \to b-0} 0\).  
    Тогда несобственный интеграл \(\int_a^b f(x,\alpha)dx\) сходится равномерно по \(\alpha \in A\).
\end{note}

\begin{note}
    Пусть \(\int_a^b f(x, \alpha) dx\) сходится равномерно на \(A\), \(\{y_n\} \subset (a, b), y_n \ra b\) и \(g_n(\alpha) = \int_a^{y_n} f(x, \alpha) dx \rightrightarrows_A I(\alpha) = \int_a^b f(x, \alpha) dx\)
\end{note}
\begin{proof}
    \(\forall \epsilon > 0 \exists N: \forall n \ge N (y_n \in (b', b))\), а значит, \(|g_n(x) - I(\alpha)| < \epsilon\) на \(A\).
\end{proof}

\begin{theorem}
    Пусть \(f\) непрерывна на \([a, b) \times A\) и \(\int_a^b f(x, \alpha)dx\) сходится равномерно на \(A\). Тогда функция \(I\) непрерывна на \(A\).
\end{theorem}
\begin{proof}
    Рассмотрим \(\{y_n\} \subset (a, b), y_n \ra b\) и определим \(g_n(\alpha) = \int_a^{y_n} f(x, \alpha) dx\). По замечанию, \(y_n \rightrightarrows I(\alpha)\). По следствию из теоремы, \(g_n\) непрерывна на \(A\). Тогда \(I\) непрерывна на \(A\) как равномерный предел непрерывных функций.
\end{proof}

\begin{theorem}
    Пусть \(f\) интегрируема на \([a, y] \times A \forall y \in (a, b), \mu(A) < \infty\) и \(\int_a^b f(x, \alpha) dx\) сходится равномерно на \(A\). Тогда
    \[\int_A I(\alpha) d\alpha \int_a^b \left( \int_A f(x, \alpha) dx \right) d\alpha\;\;\;(*)\]
\end{theorem}
\begin{proof}
    \[I(\alpha, y) = \int_a^y f(x, \alpha) dx, y \in (a, b)\]
    По теореме Фубини
    \[\int_A I(\alpha, y) d\alpha = \iint_{[a, y] \times A} f(x, \alpha )dxdy < \infty\]
    Следовательно \(I(\cdot, y)\) интегрируема на \(A\). Из условия равномерной сходимости, \(\exists b' \in [a, b) \forall y \in (b', b): |I(\alpha, y) - I(\alpha)| < 1\) на \(A\). Следовательно, \(I(\cdot, y) - I\) интегрируема на \(A\), а значит, и \(I\) интегрируема на \(A\). Далее,
    \[\left| \int_A I(\alpha, y) d\alpha - \int_A I(\alpha) d\alpha \right| \le \int_A |I(\alpha, y) - I(\alpha)| d\alpha \le \mu(A) \sup_{\alpha \in A} |I(\alpha, y) - I(\alpha)| \ra_{y \ra b - 0} 0\]
    Снова по теореме Фубини:
    \[\int_A I(\alpha, y) d\alpha = \int_a^y \left( \int_A f(x, \alpha) d\alpha \right) dx\]
    В последнем равенстве перейдем к пределу при \(y \ra b - 0\)
\end{proof}

\begin{corollary}
    Если \(f \in C([a, b] \times [c, d])\), то \((*)\) выполнеяется при условии равномерной сходимости интеграла с параметром.
\end{corollary}

\begin{theorem}
    Пусть \(f, \frac{\partial f}{\partial \alpha}\) непрерывна на \([a, b) \times A, \int_a^b \frac{\partial f}{\partial \alpha} dx\) сходится равномерно на \(A\) и \(\exists \alpha_0 \in A: \int_a^b f(x, \alpha_0)dx\) сходится. Тогда \(I\) непрерывно дифференцируема на \(A\) и \(I'(\alpha) = \int_a^b \frac{\partial f}{\partial \alpha} (x, \alpha) dx\)
\end{theorem}
\begin{proof}
    \(J(\alpha) = \int_a^b \frac{\partial f}{\partial \alpha}(x, \alpha)\) непрерывна. Применим предыдущую теорему к \(J\) на отрезке с концами \([\alpha, \alpha_0]\)
    \[\int_\alpha^{\alpha_0} J(y) dy = \int_a^b\left( \int_{\alpha_0}^\alpha f(x, \alpha)d\alpha \right) dx = \int_a^b ((f(x, \alpha) - f(x, \alpha_0)))dx = I(\alpha) - I(\alpha_0)\]
    Следовательно, \(I(\alpha) = \int_{\alpha_0}^{\alpha} J(y)dy + I(\alpha_0)\). Поэтому \(I \in C^1(A)\) и \(I'(\alpha) = J(\alpha)\)
\end{proof}

\begin{theorem}[Критерий Коши]
    \(\int_a^b f(x, \alpha)dx\) сходится равномерно на \(A\) тогда и только тогда, когда:
    \[\forall \epsilon > 0 \exists b_{\epsilon} \in [a, b) \forall \xi, \eta \in (b_\epsilon, b) \forall \alpha \in A \left( \left| \int_\xi^\eta f(x, \alpha) dx \right| < \epsilon \right)\quad(*)\]
\end{theorem}
\begin{proof}
    Пусть \(\epsilon > 0\). Из условия равномерной сходимости:
    \[\exists b_\epsilon \in [a, b)  \forall y \in (b_\epsilon, b) \forall \alpha \in A \left( \left| \int_y^b f(x, \alpha) dx \right| < \frac{\epsilon}{2} \right)\]
    Тогда для любого \([\xi, \eta] \subset (b_\epsilon, b) \forall \alpha \in A\):
    \[\left| \int_\xi^\eta f(x, \alpha) dx \right| = \left| \int_\xi^b f(x, \alpha) dx - \int_\eta^b f(x, \alpha) dx \right| \le \left| \int_\xi^b f(x, \alpha) dx \right| + \left|\int_\eta^b f(x, \alpha) dx \right| \le \frac{\epsilon}{2} + \frac{\epsilon}{2} = \epsilon\]

    В другую сторону. Пусть выполнено условие \((*)\). Заметим, что для каждого фиксированного \(\alpha \in A\) условие \((*)\) влечёт сходимость несобственного интеграла \(\int_a^b f(x,\alpha)dx\) по критерию Коши для несобственных интегралов (без параметра). Действительно, для любого \(\epsilon>0\) существует \(b_\epsilon\in[a,b)\) такое, что для любых \(\xi,\eta\in(b_\epsilon,b)\) выполнено \(\left|\int_\xi^\eta f(x,\alpha)dx\right|<\epsilon\). Это означает, что функция \(F_\alpha(y)=\int_a^y f(x,\alpha)dx\) имеет предел при \(y\to b-\). Обозначим этот предел через \(I(\alpha)=\int_a^b f(x,\alpha)dx\).

    Покажем, что сходимость равномерна. Зафиксируем \(\epsilon>0\). По условию \((*)\) найдётся \(b_\epsilon\in[a,b)\) такое, что для всех \(\xi,\eta\in(b_\epsilon,b)\) и всех \(\alpha\in A\) справедливо \(\left|\int_\xi^\eta f(x,\alpha)dx\right|<\epsilon\). Возьмём произвольное \(y\in(b_\epsilon,b)\). Для любого \(\eta\in(b_\epsilon,b)\) имеем \(\left|\int_y^\eta f(x,\alpha)dx\right|<\epsilon\). Устремим \(\eta\to b-\). Поскольку для каждого \(\alpha\) интеграл \(\int_y^\eta f\) сходится к \(\int_y^b f\), и неравенство \(\left|\int_y^\eta f\right|<\epsilon\) сохраняется в пределе (модуль непрерывен), получаем
    \[
    \left|\int_y^b f(x,\alpha)dx\right| \le \epsilon \qquad \forall \alpha\in A,\ \forall y\in(b_\epsilon,b).
    \]
    Если в условии \((*)\) взять \(\epsilon/2\) вместо \(\epsilon\), то получим оценку \(<\epsilon\). Следовательно,
    \[
    \sup_{\alpha\in A}\left|\int_y^b f(x,\alpha)dx\right| \to 0 \quad\text{при}\quad y\to b-,
    \]
    что и означает равномерную сходимость интеграла на \(A\).
\end{proof}

\begin{theorem}[Признак Дирихле]
    Пусть на \([a, b) \times A\) заданы функции \(f, g\) такие, что
    \begin{enumerate}
        \item \(F(x, \alpha) = \int_a^xf(t, \alpha)dt\) ограничен на \([a, b) \times A\).
        \item \(g(\cdot, \alpha)\) монотонна на \([a, b)\) при любом \(\alpha \in A\).
        \item \(\sup_{\alpha \in A}|g(x, \alpha)| \ra_{x \ra b - 0} 0\)
    \end{enumerate}
    Тогда \(J(\alpha) = \int_a^b f(x, \alpha)g(x, \alpha)dx\) сходится равномерно на \(A\).
\end{theorem}
\begin{proof}
    Зафиксируем \(\epsilon > 0\). Из условия 1 \(\exists C > 0: \forall x \in [a, b) \forall \alpha \in A (|F(x, \alpha)| \le C)\). Из условия 3, \(\exists b_\epsilon \in [a, b) \forall x \in (b_\epsilon, b) \forall \alpha \in A \left( |g(x, \alpha)| \le \frac{\epsilon}{8 C} \right)\). Зафиксиурем \(\alpha \in A\). Пусть теперь \([\xi, \eta] \subset (b_\epsilon, b)\). Имеем \(\left| \int_\xi^x f(t, \alpha) \right| = |F(x, \alpha) - F(\xi, \alpha)|\) и по лемме Абеля:
    \[\left| \int_\xi^\eta f(x, \alpha)g(x, \alpha)dx \right| \le 4C(|g(\xi, \alpha)| + |g(\eta, \alpha)|) < 4C\left( \frac{\epsilon}{8C} + \frac{\epsilon}{8C} \right) = \epsilon\]
    И по критерию Коши равномерной сходимости получаем желамое.
\end{proof}

\begin{example}
    Рассмотрим интеграл:
    \[\int_0^{+\infty}\frac{\sin x}{x}dx, \alpha \ge 0\]
    Рассмотрми семейство \(I(\alpha) = \int_0^{+\infty} \frac{\sin x}{x}e^{-\alpha x}dx\). Заметим, что: \(\left| \int_0^x \sin t dt \right| \le 2, \left| \frac{e^{-\alpha x}}{x} \right| \le \frac{1}{x} \ra_{x \ra +\infty} 0\) и функция \(\frac{e^{-\alpha x}}{x}\) монотонна. Тогда выполнены условия признака Дирихле, т.е. интеграл сходится равномерно (чтобы избежать проблем в точке 0, достаточно рассмотреть интеграл на луче \([1, +\infty)\)). Кроме того, \(f(x, \alpha) = \frac{\sin x}{x}e^{-\alpha x}\) непрерывна на \([0, +\infty)^2\), следовательно, по теореме о непрерывности интеграла с параметром, \(I(\alpha)\) непрерывна на \([0, +\infty)\). Докажем теперь дифференцируемость данной функции в каждой точке \(\alpha\). Заметим, что \(\frac{\partial f}{\partial \alpha} = -e{-\alpha x} \sin x\), следовательно, \(f\) дифференцируема. Возьмем \(0 < c < \alpha < d < +\infty\). На отрезке \([c, d]\) выполнено: \(\left|\frac{\partial f}{\partial \alpha}(x, \alpha)\right| \le e{-cx} \) --- интегрируемая мажоранта. Таким образом, по теореме о дифференцировании интеграла с параметром, \(I'(\alpha) = -\int_0^{+\infty}e^{-\alpha x}\sin x dx\). Заметим, что:
    \[I'(\alpha) = -\int_0^{+\infty}e^{-\alpha x}\sin xdx = \int_0^{+\infty}e^{-\alpha x}d(\cos x) = e^{-\alpha x}\cos x\big|_{0}^{+\infty} + \alpha \int_0^{+\infty}e^{-\alpha x} \cos xdx =\]
    \[= -1 + \alpha \int_0^{+\infty}e^{-\alpha x}d(\sin x) = -1 + \alpha \left( \underbrace{e^{-\alpha x}\sin x\big|_0^{+\infty}}_{0} + \alpha \int_0^{+\infty}e^{-\alpha x}\sin x dx \right)\]
    Получаем \(I'(\alpha) = -1 - \alpha^2I'(\alpha) \Ra I'(\alpha) = \frac{1}{1 + \alpha^2}\) на \(\forall [c, d] \subset (0, +\infty)\Ra \) равенство выполнено на всем луче. Тогда \(I(\alpha) = -\arctg \alpha + C\). Заметим, что \(|I(\alpha)| \le \int_0^{+\infty} e^{-\alpha x}dx = \frac{1}{\alpha} \ra_{\alpha \ra +\infty} 0\), поэтому \(C = \frac{\pi}{2}\). Итак,
    \[I(\alpha) = \frac{\pi}{2} - \arctg \alpha, \alpha \ge 0.\]
    В частности,
    \[\int_0^{+\infty} \frac{\sin x}{x} dx = I(0) = \frac{\pi}{2}\]
\end{example}

\begin{definition}
    Функция
    \[\Gamma(x) = \int_0^{+\infty} t^{x - 1}e^{-t}dt\]
    Называется гамма-функцией Эйлера
\end{definition}

\begin{note}
    \(\Gamma\) --- несобственный интеграл с особенностью в \(+\infty\) и \(0\) (при \(t < 1\)). Заметим, что \(f = t^{x - t}e^{-t} \sim t^{x - 1}\) при \(t \ra +0\), и тогда \(\int_0^{\frac{1}{2}} f(t)dt\) сходится \(\Lra x > 0\). Также, \(f = t^{x - t}e^{-t/2}e^{-t/2} = o(e^{-t/2}), t \ra +\infty \Ra \int_0^{\frac{1}{2}} f(t)dt\) сходится \(\forall x > 0\). Таким образом, \(\Gamma(x)\) определена при \(x > 0\).
\end{note}

\begin{lemma}
    \(\Gamma(x + 1) = x\Gamma(x)\) при \(x > 0\). Кроме того, \(\Gamma(n + 1) = n!\).
\end{lemma}
\begin{proof}
    Интегрирование по частям дает:
    \[\Gamma(x + 1) = \int_0^{+\infty}t^xe^{-t}dt = \underbrace{-t^{x}e^{-t}\big|_0^{+\infty}}_{0} + x\int_0^{+\infty}t^{x - 1}e^{-t}dt = x\Gamma(x)\]
    Формула для \(\Gamma(n + 1) = n!\) получается из индукции по \(n\) с базой \(\Gamma(1) = \int_0^{+\infty} e^{-t}dt = 1 = 0!\).
\end{proof}

\begin{lemma}
    \(\Gamma \in C^{\infty}(0, +\infty)\) и \(\Gamma^{(k)}(x) = \int_0^{+\infty}t^{x - 1}e^{-t}\ln^ktdt\) для \(k \in \N_0\).
\end{lemma}
\begin{proof}
    Для \(k = 0\) утверждение выполнено (по определению \(\Gamma\)). Проверим переход индукции. Для доказательства дифференцируемости, рассмотрим произвольный \(0 < c < d < +\infty\) и докажем дифференцируемость на нем.
    \[\frac{\partial}{\partial x}\left( t^{x - 1}e^{-t}\ln^kt \right) = t^{x - 1}e^{-t}\ln^{k + 1}t\]
    Тогда:
    \[|t^{x - 1}e^{-t}\ln^{k + 1}t| \le t^{c - 1}e^{-t}|\ln t|^{k + 1}, t \in (0, 1)\]
    \[|t^{x - 1}e^{-t}\ln^{k + 1}t| \le t^{d - 1}e^{-t}\ln^{k + 1} t, t \in [1, +\infty)\]
    Следовательно, по теореме о дифференцировании интеграла с параметром на любом отрезке \([c, d]\):
    \[\Gamma^{(k + 1)}(x) = \int_0^{+\infty}t^{x - 1}e^{-t}\ln^{k + 1}tdt\]
    Но тогда равенство выполнено и на всем луче \((0, +\infty)\).
\end{proof}

\begin{corollary}[Формула Эйлера-Гаусса]
    \[\Gamma(x) = \lim_{n \ra \infty} \frac{n^x\cdot n!}{x(x + 1)\dots(x + n)}, x > 0\]
\end{corollary}
\begin{proof}
    Заменой переменных \(t \leadsto zt\), получаем:
    \[\int_0^1 z^{x - 1}(1 - z)^ndz = B(x, n + 1) = \frac{\Gamma(x)\Gamma(n + 1)}{\Gamma(x + n - 1)} = \frac{\Gamma(x)\cdot n!}{\Gamma(x)x(x + 1)\dots (x + n)} = \frac{n!}{x(x + 1)\dots (x + n)}\]
    Возьемем \(g_n(t) = \left\{\begin{array}{l}
        \left( 1 - \frac{t}{n} \right)^n, 0 \le t < n \\
        0, t \ge n
    \end{array}\right.\)
    Имеем:
    \[\frac{n^x\, n!}{x(x+1)\dots(x+n)} = \int_0^n t^{x-1}\left(1-\frac{t}{n}\right)^n dt = \int_0^{\infty} t^{x-1} g_n(t)\,dt,\]
    Для каждого фиксированного \(t\ge0\) при \(n\to\infty\) имеем:
    \[g_n(t) \to e^{-t},\]
    так как известно, что \(\left(1-\frac{t}{n}\right)^n \to e^{-t}\). Кроме того, для всех \(n\) и \(t\) выполняется неравенство:
    \[0 \le g_n(t) \le e^{-t}.\]
    Действительно, для \(0\le t<n\) известно, что \(\left(1-\frac{t}{n}\right)^n \le e^{-t}\) (это следует из выпуклости экспоненты), а при \(t\ge n\) \(g_n(t)=0\le e^{-t}\). 

    Функция \(t^{x-1}e^{-t}\) интегрируема на \([0,\infty)\) при \(x>0\), так как \(\int_0^{\infty}t^{x-1}e^{-t}dt = \Gamma(x)<\infty\). Таким образом, по теореме Лебега о мажорируемой сходимости:
    \[\lim_{n\to\infty}\int_0^{\infty} t^{x-1} g_n(t)\,dt = \int_0^{\infty} t^{x-1} e^{-t}\,dt = \Gamma(x).\]

    Следовательно,
    \[\Gamma(x) = \lim_{n\to\infty} \frac{n^x\, n!}{x(x+1)\dots(x+n)},\quad x>0,\]
    что и требовалось доказать.
\end{proof}

\begin{lemma}
    \(\sin x = \prod_{k = 1}^\infty \left( 1 - \frac{x^2}{k^2\pi^2} \right), x \in (-\pi, \pi)\), где \(\prod_{n = 1}^\infty a_n = \lim_{N \ra \infty}\prod_{n = 1}^N a_n\).
\end{lemma}
\begin{proof}
    Положим \(f(x) = \cos \alpha x\). Разложим \(f\) в ряд Фурье. Для \(\alpha \notin \Z\), имеем:
    \[a_0 = \frac{2}{\pi}\int_0^{\pi} \cos\alpha x dx = \frac{2\sin \pi\alpha}{\pi\alpha}\]
    \[a_n = \frac{2}{\pi}\int_0^{\pi} \cos\alpha x \cos nx dx = \frac{1}{\pi}\int_0^{\pi} \big(\cos(\alpha + n) x + \cos(\alpha - n) x\big) dx =\]
    \[= \frac{1}{\pi}\left( \frac{\sin(\alpha + n)\pi}{\alpha + n} + \frac{\sin(\alpha - n)\pi}{\alpha - n} \right) = \frac{(-1)^n2\alpha\pi\alpha}{\pi(\alpha^2 - n^2)}\]
    Итак:
    \[\cos\alpha x = \frac{2\sin \pi\alpha}{\pi\alpha} + \sum_{n = 1}^\infty \frac{(-1)^n2\alpha\sin\pi\alpha}{\pi(\alpha^2 - n^2)}, x \in [-\pi, \pi]\]
    Подставим \(x = \pi\):
    \[\ctg \pi\alpha = \frac{1}{\pi\alpha} + \sum_{n = 1}^\infty \frac{2\alpha\pi}{\pi^2(\alpha^2 - n^2)}, x = \pi\alpha \ne \pi k, k \in \Z\]
    Заключаем, что:
    \[\ctg x - \frac{1}{x} = \sum_{n = 1}^\infty \frac{2x}{x^2 - \pi^2n^2}\]
    Теперь рассмотрим \(\phi(x) = \ctg x - \frac{1}{x}, \phi(0) = 0\). Это непрерывная на \((-\pi, \pi)\) функция. Ряд в правой части сходится равномерно на любом подотрезке \((-\pi, \pi)\). Тогда:
    \[\int_0^x \phi(t)dt = \ln\left| \frac{\sin x}{x} \right| = \sum_{n = 1}^\infty \ln\left( 1 - \frac{x^2}{\pi^2n^2} \right), x \in (-\pi, \pi)\]
    Распишем равенство посередине:
    \[\frac{\sin x}{x} = \lim_{N \ra \infty}\sum_{n = 1}^{N}\left( 1 - \frac{x^2}{\pi^2n^2} \right)\]
\end{proof}

\section{Преобразование Фурье}
\begin{definition}
    Пусть \(f \in L_1(\R)\). Преобразованием Фурье функции \(f\) называется функция \(\widehat{f}\), определяемая как:
    \[\widehat{f}(y) = \frac{1}{\sqrt{2\pi}}\int_\R f(x)e^{-iyx}dx, y \in \R\]
\end{definition}

\begin{note}
    Интеграл в правой части существует, т.к. \(|f(x)e^{ixy}| = |f(x)| \in L_1\).
\end{note}

\begin{example}
    Найдем \(\widehat{I_{[-1, 1]}}\):
    \[\widehat{I_{[-1, 1]}} = \frac{1}{\sqrt{2\pi}}\int_{-1}^1 e^{-iyx}dx = \frac{2}{\sqrt{2\pi}}\int_0^1\cos yx dx = \left.\frac{2}{\sqrt{2\pi}}\frac{\sin yx}{y}\right|_0^1 = \sqrt{\frac{2}{\pi}}\frac{\sin y}{y}, y \ne 0\]
    В силу того, что функции из \(L_1\) определены с точностью до множества меры \(0\), можно заключить:
    \[\widehat{I_{[-1, 1]}}(y) = \sqrt{\frac{2}{\pi}}\]
\end{example}

\begin{proposition}[Свойства преобразования Фурье]
    Пусть \(f, g \in L_1(\R), \alpha, \beth \in \Cm\). Тогда:
    \begin{enumerate}
        \item \(\widehat{\alpha f + \beta g} = \alpha\widehat{f} + \beta\widehat{g}\)
        \item \(\widehat{f}\) ограничена на \(\R\)
        \item Если \(f_h(x) = f(x - h), \delta_tf(x) = f(tx)\), то \(\widehat{f_h}(y) = e^{-ihy}\widehat{f}(y), \widehat{\delta_tf}(y) = \frac{1}{t}\widehat{f}\left( \frac{y}{t} \right)\)
        \item \(\widehat{f}\) непрерывна на \(\R\) и \(\widehat{f}(y) \ra 0, y \ra \pm\infty\)
        \item \(\widehat{f * g} = \sqrt{2\pi}\widehat{f}\cdot\widehat{g}\)
    \end{enumerate}
\end{proposition}
\begin{proof}
    \begin{enumerate}
        \item Вытекает из линейности интеграла Лебега
        \item \[|\widehat{f}(y)| \le \frac{1}{\sqrt{2\pi}}\int_\R |f(x)e^{-iyx}|dx = \frac{1}{\sqrt{2\pi}}\int_\R |f|dx = \frac{1}{\sqrt{2\pi}}\|f\|_1\]
        \item Вытекает из теоремы о замене переменной в интеграле Лебега
        \item \[|\widehat{f}(y + h) - \widehat{f}(y)| \le \frac{1}{\sqrt{2\pi}}\int_\R |f(x)e^{-ix(y + h)} - f(x)e^{-ixy}|dx = \frac{1}{\sqrt{2\pi}}\int_\R |f(x)||1 - e^{ihx}|dx\]
        Заметим, что \(|f(x)||1 - e^{ihx}| \le 2|f(x)|\) --- интегрируемая мажоранта. Поэтому по теореме о непрерывности интеграла с параметром, заключаем, что \(|\widehat{f}(y + h) - \widehat{f}(y)| \ra 0\), причем стремление не зависит от точки \(y\). Поэтому на самом деле \(\widehat{f}(y)\) равномерно непрерывна. Оставшееся утверждение вытекает из леммы Римана об осцилляции

        \lecture{12}

        \item Для \(\forall y \in \R: (x, t) \mapsto f(x - t)g(t)e^{-iyx}\) --- интегрируема на \(\R^2\), т.к. \(|f(x - t)g(t)e^{-iyx}| = |f(x - t)g(t)|\). Тогда по теореме Фубини:
        \[\sqrt{2\pi}\widehat{f * g} = \int_\R\left( \int_\R f(x - t)g(t)dt \right)e^{-iyx}dx = \int_\R\left( \int_\R f(x - t)e^{-iy(x - t)}dx \right)g(t)e^{-iyt}dt = 2\pi\widehat{f}\cdot\widehat{g}\]

    \end{enumerate}
\end{proof}

\begin{note}
    Отметим, что под преобразованием Фурье также понимают отображение \(F: L_1(\R) \ra C_0(\R), F[f] = \widehat{f}\), где \(C_0\) --- пространство непрерывных функций, стремящихся к \(0\) при \(x \ra \pm\infty\) с \(\sup\)-нормой.
\end{note}

\begin{problem}
    Пусть \(f_k, f \in L_1(\R), f_k \ra f\) в \(L_1\). Докажите, что \(\widehat{f}_k \rightrightarrows \widehat{f}\) на \(\R\)
\end{problem}

\begin{theorem}[Преобразование Фурье производной]
    Пусть \(f \in C^1(\R), f, f' \in L_1(\R)\). Тогда \(\widehat{f'}(y) = iy\widehat{f}(y)\) для всех \(y \in \R\).
\end{theorem}
\begin{proof}
    Покажем, что \(\lim_{x \ra \pm\infty} f(x) = 0\). По Формуле Ньютона-Лейбница: \(f(x) - f(0) = \int_0^x f'(t)dt \Ra\) существует конечный предел \(\lim_{x \ra \pm\infty} f(x)\). Если он отличен от нуля, то \(|f| \ge c > 0\) на некотором луче и тогда \(f \notin L_1(\R)\). Теперь по формуле интегрирования по частям, получаем:
    \[\sqrt{2\pi}\widehat{f'}(y) = \int_\R f'(x)d^{-iyx}dx = \underbrace{f(x)e^{-iyx}\big|_{-\infty}^{+\infty}}_0 - (-iy)\int_\R f(x) e^{-iyx}dx = \sqrt{2\pi} (iy)\widehat{f'}(y)\]
\end{proof}

\begin{note}
    Данная теорема справедлива если \(f \in L_1(\R)\) и \(f(x) = \int_{-\infty}^x \phi(x)dx\) для всех \(\phi \in L_1(\R)\). В частности, это выполняется, когда \(f\) --- непрерывна, а \(f'\) --- кусочно-непрерывна.
\end{note}

\begin{corollary}
    Пусть \(f \in C^n(\R) f, f', \dots f^{(n)} \in L_1(\R)\). Тогда \(\widehat{f^{(n)}}(y) = (iy)^n\widehat{f}(y)\) и \(\widehat{f}(y) = o\left( \frac{1}{y^n} \right)\) при \(y \ra \pm\infty\)
\end{corollary}

\begin{theorem}[Производная преобразования Фурье]
    Пусть \(f \in L_1(\R)\) и \(x \mapsto xf(x) \in L_1(\R)\). Тогда \(\widehat{f} \in C^1(\R)\) и \(\frac{d}{dy}\widehat{f} = F[-ixf(x)]\).
\end{theorem}
\begin{proof}
    Т.к. \(|-ix f(x)e^{-iyx}| = |xf(x)|\), то по признаку Лейбница дифференцирования интеграла с параметром, получаем:
    \[\sqrt{2\pi}\frac{d}{dy}\widehat{f}(y) = \int_\R \frac{\partial}{\partial y}f(x)e^{-iyx}dx = \int_\R -ixf(x)e^{-iyx}dx\]
\end{proof}

\begin{corollary}
    Пусть \(f \in L_1(\R), x \mapsto x^nf(x) \in L_1(\R)\) для некоторого \(n \in \N\). Тогда \(\widehat{f} \in C^n(\R)\).
\end{corollary}
\begin{proof}
    Т.к. \(|x|^k \le 1 + |x|^n \Ra x^kf(x)\) будут интегрируемы при \(\forall k \le n\). Осталось применить индукцию и предыдущую теорему.
\end{proof}

\begin{example}
    Найдем преобразование Фурье для \(f(x) = e^{-\frac{x^2}{2}}\)
\end{example}
\begin{solution}
    Заметим, что \(f'(x) = -xf(x)\). Преобразуем:
    \[if'(y) = -xif(x)\]
    \[i(iy)\widehat{f}(y) = \left(\widehat{f}\right)'(y)\]
    \[iy\widehat{f}(y) = -i\left(\widehat{f}\right)'(y)\]
    Таким образом, \(\widehat{f}\) удовлетворяет дифференциальному уравнению \(z'(y) = -yz(y) \Lra z(y) = Ce^{-\frac{y^2}{2}}\). Найдем \(C\):
    \[C = \widehat{f}(0) = \frac{1}{\sqrt{\pi}}\int_\R e^{-\left( \frac{x}{\sqrt{2}} \right)^2}d\left( \frac{x}{\sqrt{2}} \right) = 1\]
    Таким образом \(\widehat{f}(y) = e^{-\frac{y^2}{2}}\)
\end{solution}

\subsection{Интеграл Фурье}

Займемся вопросом восстановления функции по ее преобразованию Фурье.

\begin{definition}
    Пусть \(f\) интегрируема на \(\forall [a, b] \subset \R\) (локально интегрируема на \(\R\)). Тогда
    \[\text{v.p.}\int_{-\infty}^{+\infty}f(x)dx = \lim_{u \ra +\infty} \int_{-u}^u f(x)dx\]
\end{definition}

\begin{note}
    Если \(f \in L_1(\R)\), то \(\text{v.p.}\int_{-\infty}^{+\infty}f(x)dx = \int_{-\infty}^{+\infty}f(x)dx\) из теоремы Лебега о мажорируемой сходимости.
\end{note}

\begin{definition}
    Пусть \(f\) интегрируема, \(x \in \R\). Тогда интегралом Фурье \(f\) в точке \(x\) называется:
    \[I(f, x) = \frac{1}{\sqrt{2\pi}}\text{v.p.}\int_{-\infty}^{+\infty}\widehat{f}(y)e^{iyx}dy\]
\end{definition}

\begin{note}
    Можно построить следующее соответствие:
    \begin{center}
        \begin{tabular}{c|c} 
            Периодический случай & Непериодический случай \\ 
            \hline
            Коэффициенты Фурье & Преобразование Фурье \\ 
            Ряд Фурье & Интеграл Фурье \\ 
        \end{tabular}
    \end{center}
\end{note}

Заметим, что:
\[I(f, x) = \int_0^{+\infty} \left( \widehat{f}(y)e^{iyx} + \widehat{f}(-y)e^{-iyx} \right)dy\]
При этом:
\[\widehat{f}(y)e^{iyx} + \widehat{f}(-y)e^{-iyx} = \frac{1}{\sqrt{2\pi}}\int_\R f(t) (\underbrace{e^{iy(x - t)} + e^{-iy(x - t)}}_{2\cos y(x - t)})dy\]
Следовательно:
\[I(f, x) = \int_0^{+\infty} (a(y)\cos yx + b(y)\sin yx)dy\]
Где:
\[a(y) = \frac{1}{\pi} \int_\R f(t)\cos yt dt, b(y) = \frac{1}{\pi} \int_\R f(t)\sin yt dt\]

Запишем в удобном виде \(I_u(x) = \frac{1}{\sqrt{2\pi}}\int_{-u}^u \widehat{f}(y)e^{iyx}dy\)
\begin{lemma}
    Пусть \(f \in L_1(\R), x \in \R\). Тогда \(\forall y \in \R\) выполнено:
    \[I_u(f, x) = \frac{1}{\pi}\int_\R f(x - t)\frac{\sin ut}{t}dt\]
\end{lemma}
\begin{proof}
    Запишем \(\widehat{f}\) в виде интеграла. Имеем:
    \[I_u(f, x) = \frac{1}{2\pi}\int_{-u}^u \left( \int_\R f(s)e^{iy(x - s)}ds \right)dy\]
    Функция \((y, s) \mapsto f(s)e^{iy(x - s)}\) интегрируема на \([-u, u] \times \R\). Тогда по теореме Фубини:
    \[I_u(f, x) = \frac{1}{2\pi}\int_\R f(s)\left( \int_{-u}^u e^{-iy(x - s)}dy \right)ds =\]
    \[=\frac{1}{2\pi}\int_\R f(s) \frac{e^{iu(x - s)} - e^{-iu(x - s)}}{i(x - s)}ds = \frac{1}{2\pi}\int_\R f(s) \frac{\sin u(x - s)}{x - s}ds\]
    Заменив переменную, получаем желаемое.
\end{proof}

\begin{corollary}
    \(\forall \delta > 0\) имеем: \(I(f, x) = \frac{1}{\pi}\int_{-\delta}^\delta f(x - t)\frac{\sin ut}{t}dt + o(1), u \ra \pm\infty\)
\end{corollary}

\begin{theorem}[О равносходимости]
    Пусть \(f \in L_1(\R)\) и \(f_0 \in L_1(T)\) совпадают в некоторой окрестности точки \(x\). Тогда интеграл Фурье \(I(f, x)\) и ряд Фурье \(S(f, x)\) сходятся или расходятся одновременно и, если сходятся, то к одному значению.
\end{theorem}

\lecture{13}

\begin{proof}
    Положим \(r = [u] + \frac{1}{2}\). Т.к. \(|r - u| < \frac{1}{2}\), то \(u \ra +\infty, r \ra +\infty\). Имеем \(I_u(f, x) - I_r(f, x) = \frac{1}{\sqrt{2\pi}}\int_E \widehat{f}(y)e^{iyx}dy\), где \(E\) --- объединение двух отрезков с концами \(-u, -r\) и \(r, u\) (числа \(r, u\) могут идти в разном порядке). Т.к. \(\mu(E) \le 1\), и \(\widehat{f}(y) \ra 0\) при \(y \ra \pm\infty\), то 
    \[|I_u(f, x) - I_r(f, x)| \le \frac{1}{\sqrt{2\pi}} \int_E|\widehat{f}(y)|dy \le \sup_{E} |\widehat{f}(y)| \ra 0, u \ra +\infty\]
    По условию, \(\exists \delta \in (0, \pi]: f(x - t) = f_0(x - t)\) при \(|t| < \delta\). Тогда по предыдущему следствию:
    \[I_r(f, x) = \frac{1}{\pi}\int_{-\pi}^\pi f(x - t)\frac{\sin rt}{t}dt + o(1) = \frac{1}{\pi}\int_{-\pi}^\pi f_0(x - t) \frac{\sin\left( [u] + \frac{1}{2} \right)t}{t}dt + o(1)\]
    По \textbf{Лемме 2.4} получаем:
    \[S_{[u]}(f_0, x) = \frac{1}{\pi}\int_{-\delta}^{\delta} f_0(x - t)\frac{\sin\left( [u] + \frac{1}{2} \right)t}{t}dt + o(1)\]
    Откуда получаем, что \(I_r(f, x) - S_{[u]}(f, x) \ra 0\) при \(u \ra +\infty\). В итоге:
    \[|I_u(f, x) - S_{[u]}(f, x)| \le |I_u(f, x) - I_r(f, x)| + |I_r(f, x) - S_{[u]}(f, x)| \ra 0\]
\end{proof}

\begin{corollary}[Признак Дини]
    Пусть \(f \in L_1(\R), x \in \R\). Если для \(S \in \Cm \exists \delta > 0\) такое, что:
    \[\int_0^\delta \frac{|f(x - t) + f(x + t) - 2S|}{t}dt < +\infty\]
    То интеграл Фурье в точке \(x\) сходится к \(S\), т.е. \(I(f, x) = S\).
\end{corollary}
\begin{corollary}
    Пусть \(f_0 \in L_1(T)\) такая, что \(f_0(x - t) = f(x - t)\) при \(|t| < \delta\). Тогда по признаку Дини для рядов, \(S_n(f_0, x) \ra S\). Но тогда по предыдущей теореме, \(I_u(f, x) \ra S\) при \(u \ra +\infty\).
\end{corollary}

\begin{note}
    Аналогично можно получить и другие свойства ряда Фурье, используя теорему о равносходимости
\end{note}

\begin{corollary}[Формула обращения]
    Если \(f \in L_1(\R) \cap C(\R)\) и \(\widehat{f} \in L_1(\R)\), то \(I(f) = f\) на \(\R\), то есть для любого \(x \in \R\) выполнено:
    \[f(x) = \frac{1}{\sqrt{2\pi}}\int_{-\infty}^{+\infty}\widehat{f}(y)e^{iyx}dy\]
\end{corollary}
\begin{proof}
    Интеграл \(I(f)\) сходится (существует как интеграл Лебега), т.к. \(f \in L_1(\R)\). Пусть \(f_0 \in L_1(T)\), такая, что \(f_0 = f\) на \((x - \pi, x + \pi)\). По теореме о равносходимости, \(S(f_0, x) = I(f, x)\). Т.к. \(f_0 \in C(x - \pi, x + \pi)\), то \(S(f_0, x) = f_0(x)\) на \((x - \pi, x + \pi)\), и тогда \(I(f, x) = S(f_0, x) = f_0(x) = f(x)\), а значит, равенство есть и на \(\R\).
\end{proof}

В связи с формулой обращения, имеет смысл ввести следующее понятие:

\begin{definition}
    Пусть \(f \in L_1(\R)\). Обратным преобразованием Фурье называется 
    \[\check{f}(y) = \frac{1}{\sqrt{2\pi}}\int_\R f(x)e^{iyx}dx\]
    То есть \(\check{f} = \hat{f_\sigma}\), где \(f_\sigma(x) = f(-x)\)
\end{definition}

\begin{note}
    В условиях формулы обращения верно: \(f = I(f) = \check{\hat{f}}\).
\end{note}

В связи с обобщением формулы обращения, докажем следующее утверждение

\begin{lemma}
    Пусть \(f, g \in L_1(\R)\). Тогда 
    \[\int_\R \check{f}(x)g(x)dx = f(x)\check{g}(x)dx\]
\end{lemma}
\begin{proof}
    Т.к. \(|f(t)g(x)e^{-ixt}| = |f(t)||g(x)|\) --- интегрируемая функция, то по теореме Фубини:
    \[\int_\R\left( \int_\R f(t)e^{-itx}dt \right)g(x)dx = \int_\R\left( \int_\R g(x)e^{-itx}dt \right)f(t)dt\]
    Разделив обе части на \(\frac{1}{\sqrt{2\pi}}\), получим требуемое.
\end{proof}

\begin{theorem}[Формула обращения]
    Пусть \(f, \widehat{f} \in L_1(\R)\). Тогда \(f = \check{\hat{f}} = \hat{\check{f}}\) почти всюду.
\end{theorem}
\begin{proof}
    Рассмотрим \(\phi_\epsilon(t) = \frac{1}{\sqrt{\pi\epsilon}}e^{-\frac{t^2}{2}}\). Тогда \(\phi_\epsilon\) --- четная (а значит, \(\widehat{\phi_\epsilon}\) --- тоже). Кроме того, \(\phi_\epsilon\) --- дифференцируемая функция, поэтому по формуле обращения: \(\phi_\epsilon = \widehat{\widehat{\phi_\epsilon}}\). Для любого \(x \in \R\) имеем:
    \[(\phi_0 * f)(x) = \int_\R f(x)\widehat{\widehat{\phi_\epsilon}}(t)dt = \int_\R \widehat{f(x + t)}\widehat{\phi_\epsilon}(t)dt\]
    По формуле сдвига: \(\widehat{f(x + y)}(y) = e^{ixy}\widehat{f}(y), \widehat{\phi_\epsilon}(t) = \frac{1}{\sqrt{2\pi}}e^{-\frac{t^2\epsilon^2}{4}}\). Тогдв:
    \[(\phi_\epsilon * f)(x) = \frac{1}{\sqrt{2\pi}}\int_\R \widehat{f}(y)e^{ixt}\cdot e^{-\frac{t^2\epsilon}{4}}dt\]
    Подынтегральная функция ограничена мажорантой \(\frac{1}{\sqrt{2\pi}|\widehat{f}(y)|}\). Тогда по теореме о непрерывности интеграла с параметром можно перейти к пределу:
    \[\frac{1}{\sqrt{2\pi}}\int_\R \widehat{f}(y)e^{ixt}\cdot e^{-\frac{t^2\epsilon}{4}}dt \ra \frac{1}{\sqrt{2\pi}}\int_\R \widehat{f}(y)e^{ixt}dt\]
    \[\phi_\epsilon * f \ra_{L_1} f, \epsilon \ra 0\]
    Возьмем произвольную последовательность \(\epsilon_n \ra 0\). Тогда из доказательства теоремы о полноты \(L_p\), \(\exists \{\epsilon_{n_k}\}: \phi_{\epsilon_{n_k}} * f \ra f\) почти всюду. Положим \(\epsilon = \epsilon_{n_k}\) и получим (перейдя к пределам в левых и правых частях):
    \[f = \check{\hat{f}}\]
\end{proof}

\begin{corollary}
    Пусть \(f, g \in L_1(\R), \widehat{f} = \widehat{g}\). Тогда \(f = g\) почти всюду.
\end{corollary}
\begin{proof}
    Рассмотрим \(h = f - g\). Тогда \(\widehat{h} = 0 \Ra h = 0\) почти всюду.
\end{proof}

\begin{note}
    Можно смотреть на преобразование Фурье, как оператор \(F: L_1(\R) \ra C_0(\R)\) (\(C_0(\R)\) --- непрерывные функции, стремящиеся к \(0\) на \(\pm\infty\)). Данный оператор инъективен, но не является биекцией (задача --- покажите, что он не сюрьективен). Это создает неудобства для работы. В связи с этим, далее введется пространство Шварца \(S(\R)\), такое, что \(F: S(\R) \ra S(\R)\), на котором \(F\) уже будет являться биекцией.
\end{note}

\begin{problem}
    Пусть \(f \in C^2(\R), f, f', f'' \in L_1(\R)\). Докажите, что \(\widehat{f} \in L_1(\R)\).
\end{problem}
