\subsection{Напоминание с ОВиТМа}

\begin{definition}
    \(\mathcal{F}\) --- алгебра над \(\Omega\), если 
    \begin{enumerate}
        \item \(\Omega \in \mathcal{F}\)
        \item \(A, B \in \mathcal{F} \Ra A \cap B \in \mathcal{F}\)
        \item \(A \in \mathcal{F} \Ra \overline{A} \in \mathcal{F}\)
    \end{enumerate}
\end{definition}

\begin{note}
    Элементы \(\Omega\) называются элементарными событиями.
\end{note}

\begin{note}
    Элементы \(\mathcal{F}\) называются событиями.
\end{note}

\begin{definition}
    \(\mathcal{F}\) --- \(\sigma\)-алгебра над \(\Omega\), если 
    \begin{enumerate}
        \item \(\mathcal{F}\) --- алгебра
        \item \(A_1, \dots \in \mathcal{F} \Ra \bigcup_{i = 1}^\infty A_i \in \mathcal{F}\)
    \end{enumerate}
\end{definition}

\begin{definition}
    Функция \(P: \mathcal{F} \ra [0, 1]\) называется вероятностной мерой, если \(P(\Omega) = 1, P\) --- \(\sigma\)-аддитивна.
\end{definition}

\begin{note}
    \begin{enumerate}
        \item \(P(\emptyset) = 0\)
        \item \(A \subset B \Ra P(A) \le P(B)\) --- \textbf{монотонность меры}
        \item \(P\) конечно аддитивна
        \item Для \(P\) верна формула включений-исключений.
        \item \(P\left( \bigcup_{i = 1}^\infty A_i \right) = \le \sum_{i = 1}^\infty P(A_i)\)
    \end{enumerate}
\end{note}

\begin{theorem}[О непрерывности вероятностной меры]
    Пусть \((\Omega, \mathcal{F})\) таково, что \(\mathcal{F}\) --- алгебра над \(\Omega\), \(P\) --- мера и \(P(\Omega) = 1\). Тогда следующие условия эквивалентны:
    \begin{enumerate}
        \item \(P\) --- \(\sigma\)-аддитивна на \(\mathcal{F}\)
        \item \(P\) непрерывна в нуле, т.е. \(\bigcap A_i = \emptyset \Ra P(A_i) \ra P(A)\)
        \item \(P\) непрерывна сверху, т.е. \(\bigcap A_i = A \Ra P(A_i) \ra P(A)\)
        \item \(P\) непрерывна снизу, т.е. \(\bigcup A_i = A \Ra P(A_i) \ra P(A)\)
    \end{enumerate}
\end{theorem}

\begin{definition}
    Вероятностное пространство --- это измеримое пространство \((\Omega, \mathcal{F}, P)\), т.е. \(\Omega\) --- множество, \(\mathcal{F}\) --- некая \(\sigma\)-алгебра над \(\Omega\), \(P\) --- вероятностная мера на \((\Omega, \mathcal{F})\).
\end{definition}

\begin{definition}
    Система \(\mathcal{M}\) подмножеств \(\Omega\) называется \(\pi\)-системой, если \(\mathcal{M}\) замкнуто относительно счетного пересечения
\end{definition}

\begin{definition}
    Система \(\mathcal{M}\) подмножеств \(\Omega\) называется \(\lambda\)-системой, если
    \begin{enumerate}
        \item \(\Omega \in \mathcal{M}\)
        \item \(A, B \in \mathcal{M}, A \subset B \Ra B \setminus A \in \mathcal{M}\)
        \item \(A_i \in \mathcal{M} \Ra \bigcup A_i \in \mathcal{M}\)
    \end{enumerate}
\end{definition}

\begin{theorem}[Первая теорема о \(\lambda\)-системах]
    Система \(\mathcal{F}\) является \(\sigma\)-алгеброй над \(\Omega\) тогда и только тогда, когда она является \(\lambda\)-системой и \(\pi\)-системой.
\end{theorem}

\begin{proposition}
    Для любой системы \(\mathcal{M}\) подмножеств \(\Omega\) существует минимальная по включению, содержащаяся в \(\mathcal{M}\)
\end{proposition}

\begin{note}
    \(\sigma(\mathcal{M}), \alpha(\mathcal{M}), \pi(\mathcal{M}), \lambda(\mathcal{M})\) --- порожденные (минимальные) \(\sigma\)-алгебра, алгебра, \(\pi\)-система, \(\lambda\)-система соответственно.
\end{note}

\begin{theorem}[Вторая теорема о \(\lambda\)-системах]
    Если \(\mathcal{M}\) --- \(\pi\)-система на \(\Omega\), то \(\sigma(\mathcal{M}) = \lambda(\mathcal{M})\)
\end{theorem}
\begin{proof}
    См. доказательство из курса ОВиТМа
\end{proof}

\begin{example}[Борелевская \(\sigma\)-алгебра]
    \(\mathcal{B}(\R)\) --- \(\sigma\)-алгебра, порожденная всеми открытыми множествами (или, что равносильно, всеми полуинтервалами)
\end{example}

\begin{example}[Борелевская \(\sigma\)-алгебра в \(\R^n\)]
    \(\mathcal{B}(\R^n)\) --- \(\sigma\)-алгебра, порожденная всеми открытыми множествами (или, что равносильно, всеми кубами, где куб --- декартово произведение борелевских множеств).
\end{example}

\begin{example}
    \(\mathcal{B}(\R^\infty)\) --- \(\sigma\)-алгебра, порожденная всеми цилиндрами. Цилиндр --- множество \({x \in \R^\infty : (x_1, \dots x_n) \in B}, B \in \mathcal{B}(\R^n)\)
\end{example}


\begin{example}[Борелевская \(\sigma\)-алгебра в общем случае]
    Пусть \((S, \rho)\) --- метрическое пространство, тогда \(\mathcal{B}(S)\) --- \(\sigma\)-алгебра, порожденная всеми открытыми множествами.
\end{example}

\section{Вероятностная мера на прямой}
Пусть \(P\) --- вероятностная мера на \(\R, \mathcal{B}(\R)\)
\begin{definition}
    Функцией распределения называется функция \(F(x) = P((-\infty, x]), x \in \R\).
\end{definition}

\begin{lemma}[О свойствах функции распределения]
    \indent
    \begin{enumerate}
        \item \(F\) не убывает
        \item \(\lim_{x \ra -\infty} F(x) = 0\), \(\lim_{x \ra +\infty} F(x) = 1\)
        \item \(F\) непрерывна справа
    \end{enumerate}
\end{lemma}
\begin{proof}
    \indent
    \begin{enumerate}
        \item \(x \le y \Ra (-\infty, x] \subset (-\infty, y] \Ra F(x) \le F(y)\)
        \item \(x_n \ra -\infty \Ra (-\infty, x_n] \ra \emptyset \Ra \lim_{n \ra \infty} F(x_n) = 0 = \lim_{x \ra +\infty} F(x)\), \(x_n \ra +\infty \Ra (-\infty, x_n] \ra \R \Ra \lim_{n \ra \infty} F(x_n) = 1 = \lim_{x \ra +\infty} F(x)\)
        \item Если \(x_n \searrow x\), то \((-\infty, x_n] \searrow (-\infty, x]\) и \(F(x_n) \ra F(x)\)
    \end{enumerate}
\end{proof}

\begin{theorem}[О взаимно-однозначном соответствии функции распределения и вероятностной меры]
    Пусть \(F\) удовлетворяет условиям 1-3 из предыдущей теоремы. Тогда существует единственная вероятностная мера \(P\) на \((\R, \mathcal{B}(\R))\), т.ч. \(F\) --- её функция распределения.
\end{theorem}

\subsection{Классификация вероятностных мер}

Далее мы будем отождествлять понятия распределения и вероятностной меры.

\subsubsection{Дискретные распределения}

\begin{definition}
    Вероятностная мера \(P\) на \((R, \mathcal{B}(R))\) называется дискретной, если \(\exists X\) --- не более. чем счетое множество на \(\R\), такое, что \(P(\R \setminus X)\) и \(\forall x \in X P(\{x\}) > 0\).
\end{definition}

\paragraph{Примеры}
\begin{enumerate}
    \item Распределение Бернулли: \(P \sim Bern(p)\), если \(X = \{0, 1\}, P(\{1\}) = p, P(\{0\}) = 1 - p\)
    \item Равномерное распределение: \(X = {1, 2, \dots n}\), \(P(\{i\}) = \frac{1}{n}\)
    \item Биномиальное распределение: \(P \sim Bin(n, p)\), если \(X = \N, P(\{k\}) = C_n^kp^k(1 - p)^{n - k}, P(\{0\})\), моделирует количество успехов среди \(n\) испытаний.
    \item Пуассоновское распределение: \(P \sim Pois(n, p)\), если \(X = \N, P(\{k\}) = \frac{\lambda^k}{k!}e^{-\lambda}, P(\{0\})\), моделирует редкие события
    \item Геометрическое распределение: \(P \sim Geom(p)\), если \(X = \N, P(\{k\}) = p(1 - p)^{k-1}, P(\{0\})\), моделирует первый момент удачи в бесконечной схеме испытаний Бернулли
\end{enumerate}

\subsubsection{Абсолютно непрерывные распределения}

\begin{definition}
    \(F(x)\) называется абсолютно непрерывной, если \(\exists p(t) \ge 0: \forall x \in \R F(x) = \int_{-\infty}^x p(t)dt\). В таком случае мы говорим, что \(p(t)\) является плотностью функции \(F\) или соответствующего распределения (вероятностной меры).
\end{definition}

\begin{note}
    В таком случае \(F'(x) = p(x)\) почти всюду по мере Лебега.
\end{note}

\paragraph{Примеры}
\begin{enumerate}
    \item Равномерное распределение на отрезке \([a, b]\) --- \(U(a, b)\): 
    \[p(x) = \left\{\begin{array}{l}
        \frac{1}{b - a}, a \le x \le b \\
        0, \text{иначе}
    \end{array}\right., F(x) = \left\{\begin{array}{l}
        0, x < a \\
        \frac{x - a}{b - a}, a \le x \le b \\
        1, x > b
    \end{array}\right.\]
    Моделирует случайную точку из отрезка \([a, b]\)
    \item Нормальное распределение --- \(N(a, \sigma^2)\):
    \[p(x) = \frac{1}{\sqrt{2\pi \sigma^2s}}e^{-\frac{(x - a)^2}{2\sigma^2}}\]
    Моделирует измерение с ошибкой

    \item Экспоненциальное распределение --- \(Exp(\lambda)\):
    \[p(x) = \lambda e^{\lambda x} I\{x > 0\}\]
    \[F(x) = \left\{\begin{array}{l}
        0, x < 0 \\
        1 - e^{-\lambda x}, x \ge 0
    \end{array}\right.\]
    Моделирует время ожидание

    \item Гамма-распределение: \(\Gamma(\lambda, \alpha), \lambda, \alpha > 0\).
    \[p(x) = \frac{x^{\alpha - 1}\lambda^\alpha}{\Gamma(\alpha)}e^{-\lambda x}I\{x > 0\}\]
    Где:
    \[\Gamma(\alpha) = \int_0^{+\infty} x^{\alpha - 1}e^{-x}dx\]
    Нам в дальнейшем потребуются различные свойства \(\Gamma\)-функции: \(\Gamma(n) = (n - 1)!, \Gamma(\alpha + 1) = \alpha\Gamma(\alpha)\)

    \item Распределение Коши \(K(\sigma), \sigma > 0\)
    \[p(x) = \frac{\sigma}{\pi(x^2 + \sigma^2)}\]
    \[F(x) = \frac{1}{2} + \frac{1}{\pi}\arctan \frac{x}{\sigma}\]
    Модель
\end{enumerate}

\subsubsection{Сингулярные распределения}

\begin{definition}
    Точка \(x\) называется точкой роста фукнции распределения \(F(x)\), если \(\forall \epsilon > 0: F(x + \epsilon) - F(x - \epsilon) > 0\)
\end{definition}

\begin{definition}
    Функция распределения \(F(x)\) наызвается сингулярной, если она непрерывна, и множество точек ее роста имеет \(\mu = 0\).
\end{definition}

\paragraph{Примеры}
\begin{enumerate}
    \item Канторова лестница --- ее точками роста ялвяется канторово множество.
\end{enumerate}

\begin{theorem}[Лебега о разложении]
    Если \(F(x)\) --- функция распределения на прямой, тогда \(\exists !\) разложение вида: \(F(x) = \alpha_1F_1(x) + \alpha_2F_2(x) + \alpha_3F_3(x)\), где \(\alpha_i \ge 0, \alpha_1 + \alpha_2 + \alpha_3 = 1\) и \(F_1\) --- дискретная, \(F_2\) --- абсолютно непрерывная, \(F_3\) --- сингулярная.
\end{theorem}

\section{Вероятностные меры в \(\R^n\)}
Пусть \(P\) --- вероятностная мера на \((\R^n, \mathcal{B}(\R^n))\). 
\begin{definition}
    Функцией распределения \(P\) называется \(F(x_1, \dots x_n) = P((-\infty, x_1] \times \dots \times (-\infty, x_n])\)
\end{definition}

\begin{note}[Обозначения]
    \begin{enumerate}
        \item \(\vec{x} = (x_1, \dots x_n)\)
        \item \(\vec{x} \ge \vec{y} \Lra x_i \ge y_i\)
        \item \((-\infty, \vec{x}] = (-\infty, x_1] \times \dots \times (-\infty, x_n]\)
        \item \(\vec{x}_n \downarrow \vec{x}\) если \(\vec{x} = \lim_{n \ra \infty}\vec{x}_n\) и \(\vec{x}_n \ge \vec{x}_{n + 1}\).
    \end{enumerate}
\end{note}

\begin{definition}
    Для \(i = 1, \dots n, a_i < b_i\) введем:
    \[\Delta_{a_i, b_i}^iF(x_1, \dots x_n) = F(x_1, x_{i - 1}, b_i, x_{i + 1}, \dots x_n) - F(x_1, x_{i - 1}, a_i, x_{i + 1}, \dots x_n)\]
\end{definition}

\begin{lemma}[Свойства многомерных функций распределения]
    \begin{enumerate}
        \item Если \(\vec{x}_n \downarrow \vec{x}\), то \(F(\vec{x}_n) \ra F(\vec{x})\)
        \item \(\lim_{\begin{array}{c}
            x_1 \ra \infty \\
            \vdots \\
            x_n \ra \infty
        \end{array}} F(x_1, \dots x_n) = 1\)
        \item \(\forall i = 1, \dots n: \lim_{x_i \ra -\infty} F(x_1, \dots x_n) = 0\).
        \item Для любых \(a_i < b_i, i = 1, \dots n\):
        \[\Delta_{a_1, b_1}^1 \circ \Delta_{a_2, b_2}^2 \circ \dots \circ \Delta_{a_n, b_n}^n F(x_1, \dots x_n) \ge 0\]
    \end{enumerate}
\end{lemma}
\begin{proof}\indent
    \begin{enumerate}
        \item Если \(\vec{x}_n \downarrow \vec{x}\), то \((-\infty, \vec{x}_n] \downarrow (-\infty, \vec{x}] \Ra\) в силу непрерывности вероятностной меры, получаем \(F(\vec{x}_n) \ra F(\vec{x})\)
        \item Если \(\vec{x}_n \uparrow (+\infty, \dots +\infty)\), то \((-\infty, \vec{x}_n] \uparrow \R^n \Ra\) в силу непрерывности вероятностной меры, получаем \(F(\vec{x}_n) \ra P(\R^n) = 1\)
        \item Если \(x_i \downarrow -\infty\), то \((-\infty, \vec{x}_n] \downarrow \emptyset \Ra\) в силу непрерывности вероятностной меры, получаем \(F(\vec{x}_n) \ra P(\emptyset) = 0\)
        \item Для \(n = 2\):
        \[\Delta^1_{a_1, b_1} \circ \Delta^2_{a_2, b_2} F(x_1, x_2) = F(b_1, b_2) - F(b_1, a_2) - F(a_1, b_2) + F(a_1, a_2) = P((a_1, b_1] \times (a_2, b_2]) \ge 0\]
        В общем случае:
        \[\Delta_{a_i, b_i}^i P(A_1 \times \dots \times A_{i - 1} \times (-\infty, x_i) \times A_{i + 1} \times \dots \times A_n) = \]
        \[= P(A_1 \times \dots \times A_{i - 1} \times (a_i, b_i] \times A_{i + 1} \times \dots \times A_n)\]
        Получаем, что 
        \[\Delta_{a_1, b_1}^1 \circ \Delta_{a_2, b_2}^2 \circ \dots \circ \Delta_{a_n, b_n}^n F(x_1, \dots x_n) = P((a_1, b_1] \times \dots \times (a_n, b_n]) \ge 0\]
    \end{enumerate}
\end{proof}

\begin{theorem}[О взаимно однозначном соответствии]
    Если \(F\) удовлетворяет свойствами 1-3 из леммы, то \(\exists !\) вероятностная мера \(P\) на \((\R^n, \mathcal{B}(\R^n))\), такая, что \(F\) --- ее функция распределения
\end{theorem}
\begin{proof}
    См. ОВиТМ
\end{proof}

\begin{note}
    Свойство 3 нельзя заменить на неубывание по каждой из переменнх.
\end{note}
\begin{proof}
    Рассмотрим \(F(x, y) = \left\{\begin{array}{l}
        1, x + y \ge 0 \\
        0, x + y < 0
    \end{array}\right.\).
    Заметим, что \(F\) удовлетворяет свойствам 1, 2 и не убывает по обеим переменным. При этом, если мы возьем:
    \[\Delta^x_{-1, 1} \circ \Delta^y_{-1, 1} F(x, y) = F(1, 1) - F(-1, 1) - F(1, -1) + F(-1, -1) = 1 - 2 + 0 < 0\]
    Получаем, что \(F\) --- не двумерная функция распределения.
\end{proof}

\subsection{Примеры}
\begin{enumerate}
    \item Пусть \(F_1, F_2, \dots F_n\) --- одномерные функции распределения. Рассмотрим \(F(x_1, \dots x_n) = F_1(x_1)\dots F_n(x_n)\) --- многомерная функция распределения. Свойства 1, 2 очевидны, проверим 3:
    \[\Delta_{a_1, b_1}^1 \circ \Delta_{a_2, b_2}^2 \circ \dots \circ \Delta_{a_n, b_n}^n F(x_1, \dots x_n) = \prod_{k = 1}^n (F_k(b_k) - F_k(a_k)) \ge 0\]

    \item Пусть \(p(t_1, \dots t_n) \ge 0\), т.ч.
    \[\int_{\R^n} p(t_1, \dots t_n) dt_1dt_2\dots dt_n = 1\]
    Тогда:
    \[(*)\;\;F(x_1, \dots x_n) = \int_{-\infty}^{x_1}\int_{-\infty}^{x_2}\dots \int_{-\infty}^{x_n} p(t_1, \dots t_n)dt_1dt_2\dots dt_n\]
    Свойства 1, 2 очевидны, проверим 3:
    \[\Delta_{a_1, b_1}^1 \circ \Delta_{a_2, b_2}^2 \circ \dots \circ \Delta_{a_n, b_n}^n F(x_1, \dots x_n) = \int_{a_1}^{b_1}\int_{a_2}^{b_2}\dots \int_{a_n}^{b_n} p(t_1, \dots t_n)dt_1dt_2\dots dt_n\]
\end{enumerate}

\begin{definition}
    Если имеет место представление \((*)\), то \(p(t_1, \dots t_n)\) называется плотностью функции распределения \(F\).
\end{definition}

\section{Вероятностные меры в \(\R^\infty\)}

\begin{definition}
    Пусть \(P\) --- вероятностная мера на \((\R^{\infty}, \mathcal{B}(\R^\infty))\). Рассмотрим для \(n \in \N\) вероятностную меру \(P_n\) на \((\R^n, \mathcal{B}(\R^n))\), т.ч.
    \[P_n(B) = P(Cyl(n, B)), B \in \mathcal{B}(\R^n)\]
    Тогда можно заметить, что \(P_{n + 1}(B \times \R) = P_n(B)\).
\end{definition}

\begin{definition}
    Свойство выше называется согласованностью для последовательности вероятностных мер \(\{P_n\}\)
\end{definition}

\begin{theorem}[Колмогорова, о мерах в \(\R^\infty\)]
    Пусть \(\{P_n, n \in \N\}\) --- последовательность согласованных вероятностных мер, \(P_n\) --- мера на \(\R^n\). Тогда \(\exists !\) вероятностная мера \(P\) на \((\R^\infty, \mathcal{B}(R^\infty))\), такая, что \(\forall n \in \N: \forall B_n \in \mathcal{B}(\R^n)\):
    \[(*)\;\;\;P_n(B_n) = P(Cyl(n, B_n))\]
\end{theorem}
\begin{proof}
    Зададим меру \(P\) на цилиндрах по правилу \((*)\). Цилиндры образуют алгебру \(\mathcal{A}\). Проверим корректность задания \(P\). Если \(Cyl(n, B_n) = Cyl(n + k, B_{n + k})\), то \(B_{n + k} = B_n \times \R^k\). Тогда в силу согласованности:
    \[P_n(B_n) = P_{n + k}(B_{n + k})\]
    Проверим, что \(P\) --- конечно аддитивна на \(\mathcal{A}\). Если \(\tilde{B_1}, \dots \tilde{B_N} \in \mathcal{A}\) --- непересекаются, то будем считать, что \(\exists n: \tilde{B_i} = Cyl(n, B_i), i = 1, \dots N, B_i \in \mathcal{B}(\R^n)\). Тогда:
    \[P\left( \bigsqcup_{i = 1}^N \tilde{B_i} \right) = P\left( Cyl\left( n, \bigsqcup_{i = 1}^N \tilde{B_i} \right) \right) = P_n\left( \bigsqcup_{i = 1}^N B_i \right) = \sum_{i = 1}^N P_n(B_i) = \sum_{i = 1}^N P(\tilde{B_i})\]

    Проверим, что \(P\) непрерывна в нуле (на \(\mathcal{A}\)). Пусть \(\tilde{B_n} \downarrow \emptyset, \tilde{B_n} \in \mathcal{A}\). Без ограничения общности, считаем, что \(\tilde{B_n} = Cyl(n, B_n)\). 

    От противного. Пусть \(\lim_{n \ra \infty} P(\tilde{B_n}) = \delta > 0\). Тогда для \(\forall n \in \N\) выберем компактные \(A_n \subset \R^n\), такие, что \(A_n \subset B_n\) и \(P(B_n \setminus A_n) \le \frac{\delta}{2^{n + 1}}\). Обозначим \(\tilde{A_n} = Cyl(n, A_n)\). Тогда: \(P(\tilde{B_n} \setminus \tilde{A_n}) = P_n(B_n \setminus A_n) \le \frac{\delta}{2^{n + 1}}\). Введем \(\tilde{C_n} = \bigcap_{i = 1}^n \tilde{A_i}\). Тогда \(\tilde{C_n} \downarrow \emptyset, \tilde{C_n} = Cyl(n, C_n)\), где \(C_n = A_n \cap (A_{n - 1} \times \R) \cap (A_{n - 2} \times \R^2) \cap \dots \cap (A_{1} \times \R^{n - 1})\) --- тоже компакт в \(\R^n\). Далее:
    \[P(\tilde{B_n} \setminus \tilde{C_n}) \le \sum_{i = 1}^n P(\tilde{B_n} \setminus \tilde{A_i}) \le |B_i \subset B_n, i \ge n| \le \sum_{i = 1}^n P(\tilde{B_i} \setminus \tilde{A_i}) \le \frac{\delta}{2} \Ra \lim_{n \ra \infty} P(\tilde{C_n}) \ge \frac{delta}{2} > 0\]
    Возьмем в каждом \(\tilde{C_n}\) по точке \((x_1^{(n)}, x_2^{(n)}, \dots) \in \tilde{C_n}\). Тогда \((x_1^{(n)}, \dots x_n^{(n)}) \in C_n\). Рассмотрим последовательность \(x_1^{(n)}\). Эти все точки лежат в \(C_1\). Выберем подпоследовательность \(n_{k_i}\)  \(x_1^{(n_{k_i})} \ra x_1^0 \in C_1\). Для \(k \in \N\) построим последовательности \((x_1^{(n_{k_i})}, x_2^{(n_{k_i})}, \dots x_k^{(n_{k_i})})\), которая сходится к \(x^{(0)}_k, \dots x^{(0)}_k\) в \(С_k\). Возьмем диагональ, т.е положим \(m_k = n_{k_k}\). Тогда \(\forall s \in \N: (x_1^{(m_k)}, \dots x_s^{(m_k)}) \ra (x_1^{(0)}, \dots x_s^{(0)})= x^{(0)}\). Получаем, что \(P\) счетно-аддитивна на \(\mathcal{A}\). В таком случае по теореме о продолжении меры, \(P\) единственным образом продолжается до вероятностной меры на \(\sigma(\mathcal{A}) = \mathcal{B}(\R^\infty)\)
\end{proof}

\section{Случайные величины и векторы}
Пусть \((\Omega, \mathcal{F}, P)\) --- вероятностное пространство, \((E, \mathcal{E})\) --- измеримое пространство

\begin{definition}
    Отображение \(X: \Omega \ra E\) называется случайным элементом, если \(X\) --- измеримо, т.е. \(\forall B \in \mathcal{E}: X^{-1}(B) = {\omega: X(\omega) \in B} \in \mathcal{F}\)
\end{definition}

\begin{definition}
    Если \((E, \mathcal{E}) = (\R, \mathcal{B}(\R))\), то \(X\) называется случайной величиной.
\end{definition}

\begin{definition}
    Если \((E, \mathcal{E}) = (\R^n, \mathcal{B}(\R^n))\), то \(X\) называется случайным вектором.
\end{definition}

\begin{reminder}[Критерий измеримости отображения]
    Если \(\mathcal{M} \subset \mathcal{E}\) и \(\sigma(\mathcal{M}) = \mathcal{E}\), то \(X\) --- случайный элемент \(\Lra \forall B \in \mathcal{M} X^{-1}(B) \in \mathcal{F}\)
\end{reminder}

\begin{corollary}[Эквивалентные определения случайных величин и векторов]
    \begin{enumerate}
        \item \(\xi: \Omega \Ra \R\) --- случайная величина \(\Lra \{\xi \le x\} \in \mathcal{F} \Lra \{\xi < x\} \in \mathcal{F}\)
        \item \(\xi = (\xi_1, \dots, \xi_n)\) --- случайный вектор \(\Lra \forall i: \xi_i\) измеримо
    \end{enumerate}
\end{corollary}
\begin{proof}
    \begin{enumerate}
        \item Очевидно следует из критерия, т.к. лучи (замкнутые или открытые) порождают \(\mathcal{B}(\R)\).
        \item В одну сторону. возьмем \(B_i \in \mathcal{B}(\R)\). Т.к. \(\underbrace{\R \times \dots \times \R \times}_{i - 1}B_i \times \dots \times \R= B \in \mathcal{B}(\R^n)\), то \(\{\xi_i \in B_i\} = \{\xi \in B\} \in \mathcal{F}\). В другую: \(\{\xi \in B_1 \times \dots \times B_n\} = \bigcap_{i = 1}^n\{\xi_i \in B_i\}\). Далее, по критерию измеримости, следствие верно (т.к. полуалгебра кубов порождает \(\mathcal{B}(\R^n)\)).
    \end{enumerate}
\end{proof}

\subsection{Действия со случайными величинами}
\begin{enumerate}
    \item Если \(f: \R^n \ra \R^l\) --- борелевская, \(\xi\) --- случайная, то \(f(\xi)\) --- случайная.
    \item Арифметические операции
    \item Пусть \(\xi_n\) --- последовательность случайных величин, тогда \(\sup_n \xi_n, \inf_n \xi_n, \limsup \xi_n, \liminf \xi_n\) --- тоже случайные величины
\end{enumerate}

\subsection{Характеристики случайных величин и векторов}
\begin{definition}
    Распределением случайной величины \(\xi\) на \((\Omega, \mathcal{F}, P)\) называется вероятностная мера \(P_\xi\) на \((\R, \mathcal{B}(\R))\), определяемая по правилу \(\forall B \in \mathcal{B}(\R): P_\xi(B) = P(\xi \in B)\)
\end{definition}

\begin{definition}
    Распределением случайного вектора \(\xi\) на \((\Omega, \mathcal{F}, P)\) называется вероятностная мера \(P_\xi\) на \((\R^n, \mathcal{B}(\R^n))\), определяемая по правилу \(\forall B \in \mathcal{B}(\R^n): P_\xi(B) = P(\xi \in B)\)
\end{definition}

\begin{exercise}
    \(P_\xi\) действительно является вероятностной мерой.
\end{exercise}

\begin{definition}
    Функция распределения случайной величины \(\xi\) --- это функция \(F_\xi(x) = P_\xi((-\infty, x]) = P(\xi \le x)\).
\end{definition}

\begin{definition}
    Функция распределения случайного вектора \(\vec{\xi}\) --- это функция \(F_{\vec{\xi}} = P_{\vec{\xi}}((-\infty, \vec{x}]) = P(\vec{\xi} \le \vec{x}) = P(\xi_1 \le x_1, \xi_2 \le x_2, \dots \xi_n \le x_n)\).
\end{definition}

\begin{note}
    Обозначение \(\{\xi \le x, \eta \le y\}\) обозначает \(\{\xi \le x\} \cap \{\eta \le y\}\).
\end{note}

\begin{note}
    Случайные величины и векторы наследуют классификацию распределений.
\end{note}

\begin{definition}
    \(\sigma\)-алгебра, порожденная случайной величиной \(\xi\) называется \(\mathcal{F}_\xi = \{\{\xi \in B\}: B \in \mathcal{B}(\R)\}\).
\end{definition}

\begin{definition}
    \(\sigma\)-алгебра, порожденная случайным вектором \(\xi\) называется \(\mathcal{F}_\xi = \{\{\xi \in B\}: B \in \mathcal{B}(\R^n)\}\).
\end{definition}

\begin{exercise}
    Проверить, что \(\mathcal{F}_\xi\) действительно является \(\sigma\)-алгебой.
\end{exercise}

\begin{definition}
    Пусть \(\xi\) --- случайный вектор на \((\Omega, \mathcal{F}, P)\). Пусть \(\mathcal{C} \subset \mathcal{F}\) --- некая \(\sigma\)-алгебра. Тогда \(\xi\) является \(\mathcal{C}\)-измеримой, если \(\mathcal{F}_\xi \subset \mathcal{C}\).
\end{definition}

Возникает вопрос: а что, если \(\mathcal{C} = \mathcal{F}_\eta\)?
\begin{lemma}
    Случайная величина \(\xi\) является \(\mathcal{F}_\eta\)-измеримой \(\Lra \exists\) борелевская функция \(f\), такая, что \(\xi = f(\eta)\).
\end{lemma}
\begin{proof}\indent
    \begin{enumerate}
        \item [\(\La\)] Пусть \(B \in \mathcal{B}(\R)\). Тогда \(\{\xi \in B\} = \{f(\eta) \in B\} = \{\eta \in \underbrace{f^{-1}(B)}_{\in \mathcal{B}(\R)}\} \in \mathcal{F}_\eta\).
        \item [\(\Ra\)] Пусть \(\xi = I_A, A \in \mathcal{F}_\eta\). Тогда \(\exists B \in \mathcal{B}(\R^n)\), т.ч. \(A = \{\eta \in B\}\). Тогда \(\xi = f(\eta), f(x) = I_B\). Следовательно, все простые \(\mathcal{F}_\eta\)-измеримые случайные величины тоже являются борелевскими функциями от \(\eta\). Если \(\xi\) --- произвольная случайная величина, то рассмотрим последовательность последовательность величин \(\{\xi_n\}\), такую, что \(\forall \omega \in \Omega: \xi_n(\omega) \ra \xi(\omega)\), где \(\xi_m\) --- простые случайные величины и \(\forall m \in \N: \xi_m = g_m(\xi)\), где \(g\) --- борелевская. Далее, \(\mathcal{F}_{\xi_m} \subset \mathcal{F}_{\xi} \subset \mathcal{F}_{\eta} \Ra \exists \) борелевская функция \(f_m\), т.ч. \(\xi_m = f_m(\eta)\). Введем \(B = \{x \in \R^n: \exists \lim_{m \ra \infty} f_m(x)\}\). Положим \(f(x) = \left\{\begin{array}{l}
            \lim_{m \ra \infty} f_m(x), x \in B \\
            0, x \notin B.
        \end{array}\right.\). Заметим, что \(B \in \mathcal{B}(\R^n) \Ra f\) --- тоже борелевская функция. По построению \(\forall \omega \in \Omega: \xi(\omega) = \lim_{m \ra \infty}\xi_m(\omega) = \lim_{m \ra \infty}f_m(\eta(\omega)) = f(\eta)\).
    \end{enumerate}
\end{proof}

\section{Математическое ожидание}
Пусть \(\xi\) --- случайная величина на \((\Omega, \mathcal{F}, P)\)
\begin{definition}
    Математическое ожидание случайной величины \(\xi\) называется интеграл Лебега:
    \[\E\xi = \int_{\Omega} \xi(\omega) P(d\omega) = \int_{\Omega} \xi dP\]
\end{definition}

\begin{reminder}
    Напомним, как мы строили интеграл Лебега:
    \begin{enumerate}
        \item Для простой функции как сумму
        \item Для неотрицательной функции как предел
        \item Для функции разных знаков, разбиваем \(\xi = \xi^+ - \xi^-\)
    \end{enumerate}
    Далее говорим следующее:
    \begin{enumerate}
        \item Если \(\E \xi^{\pm}\) конечно, то \(\E\xi = \E\xi^+ - \E\xi^-\).
        \item Если одно из \(\E\xi^{\pm}\) равно \(+\infty\), то \(\E\xi = \pm\infty\) (ставим соответствующий знак).
        \item Если \(\E\xi^+ = \E\xi^- = +\infty\), то говорим, что \(\E\xi\) не определено.
    \end{enumerate}
\end{reminder}

\subsection{Свойства матожидания}
Из свойств интеграла:
\begin{enumerate}
    \item Линейность: \(\E(\alpha\xi + \beta\eta) = \alpha\E\xi + \beta\E\eta\)
    \item Сохраняет отношение порядка: \(\xi \le \eta \Ra \E\xi \le \E\eta\).
    \item \(|\E\xi| \le \E|\xi|\)
\end{enumerate}

\begin{definition}
    Событие \(A \in \mathcal{F}\) происходит почти наверное, если \(P(A) = 1\). Мы будем писать это как ''\(A\) п.н.''.
\end{definition}

И еще свойства математического ожидания:
\begin{enumerate}
    \setcounter{enumi}{3}
    \item Если \(\xi \ge 0, \E\xi = 0\), то \(\xi = 0\) п.н.
    \item Если \(\xi = \eta\) п.н., то \(\E\xi = \E\eta\).
    \item Если \(\forall A \in \mathcal{F}: \E(\xi \cdot I_A) \le \E(\eta \cdot I_A) \Ra \xi \le \eta\) п.н.
\end{enumerate}

\begin{theorem}[Замена переменных в интеграле Лебега]
    Пусть \(\xi\) --- случайный вектор из \(\R^n\) на \((\Omega, \mathcal{F}, P)\). Пусть \(f: \R^n \ra \R\) --- борелевская функция. Тогда \(\E f(\xi) = \int_{\R^n} f(x) P_\xi(dx)\).
\end{theorem}
\begin{proof}
    Пусть сначала \(f(x) = I_B(x), B \in \mathcal{B}(\R^n)\). Тогда
    \[\E f(\xi) = E \cdot I\{\xi \in B\} = P(\xi \in B) = P_\xi(B) = \int_B P\xi(dx) = \int_{\R^n} I_B(x) P_\xi(dx) = \int_{\R^n} f(x) P_\xi(dx)\]
    Заключаем, что теорема верна для простых функцйи в силу линейности матожидания и интеграла Лебега. Далее осуществляем приближение простыми функциями в общем случае.
\end{proof}

\subsection{Обобщенная плотность}

Но как посчитать последний интеграл? Здесь нам поможет обобщенная плотность. Пусть \(P\) --- вероятностная мера на \((\Omega, \mathcal{F}), \mu\) --- \(\sigma\)-конечная мера на \((\Omega, \mathcal{F})\).

\begin{definition}
    Мера \(P\) имеет обобщенную плотность по мере \(p(x)\) по мере \(\mu\), если
    \[\forall B \in \mathcal{F}: P(B) = \int_B p(x)\mu(dx)\]
\end{definition}

Смысл данного определения состоит в том, что если \(P\) --- абсолютно непрерывна относительно \(\mu\), то \(p = \frac{dP}{d\mu}\) --- производная Радона-Никодима.

\subsection{Примеры}

\begin{enumerate}
    \item \(\mu\) --- мера Лебега на \(\R \Ra\) для абсолютно-непрерывного распределения обобщенная плотность совпадает с обычной плотностью
    \item \(\mu\) --- считающая мера на \(\Z\) (дискретное распределение на \(\Z\)). Тогда обобщенная плотность вычилсяется по формуле \(p(x) = P(\{x\})\)
\end{enumerate}

\begin{theorem}[О вычислении через обобщенную плотность]
    Пусть \(P\) --- вероятностная мера на \((\R^n, \mathcal{\R^n})\), имеющая плотность \(p(x)\) по мере \(\mu\). Тогда \(\forall\) борелевской функции \(f: \R^n \ra \R\) выполнено:
    \[\int_{\R^n} f(x)P(dx) = \int_{\R^n} f(x)p(x)dx\]
\end{theorem}
\begin{proof}
    Пусть \(f(x) = I_B(x), B \in \mathcal{B}(\R^n)\). Тогда:
    \[\int_{\R^n} f(x) P(dx) = \int_B P(dx) = P(B) = \int_B p(x)\mu(dx) = \int_{\R^n} f(x)p(x)\mu(dx)\]
    Т.к. обе части требуемого линейны по \(f\), получаем, что равенство верно для простых функций. Для произвольной \(f\) осуществляем приближение простыми по стандартной схеме.
\end{proof}

\begin{corollary}
    Пусть \(\xi\) -- абсолютно непрерывный случайный вектор. с плотностью \(p(x)\). Тогда \(\forall f\) --- борелевской:
    \[\E f(\xi) = \int_{\R^n} f(x)p(x)dx\]
\end{corollary}

\begin{corollary}
    Пусть \(\xi\) --- дискретная случайная величина, сосредоточенная на некотором множестве \(\mathcal{X}\). Тогда \(\forall f\) --- борелевской:
    \[\E f(\xi) = \sum_{x \in \mathcal{X}} (f(x)P(\xi = x))\]
\end{corollary}
\begin{proof}
    \[\E f(\xi) = \int_\R f(x) P_\xi(dx) = \sum_{x \in \mathcal{X}} f(x)\underbrace{P(\xi = x)}_{\text{обобщ. плотность}} \underbrace{\mu(\{x\})}_{\text{считающая}} = \sum_{x \in \mathcal{X}} f(x) P(\xi = x)\]
\end{proof}

\begin{example}
    Пусть \(\xi\) имеет следующую функцию распределения:
    \[F(x) = \left\{\begin{array}{l}
        0, x < 0 \\
        1 - e^{-x}, x \in [0, 1) \\
        0, \text{иначе}
    \end{array}\right.\]
    Требуется найти \(\E \xi\)
\end{example}
\begin{solution}
    \[\E \xi = 1 \cdot P\left(\xi = \frac{1}{e}\right) + \int_0^1 x F'(x) dx = \frac{1}{e} + \int_0^1 xe^{-x}dx = \frac{1}{e} + (-xe^{-x} - e^{-x})|_0^1 = 1 - \frac{2}{e} + \frac{1}{e} = 1 - \frac{1}{e}\]
\end{solution}

\section{Независимость}
\subsection{Независимость событий}
Пусть \((\Omega, \mathcal{F}, P)\) --- вероятностное пространство.
\begin{definition}
    События \(A, B\) независимы в \((\Omega, \mathcal{F}, P)\), если \(P(A \cap B) = P(A)P(B)\).
\end{definition}

\begin{definition}
    События \(A_1, \dots A_n\) называются попарно независимыми, если \(\forall i \ne j: A_i, A_j\) независимы.
\end{definition}

\begin{definition}
    События \(A_1, \dots A_n\) называются независимыми в совокупности, если \(\forall k \le n \forall i_1 < \dots < i_k: P(A_{i_1} \cap \dots \cap A_{i_k}) = P(A_{i_1})\dots P(A_{i_k})\).
\end{definition}

Далее, под независимостью будем подразумевать независимость в совокупности.

\begin{definition}
    Системы событий \(\mathcal{M}_1, \dots \mathcal{M}_n\) независимы в совокупности, если \(\forall A_i \in \mathcal{M}_i\) события \(\{A_i\}\) независимы в совокупности.
\end{definition}

\begin{theorem}[критерий независимости \(\sigma\)-алгебр]
    Пусть \(\mathcal{M}_1, \dots \mathcal{M}_n\) --- \(\pi\)- системы событий на \((\Omega, \mathcal{F}, P)\). Тогда \(\mathcal{M}_1, \dots \mathcal{M}_n\) независимы \(\Lra \sigma(\mathcal{M}_1), \dots \sigma(\mathcal{M}_2)\)
\end{theorem}
\begin{proof}\indent
    \begin{enumerate}
        \item[\(\La\)] очевидно из определения
        \item[\(\Ra\)] Докажем для \(n = 2\) (для других аналогично).
        Рассмотрим
        \[\mathcal{L}_1 = \{A \in \mathcal{F}: A\text{ независимо с }\mathcal{M}_2\}\]
        По условию, \(\mathcal{M}_1 \subset \mathcal{L}_1\). Проверим, что \(\mathcal{L}_1\) --- \(\lambda\)-система.
        \begin{enumerate}
            \item \(\Omega \in \mathcal{L}_1\), т.к. оно независимо со всеми событиями.
            \item Пусть \(A, B \in \mathcal{L}_1, A \subset B\) докажем, что \(B \setminus A \in \mathcal{L}_1\). Пусть \(C \in \mathcal{M}_2\). Тогда:
            \[P((B \setminus A) \cap C) = P((B \cap C) \setminus (A \cap C)) = P(B \cap C) - P(A \cap C) = (*)\]
            Из того, что \(B, A \in \mathcal{L}_1\) получаем, что они независимы с \(C\). Тогда:
            \[(*) = P(B)P(C) - P(A)P(C) = (P(B) - P(A))P(C) = P(B \setminus A)P(C)\]
            \item Пусть \(\{A_n, n \in \N\}\) таковы, что \(A_n \uparrow A\) и \(\forall n: A_n \in \mathcal{L}_1\) докажем, что \(A \in \mathcal{L}_1\). Пусть \(C \in \mathcal{M}_2\). Тогда \(A_n \cap C \uparrow A \cap C\). В силу непрерывности вероятностной меры:
            \[P(A \cap C) = \lim_{n \ra \infty} P(A_n \cap C) = \lim_{n \ra \infty} P(A_n)P(C) = P(C)\lim_{n \ra \infty} P(A_n) = P(C)P(A)\]
            Получили, что \(A\) независимо с \(\mathcal{M}_2 \Ra A \in \mathcal{L}_1\).
        \end{enumerate}

        То есть \(\mathcal{L}_1\) --- это \(\lambda\)-система, \(\mathcal{M}_1 \subset \mathcal{L}_1\). По второй теореме о \(\pi\)-\(\lambda\)-системах:
        \[\sigma(\mathcal{M}_1) \subset \mathcal{L}_1 \Ra \sigma(\mathcal{M}_1)\text{ независимо с }\mathcal{M}_2\]

        Теперь рассмотрим \(\mathcal{L}_2 = \{A \in \mathcal{F}: A\text{ независимо с }\sigma(\mathcal{M}_1)\}\). По доказанному \(\mathcal{M}_2 \subset \mathcal{L_2s}\). Далее точно так же проверяем, что \(\mathcal{L}_2\) --- \(\lambda\)-система.
    \end{enumerate}
\end{proof}

\begin{definition}
    Пусть \(\{\mathcal{M}_\alpha, \alpha \in \mathcal{A}\}\) --- системы событий на \((\Omega, \mathcal{F}, P)\). Этот набор называется независимым, если любой конечный поднабор \(M_{\alpha_1}, \dots M_{\alpha_k}, \alpha_i \ne \alpha_j\) независим.
\end{definition}

\subsection{Независимость случайных величин и случайных векторов}
\begin{definition}
    Случайные векторы \(\xi_1, \dots \xi_n\) называются независимыми в совокупности, если порожденные ими \(\sigma\)-алгебры являются независимыми.
\end{definition}

\begin{proposition}
    Случайные векторы \(\xi_1, \dots \xi_n, \xi_i \in \R^{k_i}\) независимы в совокупности \(\Lra \forall B_1, \dots B_n, B_i \in \mathcal{B}(\R^{k_i})\) выполнено
    \[P(\xi_1 \in B_1, \dots \xi_n \in B_n) = \prod_{i = 1}^n P(\xi_i \in B_i)\]
\end{proposition}
\begin{proof}
    \begin{enumerate}
        \item[\(\Ra\)] очевидно
        \item[\(\La\)] Надо проверить, что \(\forall B_1, \dots B_n\) события \(\{\xi_i \in B_i\}\) независимы в совокупности. Если фиксирован набор \(i_1, \dots i_k\), положим \(B_j = \R^{k_j}\) для \(j \ne i_s\). Тогда получаем:
        \[P(\xi_1 \in B_1, \dots \xi_n \in B_n) = \prod_{i = 1}^n P(\xi_i \in B_i) = \prod_{s = 1}^k P(\xi_{i_s} \in B_{i_s})\]
    \end{enumerate}
\end{proof}

\begin{definition}
    Пусть \(\xi_1, \dots \xi_n\) --- случайные величины. Совместным распределением \(\xi_1, \dots \xi_n\) называется распределение вектора \((\xi_1, \dots \xi_n)\). Соответствующие функции распределения и плотности тоже называются совместными
\end{definition}

\begin{proposition}
    Если случайные величины \(\xi_1, \dots \xi_n\) независимы, то их совместное распределение есть прямое произведение распределений \(\xi_1, \dots \xi_n\), т.е:
    \[P_{(\xi_1, \dots \xi_n)} = P_{\xi_1} \times \dots \times P_{\xi_n}\]
\end{proposition}
\begin{proof}
    Для \(\forall B_1, \dots B_n \in \mathcal{B}(\R)\) имеем:
    \[P_{(\xi_1, \dots \xi_n)}(B_1 \times \dots \times B_n) = P((\xi_1, \dots \xi_n) \in B_1 \times \dots \times B_n) = \]
    \[P(\xi_1 \in B_1, \dots \xi_n \in B_n) = P(\xi_1 \in B_1)\dots P(\xi_n \in B_n) = P_{\xi_1} \times \dots \times P_{\xi_n}\]
\end{proof}

\begin{theorem}[Критерий независимости для функции распределения]
    Пусть \(\xi_1, \dots \xi_n\) --- случайные величины на \((\Omega, \mathcal{F}, P)\). Они независимы в совокупности \(\Lra \forall x_1, \dots x_n \in \R\) выполнено:
    \[P(\xi_1 \le x_1, \dots \xi_n \le x_n) = \prod_{i = 1}^n \underbrace{P(\xi_i \le x_i)}_{F_{\xi_i}(x_i)}\]
\end{theorem}
\begin{proof}
    \begin{enumerate}
        \item[\(\Ra\)] следует из определения
        \item[\(\La\)] Пусть \(F_1, \dots F_n\) --- функции распределения случайных величин \(\xi_1, \dots \xi_n\). Рассмотрим
        \[G(x_1, \dots x_n) = \prod_{i = 1}^n F_{\xi_i}(x_i) = P(\xi_1 \le x_1, \dots \xi_n \le x_n)\]
        Получили, что \(G\) --- многомерная функция распределения. Тогда легко видеть, что \(G\) --- функция распределения меры \(P_{\xi_1}\times \dots \times P_{\xi_n}\). Действительно:
        \[P_{\xi_1}\times \dots \times P_{\xi_n}((-\infty, x_1] \times \dots \times (-\infty, x_n]) = P(\xi_1 \le x_1, \dots \xi_n \le x_n) = G(x_1, \dots x_n)\]
        С другой стороны, по условию, \(G\) --- совместная функция распределения \(\xi_1, \dots \xi_n \Ra P_{\xi_1}\times \dots \times P_{\xi_n}\) --- распрелделение \(\vec{\xi} = (\xi_1, \dots \xi_n)\). Поэтому:
        \[\forall B_1, \dots B_n: P(\xi_1 \in B_1, \dots \xi_n \in B_n) = P_{\xi_1}\times \dots \times P_{\xi_n}(B_1 \times \dots \times B_n) = \prod_{i = 1}^n P(\xi_i \in B_i)\]
    \end{enumerate}
\end{proof}

\begin{theorem}[Общий случай]
    Пусть \(\vec{\xi}_1, \dots \vec{\xi}_n\) --- случайные векторы на \((\Omega, \mathcal{F}, P)\), \(\vec{\xi_i} \in \R^{k_i}\). Они независимы в совокупности \(\Lra \forall \vec{x_1}, \dots \vec{x_n}\) выполнено:
    \[P(\vec{\xi}_1 \le \vec{x_1}, \dots \vec{\xi}_n \le \vec{x_n}) = \prod_{i = 1}^n \underbrace{P(\vec{\xi}_i \le \vec{x_i})}_{F_{\vec{\xi}_i}(\vec{x_i})}\]
\end{theorem}
\begin{proof}
    Аналогично случаю для случайных величин
\end{proof}

\begin{lemma}[Функции от независимых тоже независимы]
    Пусть \(\vec{\xi_1}, \dots \vec{\xi_n}\) --- случайные векторы, \(\vec{\xi_i} \in \R^{k_i}\). Пусть \(f_i: \R^{k_i} \ra \R^{m_i}\) --- борелевские функции. Тогда \(\{f_i(\vec{\xi_i})\}\)
\end{lemma}
\begin{proof}
    Обозначим \(\eta_i = f_i(\xi_i)\). Тогда \(\mathcal{\eta_i} \subset \mathcal{F}_{\xi_i}\). Следовательно, \(\mathcal{F}_{\eta_1}, \dots \mathcal{F}_{\eta_n}\) независимы как подсистемы в независимых системах
\end{proof}

\begin{proposition}
    Пусть \(\xi_1, \dots \xi_{k_1}, \xi_{k_1 + 1}, \dots \xi_{k_2}, \dots \xi_{k_{n - 1}}, \dots \xi_{k_n}\) и \(f_1, \dots f_r\) --- борелевские функции, причем \(f_i: \R^{k_i - k_{i - 1}} \ra \R\). Тогда \(\{f_i(\xi_i)\}\) --- независимы.
\end{proposition}
\begin{proof}
    Заметим, что блоки, как случайные векторы, тоже независимы, т.к.:
    \[P(\xi_1 \le x_1, \dots \xi_{k_n} \le x_{k_n}) = \prod-{i = 1}^{k_n}P(\xi_i \le x_i) = \prod_{i = 1}^n P(\xi_{k_{i - 1} + 1} \le x_{k_{i - 1} + 1}, \dots \xi_{k_i} \le x_{k_i})\]
    Осталось воспользоваться предыдущей леммой.
\end{proof}

\begin{theorem}
    Если \(\xi, \eta\) --- независимые случайные величины и \(\E\xi, \E\eta\) конечно, то \(\E\xi\eta\) существует и \(\E\xi\eta = \E\xi \cdot \E\eta\)
\end{theorem}
\begin{proof}
    По формуле замены переменных в интеграле Лебега:
    \[\E\xi\eta = \iint_{\R^2}xy P_{\xi, \eta}(dx, dy)\]
    Из независимости случайных величин, имеем:
    \[P_{\xi, \eta} = P_\xi \times P_\eta\]
    Заметим, что по теореме Тонелли:
    \[\E|\xi|\cdot \E|\eta| = \int_\R \E|\xi||y|P_\eta(dy) = \int_\R\left( \int_\R|x|P_\xi(dx) \right)|y|P_\eta(dy) = \]
    \[= \iint_{\R^2} |xy| P_{\xi, \eta}(dx, dy) < +\infty\]
    Таким образом, по теореме Фубини:
    \[\iint_{\R^2} xy  P_\xi(dx) \times P_\eta(dy) = \int_\R \int_\R x P_\xi(dx) y P\eta(dy) = \E\xi \cdot \E\eta\]
\end{proof}

\section{Совместная плотность}
Пусть случайные величины \(\xi_1, \dots \xi_n\) имеют совместную плотность \(p(x_1, \dots x_n)\).

\begin{proposition}
    Каждая величина \(\xi_i\) имеет плотность, причем:
    \[p_i(x_i) = {\int\dots \int}_{\R^{n - 1}}p(x_1, \dots x_n)dx_1\dots dx_{i-1}dx_{i+1}\dots dx_n\]
\end{proposition}
\begin{proof}
    \[P(\xi_i \le x_i) = P((\xi_1, \dots \xi_n) \in \R \times \dots \times (-\infty, x_i] \times \dots \times \R) = \]
    \[= \int\dots\int p(y_1, \dots y_n)dy_1\dots dy_n\]
    По теореме Фубини, данный интеграл равен:
    \[\int_{-\infty}^{x_i}\left( \int\dots\int_{\R^{n - 1}}p(y_1, \dots y_n)dy_1\dots dy_{i-1}dy_{i+1}\dots dy_n \right)dy_i\]
    Получаем желаемое
\end{proof}

\begin{lemma}[Критерий независимости плотностей]
    Пусть \(\xi_1, \dots \xi_n\) --- случайные величины с плотностями \(p_1(x), \dots p_n(x)\). Тогда:
    \begin{enumerate}
        \item Если у них есть совместная плотность, \(p(x_1, \dots x_n)\) и 
        \[p(x_1, \dots x_n) = \prod_{i = 1}^n p_i(x_i)\]
        То \(\{\xi_i\}\) --- независимы в совокупности
        \item Если \(\xi_1, \dots \xi_n\) независимы в совокупности, то \(p_1(x_1)\dots p_n(x_n)\) будет их совместной плотностью.
    \end{enumerate}
\end{lemma}
\begin{proof}
    \begin{enumerate}
        \item Рассмотрим совместную функцию распределения:
        \[P(\xi_1 \le x_1, \dots \xi_n \le x_n) = \int_{-\infty}^{x_1}\dots \int_{-\infty}^{x_n} p(y_1, \dots y_n)dy_1\dots dy_n = \]
        \[= \int_{-\infty}^{x_1}\dots \int_{-\infty}^{x_n} p_1(y_1) \dots p_n(y_n)dy_1\dots dy_n = (*)\]
        По Теореме Фубини:
        \[(*) = \prod_{i = 1}^n \int_{-\infty}^{x_i}p_i(y_i) = \prod_{i = 1}^n P(\xi_i \le x_i)\]
        По критерию независимости \(\{\xi_i\}\) независимы
        \item В обратную сторону:
        \[P(\xi_1 \le x_1, \dots \xi_n \le x_n) = \prod_{i = 1}^n P(\xi_i \le x_i) = \prod_{i = 1}^n \int_{-\infty}^{x_i}p_i(y_i)dy_i = (*)\]
        По теореме Фубини:
        \[(*) = \int_{-\infty}^{x_1}\dots \int_{-\infty}^{x_n}p_1(y_1)\dots p_n(y_n)dy_1\dots dy_n\]
        Что и требовалось
    \end{enumerate}
\end{proof}

\begin{lemma}[О свертке распределений]
    Пусть \(\xi, \eta\) --- независимые случайные величины с функциями распределения \(F_\xi, F_\eta\). Тогда:
    \[F_{\xi + \eta}(x) = \int_\R F_\xi(x - y)dF_\eta(y) = \int_\R F_\eta(x - y)dF_\xi(y)\]
\end{lemma}
\begin{proof}
    \[F_{\xi + \eta} = P(\xi + \eta \le x) = \iint_{\R^2}I\{y + z \le x\}P_{\xi, \eta}(dy, dz) = \]
    \[ = \iint_{\R^2} I\{y + z \le x\}P_\xi(dy)P_\eta(dz) = (*)\]
    По теореме Фубини:
    \[(*) = \int_\R\left( \int_\R I\{y + z \le x\}P_\xi(dy) \right) P_\eta(dz) = \int_\R P(\xi \le x - z)P_\eta(dz) = \]
    \[ = \int_\R F_\xi(x - z)dF_\eta(z)\]
\end{proof}

\begin{corollary}[Формула свертки]
    Пусть \(\xi, \eta\) --- независимые случайные величины с плотностями \(p_\xi(x), p_\eta(y)\). Тогда \(\xi + \eta\) тоже имеет плотность, причем:
    \[p_{\xi + \eta} = \int_\R p_\xi(x - y)p_\eta(y)dy\]
\end{corollary}
\begin{proof}
    По лемме о свертке:
    \[F_{\xi + \eta}(x) = \int_\R F_\xi(x - z)dF_\eta(z) = \int_\R\left( \int_{-\infty}^{x - z}p_\xi(y)dy \right)p_\eta(z)dz = \]
    \[= \int_\R\left( \int_{-\infty}^{x}p_\xi(t - z)dt \right)p_\eta(z)dz\]
    По теореме Фубини:
    \[= \int_\R\left( \underbrace{\int_{-\infty}^{x}p_\xi(t - z)p_\eta(z) dt}_{\text{плотность }\xi + \eta}\right)dz\]
\end{proof}

\begin{example}
    Пусть \(\xi_1, \xi_2\) --- независимые случайные величины, \(\xi_i \sim \Gamma(\alpha, \lambda_i), i = 1,2\). Найти плотность \(\xi_1 + \xi_2\)
\end{example}
\begin{solution}
    \[p_{\xi_1 + \xi_2}(x) = \int_\R p_{\xi_1}(x - y)p_{\xi_2}(y)dy = \int_\R\frac{(x - y)^{\lambda_1 - 1}}{\Gamma(\lambda_1)\Gamma(\lambda_2)}\alpha^{\lambda_1 + \lambda_2}y^{\lambda_2 - 1}e^{-\alpha(x - y) - \alpha y}I\{y > 0, x - y > 0\}= \]
    \[ = \frac{e^{-\alpha x}}{\Gamma(\lambda_1)\Gamma(\lambda_2)}\alpha^{\lambda_1 + \lambda_2}\int_0^x(x - y)^{\lambda_1 - 1}y^{\lambda_2 - 1}dy = \left\{z = \frac{y}{x}\right\} = \frac{e^{-\alpha x}}{\Gamma(\lambda_1)\Gamma(\lambda_2)}\alpha^{\lambda_1 + \lambda_2}\int_0^1 (x - xz)^{\lambda_1 - 1}(zx)^{\lambda_2 - 1}xdz = \]
    \[ = \frac{x^{\lambda_1 + \lambda_2 - 1}}{\Gamma(\lambda_1)\Gamma(\lambda_2)}\alpha^{\lambda_1 + \lambda_2}\alpha^{\lambda_1 + \lambda_2}e^{-\alpha x} \underbrace{\int_0^1 (1 - z)^{\lambda_1 - 1}z^{\lambda_2 - 1}dz}_{B(\lambda_1, \lambda_2) = \frac{\Gamma(\lambda_1)\Gamma(\lambda_2)}{\Gamma(\lambda_1 + \lambda_2)}}\]
    Получили, что \(\xi_1 + \xi_2 \sim \Gamma(\alpha, \lambda_1 + \lambda_2)\)
\end{solution}

\subsection{Дисперсия и ковариация}
\begin{definition}
    Пусть \(\xi\) --- случайные величины с конечным \(\E\xi\). Тогда дисперсией случайной величины \(\xi\) называется:
    \[\Variance \xi = \E(\xi - \E\xi)^2\]
\end{definition}

\begin{definition}
    Ковариацией случаных величин \(\xi, \eta\) называется
    \[cov(\xi, \eta) = \E(\xi - \E\xi)(\eta - \E\eta)\]
\end{definition}

\begin{definition}
    Коэффициентом корреляции случайнх величин \(\xi, \eta\) называется
    \[\rho(\xi, \eta) = \frac{cov(\xi, \eta)}{\sqrt{\Variance \xi \cdot \Variance \eta}}\]
\end{definition}

\begin{lemma}[Свойства \(\Variance\) и \(cov\)]
    \begin{enumerate}
        \item Ковариация билинейна
        \item \(\Variance \xi = cov(\xi, \xi)\)
        \item \(cov(\xi, \eta) = \E\xi\eta - \E\xi\cdot \E\eta\), \(\Variance \xi = \E\xi^2 - (\E\xi)^2\)
        \item \(\Variance \xi \ge 0\), причем равенство выполняется тогда и только тогда, когда \(P(\xi = \E\xi) = 1\).
        \item \(c \in \R \Ra \Variance(\xi + c) = \Variance \xi, \Variance(c\xi) = c^2\Variance \xi\).
        \item Неравенство Коши-Буняковского:
        \[|E\xi\eta| \le \sqrt{\E\xi^2 \E\eta^2}\]
        Причем равенство выполняется тогда и только тогда, когда \(\xi, \eta\) линейно независимы почти наверное.
        \item \(|\rho(\xi, \eta)| \le 1\), причем равенство выполняется тогда и только тогда, когда \(\xi - \E\xi, \eta - \E\eta\) линейно независимы почти наверное.
    \end{enumerate}
\end{lemma}

\begin{definition}
    Случайные величины называются некоррелированными, если \(cov(\xi, \eta) = 0\).
\end{definition}

\begin{corollary}
    Если \(\xi, \eta\) независимы, то они некоррелированы
\end{corollary}

\begin{corollary}
    Если \(\xi_1, \dots \xi_n\) --- попарно некоррелированы (или независимые) случайные величины с конечными дисперсиями, то \(\Variance(\xi_1 + \dots + \xi_n) = \sum_{i = 1}^n \Variance \xi_i\)
\end{corollary}
\begin{proof}
    \[\Variance(\xi_1 + \dots + \xi_n) = cov\left( \sum_{i = 1}^n \xi_i, \sum_{i = 1}^n \xi_i \right) = \sum_{i, j \le n} cov(\xi_i, \xi_j) = \sum_{i = 1}^n cov(\xi_i, \xi_i) = \sum_{i = 1}^n \Variance \xi_i\]
\end{proof}

\subsection{Матрица ковариаций}

\begin{definition}
    Пусть $\vec{\xi} = (\xi_1,\dots,\xi_n)$ --- случайный вектор. Матрицей ковариаций (ковариационной матрицей) вектора $\vec{\xi}$ называется матрица $\Sigma = (\sigma_{ij})_{i,j=1}^n$, где $\sigma_{ij} = cov(\xi_i, \xi_j)$.
\end{definition}

\begin{note}
    Диагональные элементы $\sigma_{ii}$ равны $\Variance \xi_i$, а недиагональные $\sigma_{ij}$ ($i \neq j$) равны ковариации $cov(\xi_i,\xi_j)$. Матрица ковариаций часто обозначается $\Variance \vec{\xi}$ или $cov(\vec{\xi})$.
\end{note}

\begin{proposition}
    Любая матрица ковариаций симметрическая и неотрицательно определена
\end{proposition}
\begin{proof}
    Пусть \(\xi\) --- случайный вектор из \(\R^n\). Симметрия \(\Variance \xi\) следует из симметричности ковариации. Пусть \(x_1, \dots x_n \in \R\). Тогда:
    \[\sum_{i, j = 1}^n cov(\xi_i, \xi_j)x_ix_j = \sum_{i, j = 1}^n cov(x_i\xi_i, x_j\xi_j) = cov\left( \sum_{i = 1}^n x_i\xi_i, \sum_{j = 1}^n x_j\xi_j \right) = \Variance\left( \sum_{i = 1}^n x_i\xi_i \right) \ge 0\]
\end{proof}

\subsection{Три неравенства}

\begin{theorem}[Неравенство Маркова]
    Если \(\xi \ge 0\), то \(\forall a > 0\):
    \[P(\xi \ge a) \le \frac{\E\xi}{a}\]
\end{theorem}
\begin{proof}
    Заметим, что
    \[I\{\xi \ge a\} \le \frac{\xi}{a}I\{\xi \ge a\} \le \frac{\xi}{a}\]
    Тогда:
    \[P(\xi \ge a) \le \E\frac{\xi}{a} = \frac{\E\xi}{a}\]
\end{proof}

\begin{theorem}[Неравенство Чебышева]
    Если \(\Variance \xi < +\infty\), то \(\forall \epsilon > 0\):
    \[P(|\xi - \E\xi| \ge \epsilon) \le \frac{\Variance \xi}{\epsilon^2}\]
\end{theorem}
\begin{proof}
    Рассмотрим \(\eta = (\xi - \E\xi)^2\). По неравенству Маркова, получаем:
    \[P((\xi - \E\xi)^2 \ge \epsilon^2) = P(|\xi - \E\xi| \ge \epsilon) \le \frac{\Variance \xi}{\epsilon^2} = \]
\end{proof}

\begin{theorem}[Неравенство Йенсена]
    Если \(\phi\) --- выпуклая снизу функция, то 
    \[\E\phi(\xi) \ge \phi(\E\xi)\]
\end{theorem}
\begin{proof}
    Для \(x \in \R \exists \lambda(x): \forall y \in \R \phi(y) \ge \lambda(x)(y - x) + \phi(x)\). Положим \(y = \xi, x = \E\xi\):
    \[\E\phi(\xi) \ge \E\phi(\E\xi) + \E\lambda(\E\xi)(\xi - \E\xi) = \phi(\E\xi)\]
\end{proof}

\section{Условное математическое ожидание}
Пусть \(\xi, \eta\) --- простые случайные величины. Наша цель --- определить \(\E(\xi | \eta = y)\). Для начала определим условное математическое ожидание для простой случайной величины. Обычное математическое ожидание определялось так:
\[\E\xi = \sum_{i = 1}^n x_i P(\xi = x_i)\]
Поэтому разумно будет определить
\[\E(\xi | \eta = y) = \sum_{i = 1}^n x_i P(\xi = x_i | \eta = y)\]
Теперь рассмотрим случай, когда \(\xi\) --- не простая. Умножим обе части на \(P(\eta = y)\):
\[\E(\xi | \eta = y)P(\eta = y) = \sum_{i = 1}^n x_i P(\xi = x_i | \eta = y)P(\eta = y) = \sum_{i = 1}^n x_i P(\xi = x_i, \eta = y) = \]
\[= \sum_{i = 1}^n x_i \E I\{\xi = x_i, \eta = y\} = \E\left( \left( \underbrace{\sum_{i = 1}^n x_i I\{\xi = x_i\}}_{\xi} \right)I\{\eta = y\} \right) = \E(\xi I\{\eta = y\})\]

Таким образом, мы приходим к такому определнию:
\begin{definition}
    Пусть \((\Omega, \mathcal{F}, P)\) --- вероятностное пространство, \(\xi\) --- случайная величина на нем, \(\mathcal{C} \subset \mathcal{F}\) --- под-\(\sigma\)-алгебра. Условным матожиданием \(\xi\) относительно \(\sigma\)-алгербы \(\mathcal{C}\) называется случайная величина \(\E(\xi | \mathcal{C})\), обладающая двумя свойствами:
    \begin{enumerate}
        \item \textbf{Свойство измеримости:} \(\E(\xi | \mathcal{C})\) является \(\mathcal{C}\)-измеримой
        \item \textbf{Интегральное свойство:} \(\forall A \in \mathcal{C}\) выполнено:
        \[\E(\xi \cdot I_A) = \E(\E(\xi | \mathcal{C}) \cdot I_A)\]
    \end{enumerate}
\end{definition}

\begin{note}
    \(\xi \ne \E(\xi|\mathcal{C})\) в общем случае, т.к. \(\xi\) может не быть \(\mathcal{C}\)-измеримой
\end{note}

\begin{theorem}[О существовании условного математического ожидания]
    Если \(\E|\xi| < +\infty\), то \(\forall \mathcal{C} \subset \mathcal{F}\) --- под-\(\sigma\)-алгербы \(\E(\xi | \mathcal{C})\) существует и единственно с точностью до множества вероятности 0.
\end{theorem}
\begin{proof}
    Рассмотрим вероятностное пространство \((\Omega, \mathcal{C}, P)\). Функция \(Q(A) = \E(\xi \cdot I_A), A \in \mathcal{C}\) является зарядом на \((\Omega, \mathcal{C}, P)\), причем абсолютно непрервным относительно \(P\). По теореме Радона-Никодима, \(\exists !\) с точностью до множества вероятности 0 случайная величина \(\eta \in L^1(\Omega, \mathcal{C}, P)\), такая, что \(\forall A \in \mathcal{C}\):
    \[Q(A) = \int_A\eta dP = \E(\eta \cdot I_A)\]
\end{proof}

\begin{lemma}
    Если \(\E|\xi| < +\infty\) и \(\mathcal{C}\) порождена разбиением \(\{D_n | n \in \N\}\), то
    \[\E(\xi | \mathcal{C})(\omega) = \sum_{n = 1}^\infty \frac{\E(\xi \cdot I_{D_n})}{P(D_n)}I_{D_n}(\omega)\]
\end{lemma}
\begin{proof}
    Проверим, что правая часть удовледвторяет определению. Очевидно, что правая часть \(\mathcal{C}\)-измерима, как линейная комбинация несовместных индикаторов.

    Если \(A \in \mathcal{C}\), то \(A = \bigsqcup_{k = 1}^\infty D_{i_k}\), поэтому достаточно проверить интегральное свойство для \(A\) вида \(D_m, m \in \N\).
    \[\E\left( \left( \sum_{n = 1}^\infty \frac{\E(\xi I_{D_n})}{P(D_n)} I_{D_n}\right) I_{D_m}\right) = \E\left( \frac{\E(\xi I_{D_m})}{P(D_m)} I_{D_m}\right) = \frac{\E(\xi I_{D_m})}{P(D_m)} P(D_m) = \E(\xi I_{D_m})\]
\end{proof}

\begin{proposition}[Линейность]

\end{proposition}
\begin{proof}
    Очевидно, что правая часть \(\mathcal{C}\)-измерима как линейная комбинация \(\mathcal{C}\)-измеримых случайных величин. Проверим интегральное свойство: \(\forall A \in \mathcal{C}\):
    \[\E((a\xi + b\eta) I_A) = a\E(\xi I_A) + b \E(\eta I_A)= (*)\]
    По интегральному свойству, получаем:
    \[(*) = a\E(\E(\xi|\mathcal{C})I_A) + b\E(\E(\eta|\mathcal{C})I_A) = \E(a\E(\xi|\mathcal{C}) + b\E(\eta|\mathcal{C}) I_A)\]
\end{proof}

\begin{proposition}[Формула полной вероятности]
    \[\E(\E(\xi|\mathcal{C})) = \E\xi\]
\end{proposition}
\begin{proof}
    Возьмем \(A = \Omega \in \mathcal{C}\) и применим интегральное свойство:
    \[\E\xi = \E(\xi I_A) = \E(\E(\xi|\mathcal{C})I_A) = \E(\E(\xi|\mathcal{C}))\]
\end{proof}

\begin{proposition}

\begin{proof}

    \[\E(\xi I_A) = \E\xi \E I_A = \E(\E \xi I_A)\]
\end{proof}

\begin{proposition}[Сохранение отношение порядка]
    Если \(\xi \le \eta\) почти наверное, то \(\E(\xi|\mathcal{C}) \le \E(\eta|\mathcal{C})\)
\end{proposition}
\begin{proof}
    Заметим, что \(\forall A \in \mathcal{C}\):
    \[\E(\xi I_A) \le \E(\eta I_A)\]

    \[\E(\E(\xi|\mathcal{C})I_A) \le \E(\E(\eta|\mathcal{C})I_A)\]
    Но тогда по свойству матожидания в \((\Omega, \mathcal{F}, P)\), получаем, что \(\E(\xi|\mathcal{C}) \le \E(\eta|\mathcal{C})\)
\end{proof}

\begin{proposition}

\end{proposition}
\begin{proof}

\end{proof}

\begin{proposition}[Телескопическое свойство]

        \item \(\E(\E(\xi|\mathcal{C_1})|\mathcal{C_2}) = \E(\xi|\mathcal{C_1})\)
        \item \(\E(\E(\xi|\mathcal{C_2})|\mathcal{C_1}) = \E(\xi|\mathcal{C_1})\)
    \end{enumerate}
\end{proposition}
\begin{proof}

        \item \(\E(\xi|\mathcal{C_1})\) является \(\mathcal{C_2}\)-измеримой, далее применяем первое свойство
        \item \(\E(\xi|\mathcal{C_1})\) является \(\mathcal{C_1}\)-измеримой по определению. Проверим интегральное свойство: \(\forall A \in \mathcal{C_1}\):
        \[\E(\E(\xi|\mathcal{C_2})I_A) = (*)\]
        Заметим, что \(A \in \mathcal{C_1} \subset \mathcal{C_2}\), поэтому по интегральному свойству для \(\mathcal{C_2}\):
        \[(*) = \E(\xi I_A) = (*)\]
        Далее применяем интегральное свойство для \(\mathcal{C_1}\):
        \[(*) = \E(\E(\xi|\mathcal{C_1})I_A)\]
    \end{enumerate}
\end{proof}

\begin{proposition}[Предельный переход]
    \begin{enumerate}
        \item Пусть \(\{\xi_n, n \in \N\}, \xi\) таковы, что \(0 \le \xi_n \uparrow \xi\). Тогда \(\E(\xi_n|\mathcal{C}) \ra \E(\xi|\mathcal{C})\) почти наверное.
        \item Пусть \(\{\xi_n, n \in \N\}, \xi\) таковы, что \(\forall n \in \N: |\xi_n| \le \delta, \E\delta \le +\infty, \xi_n \ra \xi\) почти навреное. Тогда \(\E(\xi_n|\mathcal{C}) \ra \E(\xi|\mathcal{C})\) почти наверное.
    \end{enumerate}
\end{proposition}
\begin{proof}
    \begin{enumerate}
        \item Согласно свойству сохранения порядка, \(\E(\xi_n|\mathcal{C}) \le \E(\xi_{n + 1}|\mathcal{C})\). Но тогда почти наверное \(\exists \lim_{n} \E(\xi_n|\mathcal{C}) = \eta\). Проверим, что \(\eta = \E(\xi|\mathcal{C})\). 
        
        Случайная величина \(\eta\) является \(\mathcal{C}\)-измеримой, как предел \(\mathcal{C}\)-измеримых случайных величин. Проверим интегральное свойство:
        \[\forall A \in \mathcal{C}: 0 \le \xi_n I_A \uparrow \xi I_A, 0 \le \E(\xi_n|\mathcal{C})I_A \uparrow \eta I_A\]
        По теореме о монотонной сходимости:
        \[\E(\xi I_A) = \lim_n \E(\xi_n I_A) = (*)\]
        По интегральному свойству, получаем:
        \[(*) = \lim_n \E(\E(\xi_n|\mathcal{C})I_A) = \E(\eta I_A)\]

        \item Рассмотрим \(\psi_n = \sup_{k \ge n} |\xi_n - \xi|\). Тогда \(\xi_n \downarrow 0\) почти наверное, \(|\psi_n| \le 2\delta\) почти наверное \(\Ra\) применим пункт 1.
        \[|\E(\xi_n|\mathcal{C}) - \E(\xi|\mathcal{C})| \le \E(\psi_n|\mathcal{C}) \ra_{n \ra \infty} \E(0|\mathcal{C}) = 0\]
        Последнее равенсво выполняется почти наверное
    \end{enumerate}
\end{proof}

\begin{proposition}[Неравенство Йенсена]
    Пусть \(\phi(x)\) --- выпуклая вниз функция, тогда:
    \[\E(\phi(\xi)|\mathcal{C}) \ge \phi(\E(\xi|\mathcal{C}))\]
\end{proposition}
\begin{proof}
    В силу выпуклости \(\phi\), \(\forall x \in \R \exists \lambda(x): \forall y \in \R: \phi(y) \ge \phi(x) + \lambda(x)(y - x)\). Подставим \(y = \xi, x = \E(\xi|\mathcal{C})\) и возьмем условное математическое ожидание относительно \(\mathcal{C}\):
    \[\E(\phi(\xi)|\mathcal{C}) \ge \E(\phi(\E(\xi|\mathcal{C}))|\mathcal{C}) + \E(\lambda(\E(\xi|\mathcal{C}))(\xi - \E(\xi|\mathcal{C})) | \mathcal{C}) = \]
    \[= \phi(\E(\xi|\mathcal{C})) + \underbrace{\lambda(\E(\xi|\mathcal{C}))\E(\xi - \E(\xi|\mathcal{C}) | \mathcal{C})}_{= 0} = \phi(\E(\xi|\mathcal{C}))\]
\end{proof}

\section{Условное распределение}
\begin{definition}
    Пусть \(\xi\) --- случайная величина, \(\eta\) --- случайный вектор из \(\R^m\) на \((\Omega, \mathcal{F}, P)\). Условным распределенеим \(\xi\) относительно \(\eta\) называется \(P(B, y), B \in \mathcal{B}(\R), y \in \R^m\), такая, что:
    \begin{enumerate}
        \item \(\forall B \in \mathcal{B}(\R^m)\) функция \(P(B, \cdot)\) --- борелевская
        \item \(\forall y \in \R^m \) функция \(P(\cdot, y)\) является вероятностной мерой на \((\R, \mathcal{B}(\R))\).
        \item \(\forall B \in \mathcal{B}(\R), A \in \mathcal{B}(\R^m)\) выполняется:
        \[P(\xi \in B, \eta \in A) = \int_AP(B, y)P_\eta(dy)\]
    \end{enumerate}
\end{definition}

\begin{note}
    По сути \(P(B, y) = P(\xi \in B | \eta = y)\). Однако, если определить \(P(B, y) = \E(I\{\xi \in B\} | \eta = y)\) то, будут выполнены только свойства 1, 3. Т.к. \(\E\) определено неоднозначно, то свойство 2 может не выполняться и с этим надо что-то делать
\end{note}

\begin{note}
    \(\E(\xi|\eta = y)\) удовлетворяет естественному обобщению свойств 1-10 условного математического ожидания
\end{note}

\begin{theorem}
    Условное распределение существует
\end{theorem}
\begin{proof}
    Для каждого \(x \in \R\) рассмотрим \(F(x|y) = \E(I\{\xi \le x\} | \eta = y)\), причем из всех функций (определенных с точностью до множества меры 0) выберем в правой части любую. Пусть \(\Q = \{q_n, n \in \N\}\). Если \(q_i < q_j\) то введем множество: \(A_{ij} = \{y: F(q_i|y) > F(q_j|y)\}\) --- борелевское множество и \(P_\eta(A) = 0\). Для \(q_i \in \Q\) введем \(B_i = \{y: \lim_{n \ra \infty} F\left( q_i + \frac{1}{n}|y\right) \ne F(q_i|y)\}\). В силу того, что \(I\left\{\xi \le q_i + \frac{1}{n}\right\} \ra_{n \ra \infty} I\{\xi \le q_i\} \Ra\) по свойству 8, \(F\left( q_i + \frac{1}{n} | y \right) \ra F(q_i | y)\), \(P_\eta\)-п.н. \(\Ra P_\eta(B_i) = 0\). Наконец, введем \(C = \{y: \lim_{n \ra \infty} F(n|y) \ne 1\text{ или }\lim_{n \ra \infty}F(-n|y) \ne 0\}\). Снова, \(I\{\xi \le n\} \uparrow 1, I\{\xi \le -n\} \downarrow 0 \Ra\) по свойству 8, \(F(n|y) \ra 1, F(-n|y) \ra 0\), \(P_\eta\)-п.н. \(\Ra P_\eta(C) = 0\). Далее, введем
    \[\tilde{F}(x|y) = \left\{\begin{array}{l}
        \lim_{q \downarrow x, q \in \Q} F(q|y), y \notin D \\
        G(x), \text{ иначе}
    \end{array}\right.\]
    Где \(D = C \cup \left( \bigcup_{i} B_i \right) \cup \left( \bigcup_{i,j} A_{ij} \right)\), \(G(x)\) --- фиксированная функция распределения на \(\R\).

    Проверим, что для \(\forall y \in \R^m, \tilde{F}(x|y)\) --- это функция распределения на \(\R\). Если \(y \in D\) --- все очевидно. Если \(y \notin D\), то \(F(q|y)\) является неубывающей, непрерывной справа функцией, имеющей нужные пределы на \(\pm\infty\). Тогда \(\tilde{F}(x|y)\) будет совпадать с \(F(x|y)\), если \(x \in \Q\), и будет удовлетворять свойствам функции распределения. 

    Далее, проверим, что \(\forall x \in \R \tilde{F}(x|y) = F(x|y) = \E(I\{\xi \le x\}|\eta = y)\) почти наверное. Если \(x \in \Q\), то \(\tilde{F}(x|y) = F(x|y)\) по построению. Если \(x \notin \Q\), то рассмотрим последовательность \(x_n \downarrow x, x_n \in \Q\). Тогда \(I\{\xi \le x_n\} \ra I\{\xi \le x\} \Ra\) по свойству 8 \(P_\eta\)-п.н:
    \[\E(I\{\xi \le x\}|\eta = y) = \lim_{n \ra \infty} \E(I\{\xi \le x_n\} | \eta = y) = \lim_{n \ra \infty} F(x_n|y) = \tilde{F}(x|y)\]
    
    \(\Ra \tilde{F}(x|y)\) --- это вариант условного математического ожидания \(\E(I\{\xi \le x\}|\eta = y)\). Далее для \(\forall y \in \R^m\) рассмотрим \(Q(\cdot | y)\) --- вероятностная мера на \(\R\), соответствующая \(\tilde{F}(x|y)\). Проверим, что \(\forall B \in \mathcal{B}(\R)\), \(P_\eta\)-п.н. \(Q(B|y) = \E(I\{\xi \in B\}|\eta = y)\). Введем \(\mathcal{L} = \{B \in \mathcal{B}(\R): Q(B|y) \E(I\{\xi \in B\} | \eta = y), P_\eta\text{-п.н.}\}\)

    Докажем, что \(\mathcal{L}\) --- это \(\lambda\)-система.
    \begin{enumerate}
        \item \(\R \in \mathcal{L}\), т.к. \(Q(\R|y) = 1 = \E(1|\eta = y)\)
        \item Если \(A, B \in \mathcal{L}, A \subset B\), то \(B \setminus A \in \mathcal{L}\), т.к. \(Q(B\setminus A|y) = Q(B|y) - Q(A|y) = P_\eta\)-п.н \(= \E(I\{\xi \in B\}|\eta = y) - \E(I\{\xi \in A\}|\eta = y) = \E(I\{\xi \in B\setminus A\}|\eta = y)\).
        \item Если \(A_n \uparrow A, A_n \in \mathcal{L}\), то \(A \in \mathcal{L}\), т.к. \(Q(A|y) = \lim_{n \ra \infty} Q(A_n|y) = P_\eta\)-п.н \(= \lim_{n \ra \infty} \E(I\{\xi \in A_n\}|\eta = y) = (*)\). По свойству 8, получаем, что \((*) = \E(I\{\xi \in A\}|\eta = y)\).
    \end{enumerate}

    Итого, \(\mathcal{L}\) --- \(\lambda\)-система, содержащая все полуинтервалы \((-\infty, x] \Ra\) по второй теореме о \(\lambda\)-системах, \(\mathcal{L}\) содержит \(\mathcal{B}(\R)\). Тогда \(Q(B|y)\) удовлетворяет свойствам 1, 3, т.к. это вариант \(\E(I\{\xi \in B\} | \eta = y)\) и свойству 2 по построению.
\end{proof}

\begin{theorem}[О вычислении условного математического ожидания]
    Пусть \(\xi\) --- случайная величина, \(\eta\) --- случайный вектор из \(\R^m\), \(P(B, y)\) --- условное распределение \(\xi\) относительно \(\eta\). Тогда для любой борелевской функции \(f: \R \ra \R\) выполнено
    \[\E(f(\xi)|\eta = y) = \int_\R f(x) P(dx, y)\]
\end{theorem}
\begin{proof}
    Пусть \(f(x) = I_B, B \in \mathcal{B}(\R)\). Тогда
    \[\E(f(\xi)|\eta = y) = \E(I\{\xi \in B\}|\eta = y) = P(B, y) = \int_\R I_B(x) P(dx, y) = \int_\R f(x) P(dx, y)\]
    Обе части требуемого линейны по \(f \Ra\) равенство выполнено для простых функций. Для произвольных функций переходим к пределу от простых.
\end{proof}

\begin{definition}
    Пусть \(\xi\) --- случайная величина, \(\eta\) --- случайный вектор из \(\R^m\). Функция \(f_{\xi|\eta}(x|y)\) называется условной плотностью \(\xi\) относительно \(\eta\), если она есть плотность условного распределения (по мере \(\mu\)):
    \[\forall B \in \mathcal{B}(\R): P(\xi \in B|\eta = y) = \int_B f_{\xi|\eta}(x|y)\mu(dx)\]
\end{definition}

\begin{corollary}
    При наличии условной плотности:
    \[\E(g(\xi)|\eta = y) = \int_\R g(x) f_{\xi|\eta}(x|y)\mu(dx)\]
\end{corollary}

\begin{theorem}[Достаточное условие существования условной плотности]
    Пусть случайный вектор \((\xi, \eta)\) имеет совместную плотность \(p_{\xi, \eta}(x, y)\) по мере \(\lambda \times \mu\). Обозначим
    \[p_{\eta}(y) = \int_\R p_{\xi, \eta}(x, y) \lambda(dx)\]
    Тогда функция 
    \[p_{\xi|\eta} = \left\{\begin{array}{l}
        \frac{p_{\xi, \eta}(x, y)}{p_\eta(y)}, p_\eta(y) > 0 \\
        g(x), \text{ иначе}
    \end{array}\right.\]
    Является условной плотностью \(\xi\) относительно \(\eta\) по мере \(\lambda\) (здесь \(g(x)\) --- произвольная фиксированная плотность по \(\lambda\))
\end{theorem}
\begin{proof}
    Надо проверить, что 
    \[P(B, y) = \int_B p_{\xi|\eta}(x|y)\lambda(dx)\]
    Является условным распределением \(\xi\) относительно \(\eta\). Измеримость по \(y\) обеспечивается свойствами интеграла (теорема Фубини). Свойство меры по \(B\) обеспечивается тем фактом, что \(p_{\xi|\eta} \ge 0\) и интегрируется в 1 по \(\R\). Получаем, что это всегда плотность. Проверим последнее свойство:
    \[P(\xi \in B, \eta \in A) =\]
    Из совместной плотности получаем:
    \[ = \iint_{B \times A} p_{\xi, \eta}(x, y)\lambda(dx)\mu(dy) = \]
    Применяя теорему Фубини, получем:
    \[\int_A\left( \int_B p_{\xi, \eta}(x, y) \lambda(dx) \right) \mu(dy) = \int_A\left( \int_B p_{\xi|\eta}(x|y) \lambda(dx) \right) p_{\eta}(y)\mu(dy) = \int_A P(B, y)P_{\eta}(dy)\]
\end{proof}

\begin{note}[Схема подсчета условного матожидания]
    Пусть мы хотим посчитать \(\E(f(\xi)|\eta)\). Если есть совметная плотность \(p_{\xi, \eta}(x, y)\), то находим условную плотность \(p_{\xi|\eta}(x|y)\). После этого, вычисляем:
    \[\E(f(\xi)|\eta) = \int_\R f(x) p_{\xi|\eta}(x|y)dx\]
    После этого находим \(\phi(y)\). Ответом будет \(\phi(\eta) = \E(f(\xi)|\eta) = \E(f(\xi)|\mathcal{F}_\eta)\)
\end{note}

\begin{problem}
    Пусть \(X, Y \sim Exp(1)\) --- независимые. Требуется найти \(\E(X^2|X + Y)\)
\end{problem}
\begin{solution}
    Найдем совместную плотность \((X, X + Y)\):
    \[P((X, X + Y) \in B) = \iint_{(x, y): (x, x + y) \in B} p_X(x)p_Y(y)dxdy = \]
    \[\left\{\begin{array}{cc}
        \alpha = x & x = \alpha \\
        \beta = x + y & y = \beta - \alpha
    \end{array}, |J| = \left| \begin{array}{cc}
        1 & 0 \\
        =1 & 1
    \end{array} \right| = 1\right\} = \]
    \[= \iint_{B} \underbrace{e^{-\beta}I\{0 < \alpha < \beta\}}_{p_{X, X + Y}(\alpha, \beta)}d\alpha d\beta\]
    Получаем, что:
    \[p_{X + Y}(\beta) = \int_\R e^{-\beta}I\{0 < \alpha < \beta\} d\alpha = e^{-\beta} \int_0^\beta d\alpha = \beta e^{-\beta} I\{\beta > 0\}\]
    Тогда:
    \[p_{X|X + Y}(\alpha|\beta) = \frac{1}{\beta}I\{0 < \alpha < \beta\}\]
    И заключаем:
    \[\E(X^2 | X + Y = \beta) = \int_0^\beta \frac{\alpha^2}{\beta} d\alpha = \frac{\beta^2}{3}\]
    Получаем, что \(\E(X^2 | X + Y) = \frac{(X + Y)^2}{3}\)
\end{solution}

\section{Сходимости случайных величин}
Пусть \(\{\xi_n, n \in \N\}\) --- случайные велиичны на \((\Omega, \mathcal{F}, P)\). Последовательность \(\xi_n\) сходится к \(\xi\)
\begin{definition}
    \textit{С вероятностью 1 (почти наверное)}, если \(P(\lim_{n \ra \infty} \xi_n = \xi) = 1\)
    Обозначения: \(\xi_n \xrightarrow{\text{п.н.}} \xi, \xi_n \ra \xi\) п.н., \(\xi_n \ra \xi\) \(p\)-п.н.
\end{definition}

\begin{definition}
    \textit{По вероятности}, если \(\forall \epsilon > 0 P(|\xi_n - \xi| > \epsilon) \ra 0, n \ra \infty\)
    Обозначение: \(\xi_n \xrightarrow{P} \xi\)
\end{definition}

\begin{definition}
    \textit{В среднем порядка \(p\)}, если \(\E|\xi_n - \xi|^p \ra 0, n \ra \infty\)
    Обозначение: \(\xi_n \xrightarrow{L^p} \xi\)
\end{definition}

\begin{definition}
    \textit{По распределению}, если \(\forall f: \R \ra \R\) --- ограниченной непрерывной функции \(\E f(\xi_n) \ra \E f(\xi), n \ra \infty\)
    Обозначение: \(\xi_n \xrightarrow{d} \xi\)
\end{definition}

\begin{theorem}[Критерий сходимости почти наверное]
    \(\xi_n \xrightarrow{\text{п.н.}} \xi \Lra \forall \epsilon > 0: P\left( \sup_{k \ge n} |\xi_n - \xi| > \epsilon \right) \ra 0, n \ra \infty\)
\end{theorem}
\begin{proof}
    Было на ОВиТМе
\end{proof}

\begin{corollary}
    Если для \(\forall \epsilon > 0\) сходится ряд \(\sum_{n = 1}^\infty P(|\xi_n - \xi| \ge \epsilon)\), то \(\xi_n \xrightarrow{\text{п.н.}} \xi\).
\end{corollary}
\begin{proof}
    \[P\left( \sup_{k \ge n} |\xi_n - \xi| > \epsilon \right) = P\left( \bigcup_{k = n}^\infty \{|\xi_n - \xi| > \epsilon\} \right) \le \sum_{k = n}^\infty P(|\xi_n - \xi| > \epsilon) \ra 0\]
    Последний предел равен 0, как остаток сходящегося ряда.
\end{proof}

\begin{corollary}
    Если \(\xi_n \xrightarrow{P} \xi\), то \(\exists\) подпоследовательность \(\{\xi_{n_k}, k \in \N\}\), такая, что \(\xi_n \xrightarrow{\text{п.н.}} \xi\)
\end{corollary}
\begin{proof}
    Пусть \(n_1 = 1\). Если \(n_1, \dots n_{k - 1}\) выбраны, то возьмем \(n_k\) так, что \(n_k > n_{k - 1}\) и 
    \[P\left( |\xi_n - \xi| \ge \frac{1}{k} \right) \le \frac{1}{2^k}\]
    Если \(\epsilon > 0\) дано, то возьмем \(k_0\) такое, что \(\epsilon > \frac{1}{k_0}\) и 
    \[\sum_{k = k_0}^\infty P(|\xi_n - \xi| \ge \epsilon) \le \sum_{k = k_0}^\infty P\left(|\xi_n - \xi| \ge \frac{1}{k}\right) < +\infty\]
\end{proof}

\begin{theorem}[О взаимодействии видов сходимости]
    \begin{enumerate}
        \item \(\xi_n \xrightarrow{\text{п.н.}} \xi \Ra \xi_n \xrightarrow{P} \xi\)
        \item \(\xi_n \xrightarrow{L^p} \xi \Ra \xi_n \xrightarrow{P} \xi\)
        \item \(\xi_n \xrightarrow{P} \xi \Ra \xi_n \xrightarrow{d} \xi\)
    \end{enumerate}
\end{theorem}
\begin{proof}
    \begin{enumerate}
        \item \(P(|\xi_n - \xi| \ge \epsilon) \le P\left(\sup_{k \ge n} |\xi_k - \xi| > \frac{\epsilon}{2}\right) \ra 0, n \ra +\infty\) по критерию сходимости почти наверное.
        \item \(P(|\xi_n - \xi| \ge \epsilon) = P(|\xi_n - \xi|^p \ge \epsilon^p) = (*)\). По неравенству Маркова, получаем \((*) \le \frac{\E|\xi_n - \xi|^p}{\xi^p} \ra 0, n \ra \infty\).
        \item Пусть \(f: \R \ra \R\) --- ограниченная непрерывная функция. Пусть \(\forall x \in \R: |f(x)| \le M\). Для любого \(\epsilon > 0\) выберем \(N > 0\) так, что \(P(|\xi| > N) \le \frac{\epsilon}{4M}\). Выберем \(\delta > 0\), так, что \(\forall x, y\) с условием \(|x - y| \le \delta, |x| \le N\) выполнено \(|f(x) - f(y) \le \frac{\epsilon}{2}|\). Далее:
        \[|\E f(\xi_n) - \E f(\xi)| \le \E |f(\xi_n) - f(\xi)| = \E(|f(\xi_n) - f(\xi)|(I_{A_1} + I_{A_2} + I_{A_3}))\]
        Где \(A_1 = \{|\xi| > N\}, A_2 = \{|\xi| \le N, |\xi_n - \xi| > \delta\}, A_3 = \{|\xi| \le N, |\xi_n - \xi| < \delta\}\). Тогда:
        \[\E |f(\xi_n) - f(\xi)| = \E(|f(\xi_n) - f(\xi)|I_{A_1}) \le 2M P(A_1) \le 2M \frac{\epsilon}{4M} = \frac{\epsilon}{2}\]
        \[\E |f(\xi_n) - f(\xi)| = \E(|f(\xi_n) - f(\xi)|I_{A_2}) \le 2M P(A_2) \le 2M P(|\xi_n - \xi|) \ra 0, n \Ra \infty\]
        Последнее верно в силу того, что \(\xi_n \xrightarrow{P} \xi\). Наконец:
        \[\E |f(\xi_n) - f(\xi)| = \E(\underbrace{|f(\xi_n) - f(\xi)|}_{\frac{\epsilon}{2}}\underbrace{I_{A_3}}_{\le 1})\]
        Получили, что \(\E f(\xi_n) \ra \E f(\xi)\)
    \end{enumerate}
\end{proof}

\section{Законы больших чисел}

\subsection{ЗБЧ в форме Чебышева}

\begin{theorem}[Закон больших чисел в форме Чебышева]
    Пусть \(\{\xi_n, n \in \N\}\) --- попарно некоррелированные случайные величины, такие, что \(\exists c: \forall n \Variance \xi_n \le c\). Положим \(S_n = \xi_1 + \dots + \xi_n\). Тогда \(\forall \delta > 0\):
    \[\frac{S_n - \E S_n}{n^{\frac{1}{2} + \delta}} \xrightarrow{P} 0, n \ra \infty\]
\end{theorem}
\begin{proof}
    Для \(\epsilon > 0\):
    \[P\left( \left| \frac{S_n - \E S_n}{n} \right| \ge \epsilon\right) = (*)\]
    Применим неравенство Чебышева, получаем:
    \[(*) \le \frac{\Variance S_n}{\epsilon^2 n^{1 + 2\delta}} =\]
    Из некоррелированности получаем:
    \[= \frac{\sum_{k = 1}^n \Variance \xi_k}{\epsilon^2 n^{1 + 2\delta}} \le \frac{n c}{\epsilon^2 n^{1 + 2\delta}} = O\left( \frac{1}{n^{2\delta}} \right) \ra 0, n \ra \infty\]
\end{proof}

\begin{note}
    Закон больших чисел обосновывает принцип устойчивости частот: при большом числе испытаний среднее арифметическое наблюдаемых значений (например, частота события) сходится к математическому ожиданию. Это означает, что случайные отклонения в среднем гасятся, и результат становится предсказуемым.
\end{note}

\begin{theorem}[Усиленный закон больших чисел для некоррелированых случайнх величин]
    Пусть \(\{\xi_n, n \in \N\}\) --- попарно некоррелированные случайные величины, такие, что \(\exists c: \forall n \Variance \xi_n \le c\). Положим \(S_n = \xi_1 + \dots + \xi_n\). Тогда:
    \[\frac{S_n - \E S_n}{n} \xrightarrow{\text{п.н.}} 0, n \ra \infty\]
\end{theorem}

\begin{proof}
    Без ограничения общности, считаем, что \(\E\xi_n = 0\), иначе рассмотрим \(\xi_n' = \xi_n - \E\xi_n\). Проверим сначала, что \(\frac{S_{n^2}}{n^2} \xrightarrow{\text{п.н.}} 0\). Для \(\epsilon > 0\):
    \[P\left( \left| \frac{S_{n^2}}{n^2} \right| \ge \epsilon \right) \le \]
    Применяем неравенство Чебышева и некоррелированность:
    \[\le \frac{\Variance S_{n^2}}{n^2} = \frac{\sum_{i = 1}^{n^2}}{\Variance \xi_i}{\epsilon^2 n^4} \le \frac{n^2 c}{\epsilon^2 n^4} = O\left( \frac{1}{n^2} \right)\]
    Тогда 
    \[\sum_{n} P\left( \left| \frac{S_{n^2}}{n^2} \right| \ge \epsilon \right) < +\infty\]
    По достаточному условию получаем, что \(\frac{S_{n^2}}{n^2} \xrightarrow{\text{п.н.}} 0\). Обозначим \(m(n) = [\sqrt{n}]\). Тогда в силу доказанного:
    \[\frac{S_{m(n)^2}}{m(n)^2} \xrightarrow{\text{п.н.}} 0\]
    Теперь проверим, что \(\frac{S_n - S_{m(n)^2}}{m(n)^2} \xrightarrow{\text{п.н.}} 0\). Для \(\epsilon > 0\):
    \[P\left( \left| \frac{S_n - S_{m(n)^2}}{m(n)^2} \right| \ge \epsilon \right) \le \]
    Применяем неравенство Чебышева и некоррелированность:
    \[\frac{\Variance(S_n - S_{m(n)^2})}{\epsilon^2 m(n)^4} = \frac{\sum_{i = m(n)^2 + 1}^n \Variance \xi_n}{\epsilon^2 m(n)^4} \le \frac{c(n - (\sqrt{n} - 1)^2)}{\epsilon^2 m(n)^4} \le \frac{2c\sqrt{n}}{\epsilon m(n)^4} = O\left( n^{-\frac{3}{2}} \right)\]
    Тогда

\subsection{ЗБЧ в форме Этемади}

\begin{definition}
    Пусть \(\{A_n, n \in \N\}\) --- последовательность событий. События \(\{A_n, \text{б.ч.}\}\) называется событие, элеменатрные исходы которого встречаются бесконечно много раз в последовательности \(\{A_n\}\). Формально:
    \[\{A_n, \text{б.ч.}\} = \bigcap_{n = 1}^\infty \bigcup_{m \ge n} A_m\]
\end{definition}

\begin{lemma}[Борель-Кантелли]
    \begin{enumerate}
        \item Если \(\sum_{n = 1}^\infty P(A_n) < +\infty\), то \(P(\{A_n, \text{б.ч.}\}) = 0\).
        \item Если же \(\sum_{n = 1}^\infty P(A_n) = +\infty\) и \(A_n\) независимы, то \(P(\{A_n, \text{б.ч.}\}) = 1\).
    \end{enumerate}
\end{lemma}
\begin{proof}
    \begin{enumerate}
        \item \[P(\{A_n, \text{б.ч.}\}) = P\left( \bigcap_{n = 1}^\infty \bigcup_{m \ge n} A_m \right) = \lim_{n \ra \infty} P\left( \bigcup_{m \ge n} A_n\right) \le \lim_{n \ra \infty} \sum_{m = n}^\infty P(A_n) = 0\]
        \item \[P(\{A_n, \text{б.ч.}\}) = \lim_{n \ra \infty} P\left( \bigcup_{m \ge n} A_m\right) = \lim_{n \ra \infty} \left( 1 - P\left( \bigcap_{m \ge n} \overline{A_m}\right) \right) = 1 - \lim_{n \ra \infty} P\left( \bigcap_{m \ge n} \overline{A_m}\right) =\]
        \[= 1 - \lim_{n \ra \infty} \lim_{N \ra \infty} P\left( \bigcap_{m = n}^N \overline{A_m}\right) = 1 - \lim_{n \ra \infty} \lim_{N \ra \infty} \prod_{m = n}^N (1 - P(A_m)) \ge 1 - 1 - \lim_{n \ra \infty} \lim_{N \ra \infty} e^{\sum_{m = n}^N P(A_m)}\]

    \end{enumerate}
\end{proof}

\begin{definition}
    Пусть \(\{\xi_n, n \in \N\}\) --- последовательность случайных величин. Сигма алгеброй \(\mathcal{F}_{\ge k}\) называтся 
    \[\mathcal{F}_{\ge k} = \sigma\left( \bigcup_{n = k}^\infty \mathcal{F}_{\xi_n}\right)\]

    Хвостовой \(\sigma\)-алгеброй последовательности \(\{\xi_n, n \in \N\}\) называется:
    \[\mathcal{X} = \bigcap_{i = 1}^\infty \mathcal{F}_{\ge k}\]
\end{definition}

\begin{theorem}[Закон 0 или 1 Колмогорова]
    Если \(\{\xi_n, n \in \N\}\) --- независимые случаные величины, то для \(\forall A \in \mathcal{X}\) выполнено \(P(A) = 0\) или \(1\)
\end{theorem}
\begin{proof}
    Докажем, что \(A\) независимо с \(A\). Рассмотрим \(\sigma\)-алгебру 
    \[\mathcal{F}_{\le n} = \mathcal{F}_{(\xi_1, \dots \xi_n)} = \sigma(\xi_1, \dots \xi_n)\]
    Обозначим \(\mathcal{A} = \bigcup_{n = 1}^\infty \mathcal{F}_{\le n}\). Тогда \(\mathcal{A}\) --- алгебра \(\Ra \pi\)-система, причем \(\sigma(\mathcal{A}) = \mathcal{F}_{ge 1}\). В силу независимости \(\{\xi_n\}\), получаем, что \(\mathcal{F}_{\le n}\) независимо с \(\mathcal{F}_{\ge n + 1}\). Тогда \(\forall n: A \in \mathcal{F}_{\ge n + 1}\), \(\Ra \forall n: A\) независимо с \(\mathcal{F}_{\le n}\). Тогда \(A\) независимо с \(\mathcal{A}\). Но тогда по критерию независимости \(\sigma\)-алгебр получаем, что \(A\) независимо с \(\sigma(\mathcal{A}) = \mathcal{F}_{\ge 1}\). Но \(A \in \mathcal{F}_{\ge 1} \Ra A\) независимо с \(A\).
\end{proof}

\begin{proposition}[Критерий конечности \(\E\)]
    Пусть \(\xi\) --- случайные величины. Тогда \(\E\xi\) конечно \(\Lra \sum_{n = 1}^\infty P(|\xi| \ge n) < +\infty\) 
\end{proposition}
\begin{proof}
    \[\sum_{n = 1}^\infty P(|\xi| \ge n) = \sum_{n = 1}^\infty \sum_{k = n}^\infty P(k \le |\xi| < k + 1) = \sum_{k = 1}^\infty \sum_{n = 1}^k P(k \le |\xi| < k + 1) = \]
    \[\sum_{k = 1}^\infty k P(k \le |\xi| < k + 1) = \sum_{k = 1}^\infty \E(kI\{k \le |\xi| < k + 1\}) \le \sum_{k = 1}^\infty \E(|\xi|I\{k \le |\xi| < k + 1\}) = \E|\xi| \le \]
    \[\le \sum_{k = 1}^\infty \E((k + 1)I\{k \le |\xi| < k + 1\}) = 1 + \sum_{n = 1}^\infty P(|\xi| \ge n)\]
\end{proof}

\begin{lemma}[Тёплица]
    Пусть \(\{a_n, n \in \N\}\) --- неотрицательные числа, такие, что \(C_n = \sum_{j = 1}^n a_j \uparrow +\infty\) и \(C_n > 0 \forall n\). Если \(x_n \ra x\), то \(\frac{1}{C_n}\sum_{j = 1}^n a_jx_j \ra x\), \(n \ra \infty\)
\end{lemma}
\begin{proof}
    Пусть \(\epsilon > 0\). Выберем \(n_0\) так, что \(\forall n > n_0: |x_n - x| \le \frac{\epsilon}{2}\). Выберем \(n_1\) так, что \(\frac{1}{C_n}\sum_{j = 1}^{n_0}a_j|x_j - x| \le \frac{\epsilon}{2}\). Тогда \(\forall n \ge n_1\):
    \[\left| \frac{1}{C_n}\sum_{j = 1}^n a_jx_j - x \right| = \left| \frac{1}{C_n}\sum_{j = 1}^n a_j(x_j - x) \right| \le \frac{1}{C_n}\sum_{j = 1}^{n}a_j|x_j - x| =\]
    \[= \underbrace{\frac{1}{C_n}\sum_{j = 1}^{n_0}a_j|x_j - x|}_{\le \frac{\epsilon}{2}} + \underbrace{\frac{1}{C_n}\sum_{j = n_0 + 1}^{n}a_j}_{\le 1}\underbrace{|x_j - x|}_{\le \frac{\epsilon}{2}}\]
\end{proof}

\begin{theorem}[УЗБЧ в форме Этемади]
    Пусть \(\{\xi_n, n \in \N\}\) --- независимые одинаково распрееленные случайные величины, причем \(\E\xi_1 = \E\xi_n\) конечно. Положим \(S_n = \xi_1 + \dots + \xi_n\).:
    \[\frac{S_n}{n} \xrightarrow{\text{п.н.}} \E\xi_1, n \ra \infty\]
\end{theorem}
\begin{proof}
    Без ограничения общности, считаем, что \(\xi_n\) --- неотрицательные случайные величины, иначе рассмотрим \(\xi_n = \xi_n^+ - \xi_n^-\). Тогда \(\{\xi_n^{\pm}, n \in \N\}\) --- тоже независимые одинаково распрееленные случайные величины с конечным \(\E\).

    Рассмотрим \(\tilde{\xi_n} = \xi_nI\{\xi_n < n\}\). Тогда \(\{\tilde{\xi_n}, n \in \N\}\) --- попарно независимые случаные величины. Обозначим \(\tilde{S_n} = \tilde{\xi_1} + \dots + \xi_n^{\pm}\). Заметим, что:
    \[\sum_{n = 1}^\infty P(\xi_n \ne \tilde{\xi_n}) = \sum_{n = 1}^\infty P(\xi_n \ge n) = \sum_{n = 1}P(\xi_1 \ge n) < \infty\]
    Последнее получаем в силу \(\E\xi_1 < +\infty\). По лемме Бореля-Кантелли, получаем, что \(P(\{\xi_n \ne \tilde{\xi_n}, \text{б.ч.}\}) = 0\). Тогда \(\frac{S_n}{n}\xrightarrow{\text{п.н.}} \E\xi_1 \Lra \frac{\tilde{S_n}}{n}\xrightarrow{\text{п.н.}} \E\xi_1\)

    Из одинаковой распределенности \(\xi_i\), получаем:
    \[= \frac{c}{\epsilon^2}\sum_{j = 1}^\infty \frac{1}{j^2} \E(\xi_1^2 I\{\xi_1 < j\}) = \frac{c}{\epsilon^2}\sum_{j = 1}^\infty \frac{1}{j^2} \sum_{i = 1}^j \E(\xi_1^2 I\{i - 1 \le \xi_1 < i\}) =\]

    Таким образом, получили, что
    \[\frac{\xi_1^{\pm} + \dots + \xi_n^{\pm}}{n} \xrightarrow{\text{п.н.}}\E\xi_1^{\pm}\]
    Вычтем одну сходимость из другой и получим желаемое.
\end{proof}

\section{Слабая сходимость вероятностных мер}
Пусть \((S, \rho)\) --- метрическое пространство.
\begin{definition}
    Последовательность вероятностных мер \(\{P_n, n \in \N\}\) на \((S, \rho)\) слабо сходится к вероятностной мере \(P\), если \(\forall f: S \ra \R\) --- ограниченной непрерывной функции выполнено:
    \[\int_Sf(x)P_n(dx) \ra_{n \ra \infty} \int_Sf(x)P(dx)\]
    Обозначают \(P_n \xrightarrow{w} P\)
\end{definition}

\begin{note}
    Если \(\{\xi_n, n \in \N\}\) --- случайные величины, то \(\xi_n \xrightarrow{d} \xi \Lra P_{\xi_n} \xrightarrow{w} P_{\xi}\)
\end{note}

\begin{definition}
    Последовательность вероятностных мер \(\{P_n, n \in \N\}\) сходится в основном к вероятностной мере \(P\), если \(\forall B \in \mathcal{B}(S)\) с условием \(P(\partial B) = 0\) выполнено \(P_n(B) \ra_{n \ra \infty} P(B)\). Обозначение: \(P_n \Ra P\) (\(\partial B\) --- граница \(B\)).
\end{definition}

\begin{theorem}[Алексанрова]
    Следующие условия эквивалентны:
    \begin{enumerate}
        \item \(P_n \xrightarrow{w} P\)
        \item \(\limsup_{n} P_n(F) \le P(F)\) для любого замкнутого \(F \subset S\).
        \item \(\liminf_{n} P_n(G) \ge P(G)\) для любого открытого \(G \subset S\).
        \item \(P_n \Ra P\)
    \end{enumerate}
\end{theorem}
\begin{proof}
    \begin{enumerate}
        \item[\((1) \Ra (2)\)] Пусть \(F\) --- замкнуто. Для \(\delta > 0\) введем \(F^{\epsilon} = \{x: \rho(x, F) \le \epsilon\}\) --- \(\epsilon\)-окрестность \(F\). Для \(\epsilon > 0\) рассмотрим: \(f_\epsilon(x) = \left( 1 - \frac{\rho(x, F)}{\epsilon} \right)^+\). Это ограниченная непрерывная функция и \(I_F(x) \le f_\epsilon(x) \le I_{F^\epsilon}(x)\). Тогда
        \[\limsup_{n}P_n(F) = \limsup_{n}\int_SI_F(x)P_n(dx) \le \limsup_n \int_S f_\epsilon(x)P_n(dx) = \int_S f_\epsilon(x)P(dx) \le \int_S I_{F^\epsilon}(x)P(dx) = P(F^{\epsilon})\]
        При \(\epsilon \downarrow 0+\), \(F^\epsilon \downarrow F\) в силу замкнутости \(F\). В силе непрерывности вероятностной меры, \(P(F^{\epsilon}) \ra P(F)\).
        \item[\((2) \Lra (3)\)] Достаточно рассмотреть дополнения
        \item[\((2), (3) \Ra (4)\)] Пусть \(B \in \mathcal{B}(S)\) такое, что \(P(\partial B) = 0\). Рассмотрим \(F = B \cup \partial B, G = B \setminus \partial B\) --- замыкание и дополнение \(B\) соответственно. Тогда \(\limsup_n P_n(B) \le \limsup_n P_n(F) \le P(F) = P(B)\). Аналогично, \(\liminf_n P_n(B) \ge \liminf_n P_n(G) \ge P(F) = P(B) \Ra \exists \lim_n P_n(B) = P(B)\).
        \item[\((4) \Ra (1)\)] Пусть \(f: S \ra \R\) --- ограниченная непрерывная функция. Пусть \(M > 0\) таково, что \(\sup_{x}|f(x)| < M\). Введем \(D = \{t \in [-M, M]: P(\{x: f(x) = t\}) > 0\}\). Тогда \(D\) --- это не более чем счетное множество. Рассмотрим любое разбиение \(T = \{-M = t_0 < t_1 \dots < t_{m - 1} < t_m = M\}\), т.ч. все \(t_i \notin D\). Введем \(B_i = f^{-1}(t_{i - 1}, t_i] = \{x: f(x) \in (t_{i - 1}, t_i]\}\). Тогда \(\partial B_i \subset f^{-1}(\{t_{i - 1}\}) \cup f^{-1}(\{t_{i}\}) \Ra P(\partial B_i) = 0\). В силу \(P_n \Ra P\), верно \(P_n(B_i) \ra P(B_i)\). В итоге:
        \[\limsup_n \left| \int_S f(x)P_n(dx) - \int_S f(x)P(dx) \right| \le\]\[\le \limsup_n \Bigg( \left| \int_S f(x)P_n(dx) - \sum_{i = 1}^n t_{i - 1}P_n(B_i) \right| + \left| \sum_{i = 1}^n t_{i - 1}P_n(B_i) - \sum_{i = 1}^n t_{i - 1}P(B_i) \right| +\]
        \[+ \left| \sum_{i = 1}^n t_{i - 1}P(B_i) - \int_S f(x)P(dx) \right|\Bigg) \le 2\Delta T\]
        В силу произвольного выбора разбиения \(T\) получаем, что \(P_n \xrightarrow{w} P\).
    \end{enumerate}
\end{proof}

Пусть \(\{P_\alpha, \alpha \in \mathfrak{A}\}\) --- семейство вероятностных мер на \((\R^m, \mathcal{B}(\R^m))\)

\begin{definition}
    Семество \(\{P_\alpha, \alpha \in \mathfrak{A}\}\) называется относительно компактным, если \(\forall\) последовательности \(\{P_{\alpha_n}, n \in \N\} \subset \{P_\alpha, \alpha \in \mathfrak{A}\}\) можно извлечь слабо сходящуюся подпоследовательность.
\end{definition}

\begin{definition}
    Семество \(\{P_\alpha, \alpha \in \mathfrak{A}\}\) называется плотным, если \(\forall \epsilon > 0 \exists \) компакт \(K_\epsilon \subset \R^m\), такой, что:
    \[\sup_{\alpha \in \mathfrak{A}}P_\alpha(\R^m \setminus K_\epsilon) \le \epsilon\]
\end{definition}

\begin{theorem}[Прохорова]
    В \((\R^m, \mathcal{B}(\R^m))\) семество вероятностных мер является относительно компактным \(\Lra\) оно является плотным
\end{theorem}
\begin{proof}
    Докажем для \(m = 1\).
    \begin{enumerate}
        \item[\(\Ra\)] пусть семество \(\{P_\alpha, \alpha \in \mathfrak{A}\}\) относительно компактное, но не плотное.         \[\epsilon \le \limsup_n P_{\alpha_{n_k}}(\R\setminus(-n, n)) \le Q(\R\setminus(-n, n))\]
        Но \(Q(\R\setminus(-n, n)) \ra 0, n \ra \infty\) из непрерывности меры, получили противоречие.

        \item[\(\La\)] Пусть \(\{P_n, n \in \N\}\) --- плотная последовательность вероятностных мер. Пусть \(\{F_n, n \in \N\}\) --- соответствующие им функции распределения. Далее, \(\exists \)последовательность \(n^{(2)} \subset n^{(1)}\), такая, что \(\{F_{n^{(2)}_i}(q_2), n \in \N\}\) сходится. Аналогично строим подпоследовательность \(n^{(j)} \subset n^{(j-1)}\), такую, что \(\{F_{n^{(1)}_i}(q_s), n \in \N\}\) сходится \(\forall s \le j\).             \item Неубывание \(F\) очевидно из определения \(F\)
            \item Пусть \(x_l \downarrow x\). Пусть \(d = \lim_{l} F(x_l) > F(x)\). Тогда \(\exists y \in \Q, y > x: F(x) \le G(y) < d\).             \item Пусть \(x_0\) --- точка непрерывности \(F\). Тогда \(\forall y > x_0, y \in \Q: \limsup_n F_{n_k^{(k)}}(y) = G(y)\). Берем \(\inf\) в правой части и получаем: \(\limsup_k F_{n_k^{(k)}} \le F(x_0)\). Пусть \(x < x_0\), выберем \(y \in \Q\) так, что \(x < y < x_0\). Тогда:
            \[F(x) \le G(y) = \liminf_k F_{n_k^{(k)}}(y) \le \liminf_{k}F_{n_k^{(k)}}(x_0)\]
            Устремляя \(x \ra x_0\), получаем, что:
            \[F(x_0) \le \liminf_k F_{n_k^{(k)}}(x_0) \Ra \exists \lim_{k} F_{n_k^{(k)}}(x_0) = F(x_0)\]
            \item Последовательность \(\{F_n, n \in \N\}\) --- плотна \(\Ra \forall \epsilon > 0 \exists [-N, N]: \forall n \in \N: F_n(N) - F_n(-N) \ge 1 - \epsilon\). Выберем \(b > N, a < -N\) так, что \(a, b\) --- точки непрерывности \(F\). Тогда:
            \[F(b) - F(a) = \lim_k\left( F_{n_k^{(k)}}(b) - F_{n_k^{(k)}}(a) \right) \ge 1 - \epsilon\]
            Тогда \(\lim_{x \ra +\infty} F(x) = 1, \lim_{x \ra -\infty} F(x) = 0\)
        \end{itemize}
    \end{enumerate}
\end{proof}

\begin{corollary}
    Пусть \(\{P_n, n \in \N\}\) --- плотная последовательность вероятностных мер на \((\R^m, \mathcal{B}(\R^m))\). Пусть все слабо сходящиеся подпоследовательности \(\{P_n, n \in \N\}\) слабо сходятся к одной и той же мере \(Q\). Тогда \(P_n \xrightarrow{w} Q\).
\end{corollary}
\begin{proof}
    Пусть \(P_n \not\xrightarrow{w} Q\). Тогда \(\exists f: \R^m \ra \R\) --- ограниченная непрерывная функция, \(\exists \epsilon > 0\) и подпоследовательность \(\{n_k, k \in \N\}\), такая, что \(\forall k \in \N\):
    \[\left| \int_{R^m}f(x)P_n(dx) - \int_{R^m} f(x)Q(dx) \right| \ge \epsilon\]
    В силу теоремы Прохорова, \(\exists\) подпоследовательность \(\{P_{n_{k_s}}, s \in \N\} \subset \{P_{n_k}, k \in \N\}\), которая слабо сходится к \(Q\) (по условию). Получили противоречие \(\Ra P_n \Ra Q\).
\end{proof}

\section{Характеристические функции}

\begin{definition}
    Пусть \(\xi\) --- случайная величина. Ее характеристической функцией называется
    \[\phi(t) = \E e^{it\xi}\]
\end{definition}

\begin{note}
    Формально, можно понимать:
    \[\E e^{it\xi} = \E\cos\xi t + i\E\sin\xi t\]
\end{note}

\begin{definition}
    Пусть \(\xi = (\xi_1, \dots \xi_n)\) --- случайный вектор. Тогда его характеристической функцией называется
    \[\phi(t) = \E e^{i \langle t, \xi \rangle}\]
    Где \(t \in \R^n, \langle t, \xi \rangle\) --- скалярное произведение.
\end{definition}

\begin{definition}
    Если \(P\) --- вероятностная мера на \((\R^m, \mathcal{B}(\R^m))\), то ее характеристической функцией называется
    \[\phi(t) = \int_{\R^m} e^{i \langle t, x \rangle} P(dx)\]
\end{definition}

\begin{note}
    Характеристические функции для \(\xi\) и \(P_\xi\) совпадают.
\end{note}

\subsection{Примеры}
\begin{example}
    Пусть \(\xi \sim Bin(n, p)\). Ее характеристическая функция:
    \[\phi_\xi(t) = \E e^{it\xi} = \sum_{k = 0}^n e^{itk}P(\xi = k) = \sum_{k = 0}^n e^{itk}C_n^kp^k(1 - p)^{n - k} = (e^{it}p + 1 - p)^n\]
\end{example}

\begin{example}
    Пусть \(\xi \sim Exp(\lambda)\). Ее характеристическая функция:
    \[\phi_\xi(t) = \E e^{it\xi} = \int_0^{+\infty} e^{itx}\lambda e^{-\lambda x}dx = \frac{\lambda}{it - \lambda}e^{(it-\lambda)x}\big|_0^{+\infty} = \frac{\lambda}{\lambda - it}\]
\end{example}

\begin{example}
    Пусть \(\xi \sim \mathcal{N}(0, 1)\). Ее характеристическая функция:
    \[\phi_\xi(t) = \E e^{it\xi} = \int_\R e^{itx}\frac{1}{\sqrt{2\pi}}e^{-\frac{x^2}{2}}dx = \int_\R \cos tx \frac{1}{\sqrt{2\pi}}e^{-\frac{x^2}{2}}dx\]
    Тогда:
    \[\phi_\xi'(t) = \int_\R \sin tx \frac{1}{\sqrt{2\pi}}(-x)e^{-\frac{x^2}{2}}dx = \underbrace{\frac{1}{\sqrt{2\pi}}\sin tx e^{-\frac{x^2}{2}}\big|_{-\infty}^{+\infty}}_0 - t\underbrace{\int_\R\frac{1}{\sqrt{2\pi}}\cos tx e^{-\frac{x^2}{2}}dx}_{\phi_\xi(t)}\]
    Таким образом:
    \[\phi_\xi'(t) = -t \phi_\xi(t)\]
    \[(\ln\phi_\xi(t))' = -t\]
    И тогда \(\phi_\xi(t) = e^{-\frac{t^2}{2} + C}\). При этом \(\phi_\xi(0) = 1 \Ra C = 0 \Ra \phi_\xi(t) = e^{-\frac{t^2}{2}}\)
\end{example}

\begin{lemma}[Свойства характеристической функции]
    Пусть \(\phi(t)\) --- характеристическая функция случайной величины \(\xi\). Тогда:
    \begin{enumerate}
        \item \(|\phi(t)| \le 1 = \phi(0)\)
        \item Если \(\eta = a\xi + b, a, b \in \R\), то \(\phi_\eta(t) = e^{itb}\phi(at)\)
        \item \(\phi(t)\) равномерно непрерывна на \(\R\)
        \item \(\phi(t) = \overline{\phi(-t)}\)
        \item Единственность: \(\phi_\xi(t) = \phi_\eta(t) \Lra \xi \stackrel{d}{=} \eta\)
        \item \(\phi(t)\) действительнозначная \(\Lra\) распределение \(\xi\) симметрично, т.е. \(\xi \stackrel{d}{=} -\xi\)
        \item Если \(\E|\xi|^n < +\infty\), то для \(\forall r \le n\):
        \[\phi^{(r)}(t) = \E(i\xi)^re^{it\xi}\]
        \item Если \(\xi_1, \dots \xi_n\) --- независимые случаные величины, то 
        \[\phi_{\xi_1 + \dots + \xi_n} = \prod_{k = 1}^n \phi_{\xi_k}(t)\]
    \end{enumerate}
\end{lemma}
\begin{proof}
    \begin{enumerate}
        \item \(|\phi(t)| = |\E e^{it\xi}| \le \E\underbrace{|e^{it\xi}|}_1 = 1 = \phi(0)\)
        \item \(\phi_\eta(t) = \E e^{it\eta} = \E e^{it(a\xi + b)} = e^{itb}\E e^{iat\xi} = e^{itb}\phi(at)\)
        \item
        \[|\phi(t + h) - \phi(t)| = |\E e^{i(t + h)\xi} - \E e^{it\xi}| = |\E e^{it\xi}(e^{ih\xi} - 1)| \le \E |e^{it\xi}(e^{ih\xi} - 1)| = \E |e^{ih\xi} - 1|\]
        Заметим, что \(|e^{ih\xi} - 1| \xrightarrow{\text{п.н.}}_{h \ra 0} 0\) и \(|e^{ih\xi} - 1| \le 2\). По теореме Лебега о мажорируемой сходимости, получаем, что 
        \[\E|e^{ih\xi} - 1| \ra 0, h \ra 0\]
        \item
        \[\phi(t) = \E e^{it\xi} = \E\cos t\xi + i\E\sin t\xi = \overline{\E\cos t\xi - i\E\sin t\xi} = \overline{\phi(-t)}\]
        \item см. формулу обращения
        \item \(\Ra \phi_\xi(t) = \overline{\phi_\xi(t)} \stackrel{4)}{=} \phi_\xi(-t) \stackrel{2)}{=} \phi_{-\xi}(t)\). В силу единственности получаем, что \(\xi \stackrel{d}{=} -\xi\).
        
        \(\La \phi_{\xi}(t) \stackrel{5)}{=} \phi_{-\xi}(t) \stackrel{2)}{=} \phi_{\xi}(-t) \stackrel{4)}{=} \overline{\phi_\xi(t)} \Ra \phi_\xi(t) \in \R\)

        \item Проверим для \(r = 1\).
        \[\frac{\phi(t + h) - \phi(t)}{h} = \frac{\E e^{it\xi}(e^{ih\xi} - 1)}{h} = \E e^{it\xi}\left( \frac{e^{ih\xi} - 1}{h} \right)\]
        Заметим, что \(\frac{e^{ih\xi} - 1}{h} \xrightarrow{\text{п.н.}} (i\xi)\) при \(h \ra 0\) и 
        \[\left| e^{it\xi}\left( \frac{e^{ih\xi} - 1}{h} \right) \right| \le 2|\xi|\]
        По теореме Лебега о мажорируемой сходимости:
        \[\frac{\phi(t + h) - \phi(t)}{h} \ra_{h \ra 0} \E(i\xi)e^{it\xi}\]
        Формула для \(\phi^{(r)}(t)\) для \(r \ge 2\) доказывается аналогично по индукции.
        \item
        \[\phi_{\xi_1 + \dots + \xi_n}(t) = \E e^{i(\xi_1 + \dots + \xi_n)t} = \E\prod_{j = 1}^n e^{i\xi_jt} = (*)\]
        Из независимости \(\xi_1, \dots \xi_n\) получаем:
        \[(*) = \prod_{j = 1}^n\E e^{i\xi_j t} = \prod_{j = 1}^n \phi_{\xi_j}(t)\]
    \end{enumerate}
\end{proof}

\begin{exercise}
    Проверить, что в пункте 1 дествительно \(|\E(\xi + i\eta)| \le \E|\xi + i\eta|\)
\end{exercise}

\begin{theorem}[Формула обращения]
    Пусть \(\xi\) --- случайная величина с функцией распределения \(F(x)\) и характеристической функцией \(\phi(t)\). Тогда:
    \begin{enumerate}
        \item Если \(a < b\) --- точки непрерывности \(F(x)\), то
        \[F(b) - F(a) = \lim_{n \ra \infty}\frac{1}{2\pi}\int_\R \frac{e^{-ita} - e^{-itb}}{it}\phi(t)e^{-\frac{t^2}{2n}}dt\]
        \item Если \(\int_\R |\phi(t)|dt < +\infty\), то \(\xi\) имеет плотность, причем:
        \[p(x) = \frac{1}{2\pi}\int_\R e^{-itx}\phi(t)dt\]
    \end{enumerate}
\end{theorem}
\begin{proof}
    Пусть \(\eta \sim \mathcal{N}(0, 1)\) и независима с \(\xi\). Рассмотрим \(\xi_n = \xi + \frac{\eta}{\sqrt{n}}\). Тогда \(\xi_n \xrightarrow{\text{п.н.}} \xi \Ra \xi_n \xrightarrow{d} \xi\). Далее, \(\xi\) имеет плотность \(p_n(x)\) (упражнение). Найдем характеристическую функцию \(\xi_n\):
    \[\phi_{\xi_n}(t) \stackrel{8)}{=} \phi(t)\phi_{\frac{\eta}{\sqrt{n}}} = \phi(t)e^{-\frac{t^2}{2n}} \in L_1(\R)\]
    Также:
    \[\phi_{\xi_n}(t) = \E e^{it\eta_n} = \int_\R e^{itx}p_n(x)dx\]
    Тогда по формуле обращения преобразования Фурье:
    \[p_n(x) = \frac{1}{2\pi}\int_\R e^{-itx}\phi_{\xi_n}(t)dt\]
    Теперь докажем пункты:
    \begin{enumerate}
        \item 
        \[F(b) - F(a) = \lim_{n \ra \infty}(F_{\xi_n}(b) - F_{\xi_n}(a)) = \lim_{n \ra \infty}\int_a^b p_n(x)dx \stackrel{8)}{=} \lim_{n \ra \infty}\frac{1}{2\pi}\int_a^b \left( \int_\R e^{-itx}\phi_{\xi_n}(t)dt \right)dx = (*)\]
        По теореме Фубини:
        \[(*) = \lim_{n \ra \infty}\frac{1}{2\pi}\int_\R \left( \int_a^b e^{-itx}dx \right)\phi_{\xi_n}(t)dt = \lim_{n \ra \infty}\frac{1}{2\pi}\int_\R \frac{e^{-ita} - e^{-itb}}{it}\phi(t)e^{-\frac{t^2}{2n}}dt\]
        \item Переходим к пределу по \(n\) под знаком интеграла в следующем равенстве:
        \[\lim_{n \ra \infty}\int_a^b p_n(x)dx \stackrel{8)}{=} \lim_{n \ra \infty}\frac{1}{2\pi}\int_a^b \left( \int_\R e^{-itx}\phi_{\xi_n}(t)dt \right)dx\]
        Получим, что:
        \[F(b) - F(a) = \int_a^b\underbrace{\left( \frac{1}{2\pi} \int_\R e^{-itx}\phi(t)dt \right)}_{p(x)}dx\]
        Но тогда \(p(x)\) является плотностью \(\xi\).
    \end{enumerate}
\end{proof}

\begin{example}
    Пусть \(\xi_1, \xi_2\) --- независимые случайные величины, \(\xi_j \sim \mathcal{N}(a_j, \sigma_j^2), j = 1, 2\). Требуется найти распределение \(\xi_1 + \xi_2\).
\end{example}
\begin{solution}
    Найдем характеристическую функцию \(\xi_j\). Заметим, что:
    \[\frac{\xi_j - a_j}{\sigma_j} = \eta_j \sim \mathcal{N}(0, 1)\]
    Тогда
    \[\phi_{\xi_j}(t) \stackrel{2)}{=} \phi_{\eta_j}(\sigma_jt)e^{ia_jt} = e^{ia_jt - \frac{\sigma_j^2t}{2}}\]
    В итоге:
    \[\phi_{\xi_1 + \xi_2}(t) = \phi_{\xi_1}(t)\phi_{\xi_2}(t) = e^{i(a_1 + a_2)t - \frac{1}{2}t^2(\sigma_1^2 + \sigma_2^2)}\]
    Но тогда по единственности, \(\xi_1 + \xi_2 \sim \mathcal{N}(a_1 + a_2, \sigma_1^2 + \sigma_2^2)\)
\end{solution}

\begin{note}
    Свойство единственности работает и для случайных векторов.
\end{note}

\begin{theorem}[Критерий независимости для характеристических функций]
    Случайные величины \(\xi_1, \dots \xi_n\) независимы в совокупности \(\Lra \phi_\xi(t_1, \dots t_n) = \prod_{j = 1}^n \phi_{\xi_j}(t_j)\), где \(\xi = (\xi_1, \dots \xi_n)\).
\end{theorem}
\begin{proof}
    \begin{enumerate}
        \item[\(\Ra\)]
        \[\phi_\xi(t_1, \dots t_n) = \E e^{i \langle \xi, t \rangle} = \E\prod_{j = 1}^n e^{i\xi_jt_j} = (*)\]
        Из независимости \(\xi_1, \dots \xi_n\) получаем:
        \[(*) = \prod_{j = 1}^n\E e^{i\xi_j t_j} = \prod_{j = 1}^n \phi_{\xi_j}(t_j)\]
        \item[\(\La\)] Рассмотрим случайный вектор \(\eta = (\eta_1, \dots \eta_n)\), такой, что \(\eta_j\) --- независимые случаные величины и \(\forall j = 1, \dots n: \eta_j \stackrel{d}{=} \xi_j\). Тогда:
        \[\phi_\xi(t_1, \dots t_n) = \prod_{j = 1}^n \phi_{\xi_j}(t_j) = (*)\]
        Из одинаковой распределенности получаем:
        \[(*) = \prod_{j = 1}^n \phi_{\eta_j}(t_j) \stackrel{\Ra}{=}\phi_\eta(t_1, \dots t_n)\]
        По свойству единственности получаем, что
        \[F_\xi(x_1, \dots x_n) = F_\eta(x_1, \dots x_n) = \prod_{j = 1}^n F_{\eta_j}(x_j) = (*)\]
        И из одинаковой распределенности получаем:
        \[(*)\prod_{j = 1}^n F_{\xi_j}(x_j)\]
        Но тогда \(\xi_1, \dots \xi_n\) независимы в совокупности.
    \end{enumerate}
\end{proof}
